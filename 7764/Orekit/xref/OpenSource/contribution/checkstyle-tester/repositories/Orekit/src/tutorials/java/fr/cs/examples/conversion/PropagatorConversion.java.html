55  /** Orekit tutorial for propagator conversion.
56   * @author Pascal Parraud
57   */
58  public class PropagatorConversion {
59  
60      /** Program entry point.
61       * @param args program arguments (unused here)
62       */
63      public static void main(String[] args) {
64          try {
65  
66              // configure Orekit
67              Autoconfiguration.configureOrekit();
68  
69              // gravity field
70              NormalizedSphericalHarmonicsProvider provider = GravityFieldFactory.getNormalizedProvider(2, 0);
71              double mu =  provider.getMu();
72  
73              // inertial frame
74              Frame inertialFrame = FramesFactory.getEME2000();
75  
76              // Initial date
77              AbsoluteDate initialDate = new AbsoluteDate(2004, 01, 01, 23, 30, 00.000,
78                                                          TimeScalesFactory.getUTC());
79  
80              // Initial orbit (GTO)
81              final double a     = 24396159;                // semi major axis in meters
82              final double e     = 0.72831215;              // eccentricity
83              final double i     = FastMath.toRadians(7);   // inclination
84              final double omega = FastMath.toRadians(180); // perigee argument
85              final double raan  = FastMath.toRadians(261); // right ascention of ascending node
86              final double lM    = 0;                       // mean anomaly
87              Orbit initialOrbit = new KeplerianOrbit(a, e, i, omega, raan, lM, PositionAngle.MEAN,
88                                                      inertialFrame, initialDate, mu);
89              final double period = initialOrbit.getKeplerianPeriod();
90  
91              // Initial state definition
92              final SpacecraftState initialState = new SpacecraftState(initialOrbit);
93  
94              // Adaptive step integrator with a minimum step of 0.001 and a maximum step of 1000
95              final double minStep    = 0.001;
96              final double maxStep    = 1000.;
97              final double dP         = 1.e-2;
98              final OrbitType orbType = OrbitType.CARTESIAN;
99              final double[][] tol = NumericalPropagator.tolerances(dP, initialOrbit, orbType);
100             final AbstractIntegrator integrator = new DormandPrince853Integrator(minStep, maxStep,
101                                                                                  tol[0], tol[1]);
102 
103             // Propagator
104             NumericalPropagator numProp = new NumericalPropagator(integrator);
105             numProp.setInitialState(initialState);
106             numProp.setOrbitType(orbType);
107 
108             // Force Models:
109             // 1 - Perturbing gravity field (only J2 is considered here)
110             ForceModel gravity = new HolmesFeatherstoneAttractionModel(FramesFactory.getITRF(IERSConventions.IERS_2010, true), provider);
111 
112             // Add force models to the propagator
113             numProp.addForceModel(gravity);
114             
115             // Propagator factory
116             PropagatorBuilder builder = new KeplerianPropagatorBuilder(initialOrbit, PositionAngle.TRUE, dP);
117 
118             // Propagator converter
119             PropagatorConverter fitter = new FiniteDifferencePropagatorConverter(builder, 1.e-6, 5000);
120             
121             // Resulting propagator
122             KeplerianPropagator kepProp = (KeplerianPropagator)fitter.convert(numProp, 2*period, 251);
123 
124             // Step handlers
125             StatesHandler numStepHandler = new StatesHandler();
126             StatesHandler kepStepHandler = new StatesHandler();
127 
128             // Set up operating mode for the propagator as master mode
129             // with fixed step and specialized step handler
130             numProp.setMasterMode(60., numStepHandler);
131             kepProp.setMasterMode(60., kepStepHandler);
132 
133             // Extrapolate from the initial to the final date
134             numProp.propagate(initialDate.shiftedBy(10.*period));
135             kepProp.propagate(initialDate.shiftedBy(10.*period));
136 
137             // retrieve the states
138             List<SpacecraftState> numStates = numStepHandler.getStates();
139             List<SpacecraftState> kepStates = kepStepHandler.getStates();
140 
141             // Print the results on the output file
142             File output = new File(new File(System.getProperty("user.home")), "elements.dat");
143             try (final PrintStream stream = new PrintStream(output, "UTF-8")) {
144                 stream.println("# date Anum Akep Enum Ekep Inum Ikep LMnum LMkep");
145                 for (SpacecraftState numState : numStates) {
146                     for (SpacecraftState kepState : kepStates) {
147                         if (numState.getDate().compareTo(kepState.getDate()) == 0) {
148                             stream.println(numState.getDate()
149                                            + " " + numState.getA()
150                                            + " " + kepState.getA()
151                                            + " " + numState.getE()
152                                            + " " + kepState.getE()
153                                            + " " + FastMath.toDegrees(numState.getI())
154                                            + " " + FastMath.toDegrees(kepState.getI())
155                                            + " " + FastMath.toDegrees(MathUtils.normalizeAngle(numState.getLM(), FastMath.PI))
156                                            + " " + FastMath.toDegrees(MathUtils.normalizeAngle(kepState.getLM(), FastMath.PI)));
157                             break;
158                         }
159                     }
160                 }
161             }
162             System.out.println("Results saved as file " + output);
163 
164             File output1 = new File(new File(System.getProperty("user.home")), "elts_pv.dat");
165             try (final PrintStream stream = new PrintStream(output1, "UTF-8")) {
166                 stream.println("# date pxn pyn pzn vxn vyn vzn pxk pyk pzk vxk vyk vzk");
167                 for (SpacecraftState numState : numStates) {
168                     for (SpacecraftState kepState : kepStates) {
169                         if (numState.getDate().compareTo(kepState.getDate()) == 0) {
170                             final double pxn = numState.getPVCoordinates().getPosition().getX();
171                             final double pyn = numState.getPVCoordinates().getPosition().getY();
172                             final double pzn = numState.getPVCoordinates().getPosition().getZ();
173                             final double vxn = numState.getPVCoordinates().getVelocity().getX();
174                             final double vyn = numState.getPVCoordinates().getVelocity().getY();
175                             final double vzn = numState.getPVCoordinates().getVelocity().getZ();
176                             final double pxk = kepState.getPVCoordinates().getPosition().getX();
177                             final double pyk = kepState.getPVCoordinates().getPosition().getY();
178                             final double pzk = kepState.getPVCoordinates().getPosition().getZ();
179                             final double vxk = kepState.getPVCoordinates().getVelocity().getX();
180                             final double vyk = kepState.getPVCoordinates().getVelocity().getY();
181                             final double vzk = kepState.getPVCoordinates().getVelocity().getZ();
182                             stream.println(numState.getDate()
183                                            + " " + pxn + " " + pyn + " " + pzn
184                                            + " " + vxn + " " + vyn + " " + vzn
185                                            + " " + pxk + " " + pyk + " " + pzk
186                                            + " " + vxk + " " + vyk + " " + vzk);
187                             break;
188                         }
189                     }
190                 }
191             }
192             System.out.println("Results saved as file " + output1);
193 
194         } catch (OrekitException oe) {
195             System.err.println(oe.getLocalizedMessage());
196             System.exit(1);
197         } catch (IOException ioe) {
198             System.err.println(ioe.getLocalizedMessage());
199             System.exit(1);
200         }
201     }
202 
203     /** Specialized step handler.
204      * <p>This class extends the step handler in order to handle states at each step.<p>
205      */
206     private static class StatesHandler implements OrekitFixedStepHandler {
207 
208         /** Points. */
209         private final List<SpacecraftState> states;
210 
211         public StatesHandler() {
212             // prepare an empty list of SpacecraftState
213             states = new ArrayList<SpacecraftState>();
214         }
215 
216         public void init(final SpacecraftState s0, final AbsoluteDate t) {
217         }
218 
219         public void handleStep(SpacecraftState currentState, boolean isLast) {
220 
221             // add the current state
222             states.add(currentState);
223 
224         }
225 
226         /** Get the list of handled orbital elements.
227          * @return orbital elements list
228          */
229         public List<SpacecraftState> getStates() {
230             return states;
231         }
232 
233     }
