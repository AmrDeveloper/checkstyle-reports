89      public static void main(String[] args) {
90          try {
91

93                  System.err.println("usage: java fr.cs.examples.bodies.Phasing filename");

95              }
96  
97              // configure Orekit
98              Autoconfiguration.configureOrekit();

100             // input/out
101             URL url = Phasing.class.getResource("/" + args[0]);
102             if (url == null) {
103                 System.err.println(args[0] + " not found");
104                 System.exit(1);
105             }
106             File input  = new File(url.toURI().getPath());
107 
108             new Phasing().run(input);

110         } catch (URISyntaxException use) {
111             System.err.println(use.getLocalizedMessage());
112             System.exit(1);
113         } catch (IOException ioe) {

115             System.exit(1);

125             System.err.println(oe.getLocalizedMessage());
126             System.exit(1);
127         }

154         IERSConventions conventions = IERSConventions.IERS_2010;

162     private void run(final File input)
163             throws IOException, IllegalArgumentException, ParseException, OrekitException {
164 
165         // read input parameters
166         KeyValueFileParser<ParameterKey> parser =
167                 new KeyValueFileParser<ParameterKey>(ParameterKey.class);
168         try (final FileInputStream fis = new FileInputStream(input)) {
169             parser.parseInput(input.getAbsolutePath(), fis);
170         }
171         TimeScale utc = TimeScalesFactory.getUTC();
172 
173        // simulation properties
174         AbsoluteDate date       = parser.getDate(ParameterKey.ORBIT_DATE, utc);
175         int          nbOrbits   = parser.getInt(ParameterKey.PHASING_ORBITS_NUMBER);
176         int          nbDays     = parser.getInt(ParameterKey.PHASING_DAYS_NUMBER);
177         double       latitude   = parser.getAngle(ParameterKey.SUN_SYNCHRONOUS_REFERENCE_LATITUDE);
178         boolean      ascending  = parser.getBoolean(ParameterKey.SUN_SYNCHRONOUS_REFERENCE_ASCENDING);
179         double       mst        = parser.getTime(ParameterKey.SUN_SYNCHRONOUS_MEAN_SOLAR_TIME).getSecondsInUTCDay() / 3600;
180         int          degree     = parser.getInt(ParameterKey.GRAVITY_FIELD_DEGREE);
181         int          order      = parser.getInt(ParameterKey.GRAVITY_FIELD_ORDER);
182         String       gridOutput = parser.getString(ParameterKey.GRID_OUTPUT);
183         double[]     gridLatitudes = new double[] {
184             parser.getAngle(ParameterKey.GRID_LATITUDE_1),
185             parser.getAngle(ParameterKey.GRID_LATITUDE_2),
186             parser.getAngle(ParameterKey.GRID_LATITUDE_3),
187             parser.getAngle(ParameterKey.GRID_LATITUDE_4),
188             parser.getAngle(ParameterKey.GRID_LATITUDE_5)
189         };
190         boolean[]    gridAscending = new boolean[] {
191             parser.getBoolean(ParameterKey.GRID_ASCENDING_1),
192             parser.getBoolean(ParameterKey.GRID_ASCENDING_2),
193             parser.getBoolean(ParameterKey.GRID_ASCENDING_3),
194             parser.getBoolean(ParameterKey.GRID_ASCENDING_4),
195             parser.getBoolean(ParameterKey.GRID_ASCENDING_5)
196         };
197 
198         gravityField = GravityFieldFactory.getNormalizedProvider(degree, order);
199 
200         // initial guess for orbit
201         CircularOrbit orbit = guessOrbit(date, FramesFactory.getEME2000(), nbOrbits, nbDays,
202                                          latitude, ascending, mst);
203         System.out.println("initial orbit: " + orbit);
204         System.out.println("please wait while orbit is adjusted...");
205         System.out.println();
206 
207         // numerical model for improving orbit
208         double[][] tolerances = NumericalPropagator.tolerances(0.1, orbit, OrbitType.CIRCULAR);
209         DormandPrince853Integrator integrator =
210                 new DormandPrince853Integrator(1.0e-4 * orbit.getKeplerianPeriod(),
211                                                1.0e-1 * orbit.getKeplerianPeriod(),
212                                                tolerances[0], tolerances[1]);
213         integrator.setInitialStepSize(1.0e-2 * orbit.getKeplerianPeriod());
214         NumericalPropagator propagator = new NumericalPropagator(integrator);
215         propagator.addForceModel(new HolmesFeatherstoneAttractionModel(FramesFactory.getGTOD(IERSConventions.IERS_2010, true),
216                                                                        gravityField));
217         propagator.addForceModel(new ThirdBodyAttraction(CelestialBodyFactory.getSun()));
218         propagator.addForceModel(new ThirdBodyAttraction(CelestialBodyFactory.getMoon()));
219 
220         double deltaP = Double.POSITIVE_INFINITY;
221         double deltaV = Double.POSITIVE_INFINITY;
222 
223         int counter = 0;
224         DecimalFormat f = new DecimalFormat("0.000E00", new DecimalFormatSymbols(Locale.US));
225         while (deltaP > 3.0e-1 || deltaV > 3.0e-4) {
226 
227             CircularOrbit previous = orbit;
228 
229             CircularOrbit tmp1 = improveEarthPhasing(previous, nbOrbits, nbDays, propagator);
230             CircularOrbit tmp2 = improveSunSynchronization(tmp1, nbOrbits * tmp1.getKeplerianPeriod(),
231                                               latitude, ascending, mst, propagator);
232             orbit = improveFrozenEccentricity(tmp2, nbOrbits * tmp2.getKeplerianPeriod(), propagator);
233             double da  = orbit.getA() - previous.getA();
234             double dex = orbit.getCircularEx() - previous.getCircularEx();
235             double dey = orbit.getCircularEy() - previous.getCircularEy();
236             double di  = FastMath.toDegrees(orbit.getI() - previous.getI());
237             double dr  = FastMath.toDegrees(orbit.getRightAscensionOfAscendingNode() -
238                                            previous.getRightAscensionOfAscendingNode());
239             System.out.println(" iteration " + (++counter) + ": deltaA = " + f.format(da) +
240                                " m, deltaEx = " + f.format(dex) + ", deltaEy = " + f.format(dey) +
241                                ", deltaI = " + f.format(di) + " deg, deltaRAAN = " + f.format(dr) +
242                                " deg");
243 
244             PVCoordinates delta = new PVCoordinates(previous.getPVCoordinates(),
245                                                     orbit.getPVCoordinates());
246             deltaP = delta.getPosition().getNorm();
247             deltaV = delta.getVelocity().getNorm();
248 
249         }
250 
251         // final orbit
252         System.out.println();
253         System.out.println("final orbit (osculating): " + orbit);
254 
255         // generate the ground track grid file
256         try (PrintStream output = new PrintStream(new File(input.getParent(), gridOutput), "UTF-8")) {
257             for (int i = 0; i < gridLatitudes.length; ++i) {
258                 printGridPoints(output, gridLatitudes[i], gridAscending[i], orbit, propagator, nbOrbits);
259             }
260         }
261 
262     }

264     /** Guess an initial orbit from theoretical model.
265      * @param date orbit date
266      * @param frame frame to use for defining orbit
267      * @param nbOrbits number of orbits in the phasing cycle
268      * @param nbDays number of days in the phasing cycle
269      * @param latitude reference latitude for Sun synchronous orbit
270      * @param ascending if true, crossing latitude is from South to North
271      * @param mst desired mean solar time at reference latitude crossing
272      * @return an initial guess of Earth phased, Sun synchronous orbit
273      * @exception OrekitException if mean solar time cannot be computed
274      */
275     private CircularOrbit guessOrbit(AbsoluteDate date, Frame frame, int nbOrbits, int nbDays,
276                                      double latitude, boolean ascending, double mst)
277         throws OrekitException {
278 
279         double mu = gravityField.getMu();
280         NormalizedSphericalHarmonicsProvider.NormalizedSphericalHarmonics harmonics =
281                 gravityField.onDate(date);
282 
283         // initial semi major axis guess based on Keplerian period
284         double period0 = (nbDays * Constants.JULIAN_DAY) / nbOrbits;
285         double n0      = 2 * FastMath.PI / period0;
286         double a0      = FastMath.cbrt(mu / (n0 * n0));
287 
288         // initial inclination guess based on ascending node drift due to J2
289         double[][] unnormalization = GravityFieldFactory.getUnnormalizationFactors(3, 0);
290         double j2       = -unnormalization[2][0] * harmonics.getNormalizedCnm(2, 0);
291         double j3       = -unnormalization[3][0] * harmonics.getNormalizedCnm(3, 0);
292         double raanRate = 2 * FastMath.PI / Constants.JULIAN_YEAR;
293         double ae       = gravityField.getAe();
294         double i0       = FastMath.acos(-raanRate * a0 * a0 / (1.5 * ae * ae * j2 * n0));
295 
296         // initial eccentricity guess based on J2 and J3
297         double ex0   = 0;
298         double ey0   = -j3 * ae * FastMath.sin(i0) / (2 * a0 * j2);
299 
300         // initial ascending node guess based on mean solar time
301         double alpha0 = FastMath.asin(FastMath.sin(latitude) / FastMath.sin(i0));
302         if (!ascending) {
303             alpha0 = FastMath.PI - alpha0;
304         }
305         double h = meanSolarTime(new CircularOrbit(a0, ex0, ey0, i0, 0.0, alpha0,
306                                                    PositionAngle.TRUE, frame, date, mu));
307         double raan0 = FastMath.PI * (mst - h) / 12.0;
308 
309         return new CircularOrbit(a0, ex0, ey0, i0, raan0, alpha0,
310                                  PositionAngle.TRUE, frame, date, mu);
311 
312     }

314     /** Improve orbit to better match Earth phasing parameters.
315      * @param previous previous orbit
316      * @param nbOrbits number of orbits in the phasing cycle
317      * @param nbDays number of days in the phasing cycle
318      * @param propagator propagator to use
319      * @return an improved Earth phased orbit
320      * @exception OrekitException if orbit cannot be propagated
321      */
322     private CircularOrbit improveEarthPhasing(CircularOrbit previous, int nbOrbits, int nbDays,
323                                               Propagator propagator)
324         throws OrekitException {
325 
326         propagator.resetInitialState(new SpacecraftState(previous));
327 
328         // find first ascending node
329         double period = previous.getKeplerianPeriod();
330         SpacecraftState firstState = findFirstCrossing(0.0, true, previous.getDate(),
331                                                        previous.getDate().shiftedBy(2 * period),
332                                                        0.01 * period, propagator);
333 
334         // go to next cycle, one orbit at a time
335         SpacecraftState state = firstState;
336         for (int i = 0; i < nbOrbits; ++i) {
337             final AbsoluteDate previousDate = state.getDate();
338             state = findLatitudeCrossing(0.0, previousDate.shiftedBy(period),
339                                          previousDate.shiftedBy(2 * period),
340                                          0.01 * period, period, propagator);
341             period = state.getDate().durationFrom(previousDate);
342         }
343 
344         double cycleDuration = state.getDate().durationFrom(firstState.getDate());
345         double deltaT;
346         if (((int) FastMath.rint(cycleDuration / Constants.JULIAN_DAY)) != nbDays) {
347             // we are very far from expected duration
348             deltaT = nbDays * Constants.JULIAN_DAY - cycleDuration;
349         } else {
350             // we are close to expected duration
351             GeodeticPoint startPoint = earth.transform(firstState.getPVCoordinates().getPosition(),
352                                                        firstState.getFrame(), firstState.getDate());
353             GeodeticPoint endPoint   = earth.transform(state.getPVCoordinates().getPosition(),
354                                                        state.getFrame(), state.getDate());
355             double deltaL =
356                     MathUtils.normalizeAngle(endPoint.getLongitude() - startPoint.getLongitude(), 0.0);
357             deltaT = deltaL * Constants.JULIAN_DAY / (2 * FastMath.PI);
358         }
359 
360         double deltaA = 2 * previous.getA() * deltaT / (3 * nbOrbits * previous.getKeplerianPeriod());
361         return new CircularOrbit(previous.getA() + deltaA,
362                                  previous.getCircularEx(), previous.getCircularEy(),
363                                  previous.getI(), previous.getRightAscensionOfAscendingNode(),
364                                  previous.getAlphaV(), PositionAngle.TRUE,
365                                  previous.getFrame(), previous.getDate(),
366                                  previous.getMu());
367 
368     }

370     /** Improve orbit to better match phasing parameters.
371      * @param previous previous orbit
372      * @param duration sampling duration
373      * @param latitude reference latitude for Sun synchronous orbit
374      * @param ascending if true, crossing latitude is from South to North
375      * @param mst desired mean solar time at reference latitude crossing
376      * @param propagator propagator to use
377      * @return an improved Earth phased, Sun synchronous orbit
378      * @exception OrekitException if orbit cannot be propagated
379      */
380     private CircularOrbit improveSunSynchronization(CircularOrbit previous, double duration,
381                                                     double latitude, boolean ascending, double mst,
382                                                     Propagator propagator)
383         throws OrekitException {
384 
385         propagator.resetInitialState(new SpacecraftState(previous));
386         AbsoluteDate start = previous.getDate();
387 
388         // find the first latitude crossing
389         double period   = previous.getKeplerianPeriod();
390         double stepSize = period / 100;
391         SpacecraftState crossing =
392                 findFirstCrossing(latitude, ascending, start, start.shiftedBy(2 * period),
393                                   stepSize, propagator);
394 
395         // find all other latitude crossings from regular schedule
396         SecularAndHarmonic mstModel = new SecularAndHarmonic(2,
397                                                              2.0 * FastMath.PI / Constants.JULIAN_YEAR,
398                                                              4.0 * FastMath.PI / Constants.JULIAN_YEAR,
399                                                              2.0 * FastMath.PI / Constants.JULIAN_DAY,
400                                                              4.0 * FastMath.PI / Constants.JULIAN_DAY);
401         mstModel.resetFitting(start, new double[] {
402             mst, -1.0e-10, -1.0e-17,
403             1.0e-3, 1.0e-3, 1.0e-5, 1.0e-5, 1.0e-5, 1.0e-5, 1.0e-5, 1.0e-5
404         });
405         while (crossing != null && crossing.getDate().durationFrom(start) < duration) {
406             final AbsoluteDate previousDate = crossing.getDate();
407             crossing = findLatitudeCrossing(latitude, previousDate.shiftedBy(period),
408                                             previousDate.shiftedBy(2 * period),
409                                             stepSize, period / 8, propagator);
410             if (crossing != null) {
411 
412                 // store current point
413                 mstModel.addPoint(crossing.getDate(), meanSolarTime(crossing.getOrbit()));
414 
415                 // use the same time separation to pinpoint next crossing
416                 period = crossing.getDate().durationFrom(previousDate);
417 
418             }
419 
420         }
421 
422         // fit the mean solar time to a parabolic plus medium periods model
423         // we will only use the linear part for the correction
424         mstModel.fit();
425         final double[] fittedH = mstModel.approximateAsPolynomialOnly(1, start, 2, 2,
426                                                                       start, start.shiftedBy(duration),
427                                                                       stepSize);
428 
429         // solar time bias must be compensated by shifting ascending node
430         double deltaRaan = FastMath.PI * (mst - fittedH[0]) / 12;
431 
432         // solar time slope must be compensated by changing inclination
433         // linearized relationship between hDot and inclination:
434         // hDot = alphaDot - raanDot where alphaDot is the angular rate of Sun right ascension
435         // and raanDot is the angular rate of ascending node right ascension. So hDot evolution
436         // is the opposite of raan evolution, which itself is proportional to cos(i) due to J2
437         // effect. So hDot = alphaDot - k cos(i) and hence Delta hDot = -k sin(i) Delta i
438         // so Delta hDot / Delta i = (alphaDot - hDot) tan(i)
439         double dhDotDi = (24.0 / Constants.JULIAN_YEAR - fittedH[1]) * FastMath.tan(previous.getI());
440 
441         // compute inclination offset needed to achieve station-keeping target
442         final double deltaI = fittedH[1] / dhDotDi;
443 
444         return new CircularOrbit(previous.getA(),
445                                  previous.getCircularEx(), previous.getCircularEy(),
446                                  previous.getI() + deltaI,
447                                  previous.getRightAscensionOfAscendingNode() + deltaRaan,
448                                  previous.getAlphaV(), PositionAngle.TRUE,
449                                  previous.getFrame(), previous.getDate(),
450                                  previous.getMu());
451 
452     }

454     /** Improve orbit to better match frozen eccentricity property.
455      * @param previous previous orbit
456      * @param duration sampling duration
457      * @param propagator propagator to use
458      * @return an improved Earth phased, Sun synchronous orbit with frozen eccentricity
459      * @exception OrekitException if orbit cannot be propagated
460      */
461     private CircularOrbit improveFrozenEccentricity(CircularOrbit previous, double duration,
462                                                     Propagator propagator)
463         throws OrekitException {
464 
465         propagator.resetInitialState(new SpacecraftState(previous));
466         AbsoluteDate start = previous.getDate();
467 
468         NormalizedSphericalHarmonicsProvider.NormalizedSphericalHarmonics harmonics =
469                 gravityField.onDate(previous.getDate());
470         double[][] unnormalization = GravityFieldFactory.getUnnormalizationFactors(2, 0);
471         double a    = previous.getA();
472         double sinI = FastMath.sin(previous.getI());
473         double aeOa = gravityField.getAe() / a;
474         double mu   = gravityField.getMu();
475         double n    = FastMath.sqrt(mu / a) / a;
476         double j2   = -unnormalization[2][0] * harmonics.getNormalizedCnm(2, 0);
477         double frozenPulsation = 3 * n * j2 * aeOa * aeOa * (1 - 1.25 * sinI * sinI);
478 
479         // fit the eccentricity to an harmonic model with short and medium periods
480         // we will only use the medium periods part for the correction
481         SecularAndHarmonic exModel = new SecularAndHarmonic(0, frozenPulsation, n, 2 * n);
482         SecularAndHarmonic eyModel = new SecularAndHarmonic(0, frozenPulsation, n, 2 * n);
483         exModel.resetFitting(start, new double[] {
484             previous.getCircularEx(), -1.0e-10, 1.0e-5,
485             1.0e-5, 1.0e-5, 1.0e-5, 1.0e-5
486         });
487         eyModel.resetFitting(start, new double[] {
488             previous.getCircularEy(), -1.0e-10, 1.0e-5,
489             1.0e-5, 1.0e-5, 1.0e-5, 1.0e-5
490         });
491 
492         final double step = previous.getKeplerianPeriod() / 5;
493         for (double dt = 0; dt < duration; dt += step) {
494             final SpacecraftState state = propagator.propagate(start.shiftedBy(dt));
495             final CircularOrbit orbit   = (CircularOrbit) OrbitType.CIRCULAR.convertType(state.getOrbit());
496             exModel.addPoint(state.getDate(), orbit.getCircularEx());
497             eyModel.addPoint(state.getDate(), orbit.getCircularEy());
498         }
499 
500         // adjust eccentricity
501         exModel.fit();
502         double dex = -exModel.getFittedParameters()[1];
503         eyModel.fit();
504         double dey = -eyModel.getFittedParameters()[1];
505 
506         // put the eccentricity at center of frozen center
507         return new CircularOrbit(previous.getA(),
508                                  previous.getCircularEx() + dex,
509                                  previous.getCircularEy() + dey,
510                                  previous.getI(), previous.getRightAscensionOfAscendingNode(),
511                                  previous.getAlphaV(), PositionAngle.TRUE,
512                                  previous.getFrame(), previous.getDate(),
513                                  previous.getMu());
514 
515     }

517     /** Print ground track grid point
518      * @param out output stream
519      * @param latitude point latitude
520      * @param ascending indicator for latitude crossing direction
521      * @param orbit phased orbit
522      * @param propagator propagator for orbit
523      * @param nbOrbits number of orbits in the cycle
524      * @exception OrekitException if orbit cannot be propagated
525      */
526     private void printGridPoints(final PrintStream out,
527                                  final double latitude, final boolean ascending,
528                                  final Orbit orbit, final Propagator propagator, int nbOrbits)
529         throws OrekitException {
530 
531         propagator.resetInitialState(new SpacecraftState(orbit));
532         AbsoluteDate start = orbit.getDate();
533 
534         // find the first latitude crossing
535         double period   = orbit.getKeplerianPeriod();
536         double stepSize = period / 100;
537         SpacecraftState crossing =
538                 findFirstCrossing(latitude, ascending, start, start.shiftedBy(2 * period),
539                                   stepSize, propagator);
540 
541         // find all other latitude crossings from regular schedule
542         DecimalFormat fTime  = new DecimalFormat("0000000.000", new DecimalFormatSymbols(Locale.US));
543         DecimalFormat fAngle = new DecimalFormat("000.00000",   new DecimalFormatSymbols(Locale.US));
544         while (nbOrbits-- > 0) {
545 
546             GeodeticPoint gp = earth.transform(crossing.getPVCoordinates().getPosition(),
547                                                crossing.getFrame(), crossing.getDate());
548             out.println(fTime.format(crossing.getDate().durationFrom(start)) +
549                         " " + fAngle.format(FastMath.toDegrees(gp.getLatitude())) +
550                         " " + fAngle.format(FastMath.toDegrees(gp.getLongitude())) +
551                         " " + ascending);
552 
553             final AbsoluteDate previousDate = crossing.getDate();
554             crossing = findLatitudeCrossing(latitude, previousDate.shiftedBy(period),
555                                             previousDate.shiftedBy(2 * period),
556                                             stepSize, period / 8, propagator);
557             period = crossing.getDate().durationFrom(previousDate);
558 
559         }
560 
561     }

563     /** Compute the mean solar time.
564      * @param orbit current orbit
565      * @return mean solar time
566      * @exception OrekitException if state cannot be converted
567      */
568     private double meanSolarTime(final Orbit orbit)
569         throws OrekitException {
570 
571         // compute angle between Sun and spacecraft in the equatorial plane
572         final Vector3D position = orbit.getPVCoordinates().getPosition();
573         final double time       = orbit.getDate().getComponents(TimeScalesFactory.getUTC()).getTime().getSecondsInUTCDay();
574         final double theta      = gmst.value(orbit.getDate()).getValue();
575         final double sunAlpha   = theta + FastMath.PI * (1 - time / (Constants.JULIAN_DAY * 0.5));
576         final double dAlpha     = MathUtils.normalizeAngle(position.getAlpha() - sunAlpha, 0);
577 
578         // convert the angle to solar time
579         return 12.0 * (1.0 + dAlpha / FastMath.PI);
580 
581     }

583     /**
584      * Find the first crossing of the reference latitude.
585      * @param latitude latitude to search for
586      * @param ascending indicator for desired crossing direction
587      * @param searchStart search start
588      * @param end maximal date not to overtake
589      * @param stepSize step size to use
590      * @param propagator propagator
591      * @return first crossing
592      * @throws OrekitException if state cannot be propagated
593      */
594     private SpacecraftState findFirstCrossing(final double latitude, final boolean ascending,
595                                               final AbsoluteDate searchStart, final AbsoluteDate end,
596                                               final double stepSize, final Propagator propagator)
597         throws OrekitException {
598 
599         double previousLatitude = Double.NaN;
600         for (AbsoluteDate date = searchStart; date.compareTo(end) < 0; date = date.shiftedBy(stepSize)) {
601             final PVCoordinates pv       = propagator.propagate(date).getPVCoordinates(earth.getBodyFrame());
602             final double currentLatitude = earth.transform(pv.getPosition(), earth.getBodyFrame(), date).getLatitude();
603             if (((previousLatitude <= latitude) && (currentLatitude >= latitude) &&  ascending) ||
604                 ((previousLatitude >= latitude) && (currentLatitude <= latitude) && !ascending)) {
605                 return findLatitudeCrossing(latitude, date.shiftedBy(-0.5 * stepSize), end,
606                                             0.5 * stepSize, 2 * stepSize, propagator);
607             }
608             previousLatitude = currentLatitude;
609         }
610 
611         throw new OrekitException(LocalizedCoreFormats.SIMPLE_MESSAGE,
612                                   "latitude " + FastMath.toDegrees(latitude) + " never crossed");
613 
614     }

617     /**
618      * Find the state at which the reference latitude is crossed.
619      * @param latitude latitude to search for
620      * @param guessDate guess date for the crossing
621      * @param endDate maximal date not to overtake
622      * @param shift shift value used to evaluate the latitude function bracketing around the guess date
623      * @param maxShift maximum value that the shift value can take
624      * @param propagator propagator used
625      * @return state at latitude crossing time
626      * @throws OrekitException if state cannot be propagated
627      * @throws MathRuntimeException if latitude cannot be bracketed in the search interval
628      */
629     private SpacecraftState findLatitudeCrossing(final double latitude,
630                                                  final AbsoluteDate guessDate, final AbsoluteDate endDate,
631                                                  final double shift, final double maxShift,
632                                                  final Propagator propagator)
633         throws OrekitException, MathRuntimeException {
634 
635         // function evaluating to 0 at latitude crossings
636         final UnivariateFunction latitudeFunction = new UnivariateFunction() {
637             /** {@inheritDoc} */
638             public double value(double x) {
639                 try {
640                     final SpacecraftState state = propagator.propagate(guessDate.shiftedBy(x));
641                     final Vector3D position = state.getPVCoordinates(earth.getBodyFrame()).getPosition();
642                     final GeodeticPoint point = earth.transform(position, earth.getBodyFrame(), state.getDate());
643                     return point.getLatitude() - latitude;
644                 } catch (OrekitException oe) {
645                     throw new RuntimeException(oe);
646                 }
647             }
648         };
649 
650         // try to bracket the encounter
651         double span;
652         if (guessDate.shiftedBy(shift).compareTo(endDate) > 0) {
653             // Take a 1e-3 security margin
654             span = endDate.durationFrom(guessDate) - 1e-3;
655         } else {
656             span = shift;
657         }
658 
659         while (!UnivariateSolverUtils.isBracketing(latitudeFunction, -span, span)) {
660 
661             if (2 * span > maxShift) {
662                 // let the Hipparchus exception be thrown
663                 UnivariateSolverUtils.verifyBracketing(latitudeFunction, -span, span);
664             } else if (guessDate.shiftedBy(2 * span).compareTo(endDate) > 0) {
665                 // Out of range :
666                 return null;
667             }
668 
669             // expand the search interval
670             span *= 2;
671 
672         }
673 
674         // find the encounter in the bracketed interval
675         final BaseUnivariateSolver<UnivariateFunction> solver =
676                 new BracketingNthOrderBrentSolver(0.1, 5);
677         final double dt = solver.solve(1000, latitudeFunction,-span, span);
678         return propagator.propagate(guessDate.shiftedBy(dt));
679 
680     }
