18  package fr.cs.examples.frames;
19  
20  import java.util.Locale;
21  
22  import org.hipparchus.geometry.euclidean.threed.Rotation;
23  import org.hipparchus.geometry.euclidean.threed.RotationConvention;
24  import org.hipparchus.geometry.euclidean.threed.Vector3D;
25  import org.hipparchus.util.FastMath;
26  import org.orekit.errors.OrekitException;
27  import org.orekit.frames.Frame;
28  import org.orekit.frames.FramesFactory;
29  import org.orekit.frames.Transform;
30  import org.orekit.frames.UpdatableFrame;
31  import org.orekit.time.AbsoluteDate;
32  import org.orekit.time.TimeScale;
33  import org.orekit.time.TimeScalesFactory;
34  import org.orekit.utils.IERSConventions;
35  import org.orekit.utils.PVCoordinates;
36  
37  import fr.cs.examples.Autoconfiguration;
38  
39  /** Orekit tutorial for advanced frames support.
40   * <p>This tutorial shows a smart usage of frames and transforms.</p>
41   * @author Pascal Parraud
42   */
43  public class Frames2 {
44  
45      public static void main(String[] args) {
46          try {
47  
48              // configure Orekit
49              Autoconfiguration.configureOrekit();
50  
51              // Considering the following Computing/Measurement date in UTC time scale
52              TimeScale utc = TimeScalesFactory.getUTC();
53              AbsoluteDate date = new AbsoluteDate(2008, 10, 01, 12, 00, 00.000, utc);
54  
55              // The Center of Gravity frame has its origin at the satellite center of gravity (CoG)
56              // and its axes parallel to EME2000. It is derived from EME2000 frame at any moment
57              // by an unknown transform which depends on the current position and the velocity.
58              // Let's initialize this transform by the identity transform.
59              UpdatableFrame cogFrame = new UpdatableFrame(FramesFactory.getEME2000(), Transform.IDENTITY, "LOF", false);
60  
61              // The satellite frame, with origin also at the CoG, depends on attitude.
62              // For the sake of this tutorial, we consider a simple inertial attitude here
63              Transform cogToSat = new Transform(date, new Rotation(0.6, 0.48, 0, 0.64, false));
64              Frame satFrame = new Frame(cogFrame, cogToSat, "sat", false);
65  
66              // Finally, the GPS antenna frame can be defined from the satellite frame by 2 transforms:
67              // a translation and a rotation
68              Transform translateGPS = new Transform(date, new Vector3D(0, 0, 1));
69              Transform rotateGPS    = new Transform(date, new Rotation(new Vector3D(0, 1, 3),
70                                                                        FastMath.toRadians(10),
71                                                                        RotationConvention.VECTOR_OPERATOR));
72              Frame gpsFrame         = new Frame(satFrame, new Transform(date, translateGPS, rotateGPS), "GPS", false);
73  
74  
75              // Let's get the satellite position and velocity in ITRF as measured by GPS antenna at this moment:
76              final Vector3D position = new Vector3D(-6142438.668, 3492467.560, -25767.25680);
77              final Vector3D velocity = new Vector3D(505.8479685, 942.7809215, 7435.922231);
78              System.out.format(Locale.US, "GPS antenna position in ITRF:    %12.3f %12.3f %12.3f%n",
79                                position.getX(), position.getY(), position.getZ());
80              System.out.format(Locale.US, "GPS antenna velocity in ITRF:    %12.7f %12.7f %12.7f%n",
81                                velocity.getX(), velocity.getY(), velocity.getZ());
82  
83              // The transform from GPS frame to ITRF frame at this moment is defined by
84              // a translation and a rotation. The translation is directly provided by the
85              // GPS measurement above. The rotation is extracted from the existing tree, where
86              // we know all rotations are already up to date, even if one translation is still
87              // unknown. We combine the extracted rotation and the measured translation by
88              // applying the rotation first because the position/velocity vector are given in
89              // ITRF frame not in GPS antenna frame:
90              Transform measuredTranslation = new Transform(date, position, velocity);
91              Transform formerTransform     = gpsFrame.getTransformTo(FramesFactory.getITRF(IERSConventions.IERS_2010, true), date);
92              Transform preservedRotation   = new Transform(date,
93                                                            formerTransform.getRotation(),
94                                                            formerTransform.getRotationRate());
95              Transform gpsToItrf           = new Transform(date, preservedRotation, measuredTranslation);
96  
97              // So we can update the transform from EME2000 to CoG frame
98              cogFrame.updateTransform(gpsFrame, FramesFactory.getITRF(IERSConventions.IERS_2010, true), gpsToItrf, date);
99  
100             // And we can get the position and velocity of satellite CoG in EME2000 frame
101             PVCoordinates origin  = PVCoordinates.ZERO;
102             Transform cogToItrf   = cogFrame.getTransformTo(FramesFactory.getITRF(IERSConventions.IERS_2010, true), date);
103             PVCoordinates satItrf = cogToItrf.transformPVCoordinates(origin);
104             System.out.format(Locale.US, "Satellite   position in ITRF:    %12.3f %12.3f %12.3f%n",
105                               satItrf.getPosition().getX(), satItrf.getPosition().getY(), satItrf.getPosition().getZ());
106             System.out.format(Locale.US, "Satellite   velocity in ITRF:    %12.7f %12.7f %12.7f%n",
107                               satItrf.getVelocity().getX(), satItrf.getVelocity().getY(), satItrf.getVelocity().getZ());
108 
109             Transform cogToEme2000   = cogFrame.getTransformTo(FramesFactory.getEME2000(), date);
110             PVCoordinates satEME2000 = cogToEme2000.transformPVCoordinates(origin);
111             System.out.format(Locale.US, "Satellite   position in EME2000: %12.3f %12.3f %12.3f%n",
112                               satEME2000.getPosition().getX(), satEME2000.getPosition().getY(), satEME2000.getPosition().getZ());
113             System.out.format(Locale.US, "Satellite   velocity in EME2000: %12.7f %12.7f %12.7f%n",
114                               satEME2000.getVelocity().getX(), satEME2000.getVelocity().getY(), satEME2000.getVelocity().getZ());
115 
116         } catch (OrekitException oe) {
117             System.err.println(oe.getMessage());
118         }
119     }
120 
121 }
