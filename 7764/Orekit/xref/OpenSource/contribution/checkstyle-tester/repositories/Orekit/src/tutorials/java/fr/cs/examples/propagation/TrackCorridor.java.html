69      public static void main(String[] args) {

76              File input  = new File(TrackCorridor.class.getResource("/track-corridor.in").toURI().getPath());

79              new TrackCorridor().run(input, output, ",");

85              System.exit(1);

88              System.exit(1);
89          } catch (IllegalArgumentException iae) {
90              System.err.println(iae.getLocalizedMessage());
91              System.exit(1);
92          } catch (OrekitException oe) {
93              System.err.println(oe.getLocalizedMessage());
94              System.exit(1);
95          }
96      }

98      /** Input parameter keys. */
99      private static enum ParameterKey {
100 
101         TLE_LINE1,
102         TLE_LINE2,
103         ORBIT_CIRCULAR_DATE,
104         ORBIT_CIRCULAR_A,
105         ORBIT_CIRCULAR_EX,
106         ORBIT_CIRCULAR_EY,
107         ORBIT_CIRCULAR_I,
108         ORBIT_CIRCULAR_RAAN,
109         ORBIT_CIRCULAR_ALPHA,
110         START_DATE,
111         DURATION,
112         STEP,
113         ANGULAR_OFFSET;
114 
115     }
116 
117     private void run(final File input, final File output, final String separator)
118             throws IOException, IllegalArgumentException, OrekitException {
119 
120         // read input parameters
121         KeyValueFileParser<ParameterKey> parser =
122                 new KeyValueFileParser<ParameterKey>(ParameterKey.class);
123         try (final FileInputStream fis = new FileInputStream(input)) {
124             parser.parseInput(input.getAbsolutePath(), fis);
125         }
126         TimeScale utc = TimeScalesFactory.getUTC();
127 
128         Propagator propagator;
129         if (parser.containsKey(ParameterKey.TLE_LINE1)) {
130             propagator = createPropagator(parser.getString(ParameterKey.TLE_LINE1),
131                                           parser.getString(ParameterKey.TLE_LINE2));
132         } else {
133             propagator = createPropagator(parser.getDate(ParameterKey.ORBIT_CIRCULAR_DATE, utc),
134                                           parser.getDouble(ParameterKey.ORBIT_CIRCULAR_A),
135                                           parser.getDouble(ParameterKey.ORBIT_CIRCULAR_EX),
136                                           parser.getDouble(ParameterKey.ORBIT_CIRCULAR_EY),
137                                           parser.getAngle(ParameterKey.ORBIT_CIRCULAR_I),
138                                           parser.getAngle(ParameterKey.ORBIT_CIRCULAR_RAAN),
139                                           parser.getAngle(ParameterKey.ORBIT_CIRCULAR_ALPHA));
140         }
141 
142         // simulation properties
143         AbsoluteDate start = parser.getDate(ParameterKey.START_DATE, utc);
144         double duration    = parser.getDouble(ParameterKey.DURATION);
145         double step        = parser.getDouble(ParameterKey.STEP);
146         double angle       = parser.getAngle(ParameterKey.ANGULAR_OFFSET);
147 
148         // set up a handler to gather all corridor points
149         CorridorHandler handler = new CorridorHandler(angle);
150         propagator.setMasterMode(step, handler);
151 
152         // perform propagation, letting the step handler populate the corridor
153         propagator.propagate(start, start.shiftedBy(duration));
154 
155         // retrieve the built corridor
156         List<CorridorPoint> corridor = handler.getCorridor();
157 
158         // create a 7 columns csv file representing the corridor in the user home directory, with
159         // date in column 1 (in ISO-8601 format)
160         // left limit latitude in column 2 and left limit longitude in column 3
161         // center track latitude in column 4 and center track longitude in column 5
162         // right limit latitude in column 6 and right limit longitude in column 7
163         DecimalFormat format = new DecimalFormat("#00.00000", new DecimalFormatSymbols(Locale.US));
164         try (final PrintStream stream = new PrintStream(output, "UTF-8")) {
165             for (CorridorPoint p : corridor) {
166                 stream.println(p.getDate() + separator +
167                                format.format(FastMath.toDegrees(p.getLeft().getLatitude()))    + separator +
168                                format.format(FastMath.toDegrees(p.getLeft().getLongitude()))   + separator +
169                                format.format(FastMath.toDegrees(p.getCenter().getLatitude()))  + separator +
170                                format.format(FastMath.toDegrees(p.getCenter().getLongitude())) + separator +
171                                format.format(FastMath.toDegrees(p.getRight().getLatitude()))   + separator +
172                                format.format(FastMath.toDegrees(p.getRight().getLongitude())));
173             }
174         }
175 
176     }

178     /** Create an orbit propagator for a circular orbit
179      * @param a  semi-major axis (m)
180      * @param ex e cos(ω), first component of circular eccentricity vector
181      * @param ey e sin(ω), second component of circular eccentricity vector
182      * @param i inclination (rad)
183      * @param raan right ascension of ascending node (Ω, rad)
184      * @param alpha  an + ω, mean latitude argument (rad)
185      * @param date date of the orbital parameters
186      * @return an orbit propagator
187      * @exception OrekitException if propagator cannot be built
188      */
189     private Propagator createPropagator(final AbsoluteDate date,
190                                         final double a, final double ex, final double ey,
191                                         final double i, final double raan,
192                                         final double alpha)
193         throws OrekitException {
194 
195         // create orbit
196         Orbit initialOrbit = new CircularOrbit(a, ex, ey, i, raan, alpha, PositionAngle.MEAN,
197                                                FramesFactory.getEME2000(), date,
198                                                Constants.EIGEN5C_EARTH_MU);
199 
200         // create propagator
201         Propagator propagator =
202                 new EcksteinHechlerPropagator(initialOrbit,
203                                               new LofOffset(initialOrbit.getFrame(), LOFType.TNW),
204                                               Constants.EIGEN5C_EARTH_EQUATORIAL_RADIUS,
205                                               Constants.EIGEN5C_EARTH_MU,  Constants.EIGEN5C_EARTH_C20,
206                                               Constants.EIGEN5C_EARTH_C30, Constants.EIGEN5C_EARTH_C40,
207                                               Constants.EIGEN5C_EARTH_C50, Constants.EIGEN5C_EARTH_C60);
208 
209         return propagator;
210 
211     }
212 
213     /** Create an orbit propagator for a TLE orbit
214      * @param line1 firs line of the TLE
215      * @param line2 second line of the TLE
216      * @return an orbit propagator
217      * @exception OrekitException if the TLE lines are corrupted (wrong checksums ...)
218      */
219     private Propagator createPropagator(final String line1, final String line2)
220         throws OrekitException {
221 
222         // create pseudo-orbit
223         TLE tle = new TLE(line1, line2);
224 
225         // create propagator
226         Propagator propagator = TLEPropagator.selectExtrapolator(tle);
227 
228         return propagator;
229 
230     }

232     /** Step handler storing corridor points. */
233     private static class CorridorHandler implements OrekitFixedStepHandler {
234 
235         /** Earth model. */
236         private final BodyShape earth;
237 
238         /** Radial offset from satellite to some distant point at specified angular offset. */
239         private final double deltaR;
240 
241         /** Cross-track offset from satellite to some distant point at specified angular offset. */
242         private final double deltaC;
243 
244         /** Corridor. */
245         private final List<CorridorPoint> corridor;
246 
247         /** simple constructor.
248          * @param angle angular offset of corridor boundaries
249          * @exception OrekitException if Earth frame cannot be built
250          */
251         public CorridorHandler(final double angle) throws OrekitException {
252 
253             // set up Earth model
254             earth = new OneAxisEllipsoid(Constants.WGS84_EARTH_EQUATORIAL_RADIUS,
255                                          Constants.WGS84_EARTH_FLATTENING,
256                                          FramesFactory.getITRF(IERSConventions.IERS_2010, false));
257 
258             // set up position offsets, using Earth radius as an arbitrary distance
259             deltaR = Constants.WGS84_EARTH_EQUATORIAL_RADIUS * FastMath.cos(angle);
260             deltaC = Constants.WGS84_EARTH_EQUATORIAL_RADIUS * FastMath.sin(angle);
261 
262             // prepare an empty corridor
263             corridor = new ArrayList<TrackCorridor.CorridorPoint>();
264 
265         }
266 
267         /** {@inheritDoc} */
268         public void init(final SpacecraftState s0, final AbsoluteDate t) {
269         }
270 
271         /** {@inheritDoc} */
272         public void handleStep(SpacecraftState currentState, boolean isLast)
273             throws OrekitException {
274 
275             // compute sub-satellite track
276             AbsoluteDate  date    = currentState.getDate();
277             PVCoordinates pvInert = currentState.getPVCoordinates();
278             Transform     t       = currentState.getFrame().getTransformTo(earth.getBodyFrame(), date);
279             Vector3D      p       = t.transformPosition(pvInert.getPosition());
280             Vector3D      v       = t.transformVector(pvInert.getVelocity());
281             GeodeticPoint center  = earth.transform(p, earth.getBodyFrame(), date);
282 
283             // compute left and right corridor points
284             Vector3D      nadir      = p.normalize().negate();
285             Vector3D      crossTrack = p.crossProduct(v).normalize();
286             Line          leftLine   = new Line(p, new Vector3D(1.0, p, deltaR, nadir,  deltaC, crossTrack), 1.0e-10);
287             GeodeticPoint left       = earth.getIntersectionPoint(leftLine, p, earth.getBodyFrame(), date);
288             Line          rightLine  = new Line(p, new Vector3D(1.0, p, deltaR, nadir, -deltaC, crossTrack), 1.0e-10);
289             GeodeticPoint right      = earth.getIntersectionPoint(rightLine, p, earth.getBodyFrame(), date);
290 
291             // add the corridor points
292             corridor.add(new CorridorPoint(date, left, center, right));
293 
294         }
295 
296         /** Get the corridor.
297          * @return build corridor
298          */
299         public List<CorridorPoint> getCorridor() {
300             return corridor;
301         }
302 
303     }

305     /** Container for corridor points. */
306     private static class CorridorPoint {
307 
308         /** Point date. */
309         private final AbsoluteDate date;
310 
311         /** Left limit. */
312         private final GeodeticPoint left;
313 
314         /** Central track point. */
315         private final GeodeticPoint center;
316 
317         /** Right limit. */
318         private final GeodeticPoint right;
319 
320         /** Simple constructor.
321          * @param date point date
322          * @param left left limit
323          * @param center central track point
324          * @param right right limit
325          */
326         public CorridorPoint(AbsoluteDate date, GeodeticPoint left,
327                              GeodeticPoint center, GeodeticPoint right) {
328             this.date   = date;
329             this.left   = left;
330             this.center = center;
331             this.right  = right;
332         }
333 
334         /** Get point date. */
335         public AbsoluteDate getDate() {
336             return date;
337         }
338 
339         /** Get left limit. */
340         public GeodeticPoint getLeft() {
341             return left;
342         }
343 
344         /** Get central track point. */
345         public GeodeticPoint getCenter() {
346             return center;
347         }
348 
349         /** Get right limit. */
350         public GeodeticPoint getRight() {
351             return right;
352         }
