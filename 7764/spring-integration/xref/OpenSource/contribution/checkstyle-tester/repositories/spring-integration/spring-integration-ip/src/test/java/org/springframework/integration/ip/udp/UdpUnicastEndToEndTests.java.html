44  /**
45   * Sends and receives a simple message through to the Udp channel adapters.
46   * If run as a JUnit just sends one message and terminates (see console).
47   *
48   * If run from main(),
49   * hangs around for a couple of minutes to allow console interaction - enter a message on the
50   * console and you should see it go through the outbound context, over UDP, and
51   * received in the other context (and written back to the console).
52   *
53   * @author Gary Russell
54   * @author Artem Bilan
55   * @author Gunnar Hillert
56   * @since 2.0
57   */
58  public class UdpUnicastEndToEndTests implements Runnable {
59  
60  	private String testingIpText;
61  
62  	private Message<byte[]> finalMessage;
63  
64  	private CountDownLatch sentFirst = new CountDownLatch(1);
65  
66  	private CountDownLatch firstReceived = new CountDownLatch(1);
67  
68  	private CountDownLatch doneProcessing = new CountDownLatch(1);
69  
70  	private boolean okToRun = true;
71  
72  	private CountDownLatch readyToReceive = new CountDownLatch(1);
73  
74  	private volatile int receiverPort;
75  
76  	private static long hangAroundFor = 10;
77  
78  	@Before
79  	public void setup() {
80  		this.testingIpText = null;
81  		this.finalMessage = null;
82  		this.sentFirst = new CountDownLatch(1);
83  		this.firstReceived = new CountDownLatch(1);
84  		this.doneProcessing = new CountDownLatch(1);
85  		this.okToRun = true;
86  		this.readyToReceive = new CountDownLatch(1);
87  	}
88  
89  	@Test
90  	public void runIt() throws Exception {
91  		String location = "org/springframework/integration/ip/udp/testIp-out-context.xml";
92  		test(location);
93  	}
94  
95  	@Test
96  	public void tesUdpOutboundChannelAdapterWithinChain() throws Exception {
97  		String location = "org/springframework/integration/ip/udp/testIp-out-within-chain-context.xml";
98  		test(location);
99  	}
100 
101 	private void test(String location) throws InterruptedException, Exception {
102 		UdpUnicastEndToEndTests launcher = new UdpUnicastEndToEndTests();
103 		Thread t = new Thread(launcher);
104 		t.start(); // launch the receiver
105 		int n = 0;
106 		while (n++ < 100 && launcher.getReceiverPort() == 0) {
107 			Thread.sleep(100);
108 		}
109 		assertThat(n < 100).as("Receiver failed to listen").isTrue();
110 
111 		ClassPathXmlApplicationContext applicationContext = createContext(launcher, location);
112 		launcher.launchSender(applicationContext);
113 		applicationContext.close();
114 	}
115 
116 	private ClassPathXmlApplicationContext createContext(UdpUnicastEndToEndTests launcher, String location) {
117 		ClassPathXmlApplicationContext applicationContext = new ClassPathXmlApplicationContext();
118 		applicationContext.setConfigLocation(location);
119 		StandardEnvironment env = new StandardEnvironment();
120 		Properties props = new Properties();
121 		props.setProperty("port", Integer.toString(launcher.getReceiverPort()));
122 		PropertiesPropertySource pps = new PropertiesPropertySource("ftpprops", props);
123 		env.getPropertySources().addLast(pps);
124 		applicationContext.setEnvironment(env);
125 		applicationContext.refresh();
126 		return applicationContext;
127 	}
128 
129 
130 	public void launchSender(ApplicationContext applicationContext) throws Exception {
131 		DestinationResolver<MessageChannel> channelResolver = new BeanFactoryChannelResolver(applicationContext);
132 		MessageChannel inputChannel = channelResolver.resolveDestination("inputChannel");
133 		if (!readyToReceive.await(30, TimeUnit.SECONDS)) {
134 			fail("Receiver failed to start in 30s");
135 		}
136 		try {
137 			testingIpText = ">>>>>>> Testing IP " + new Date();
138 			inputChannel.send(new GenericMessage<String>(testingIpText));
139 			sentFirst.countDown();
140 			try {
141 				Thread.sleep(hangAroundFor); // give some time for console interaction
142 			}
143 			catch (InterruptedException e) {
144 				e.printStackTrace();
145 			}
146 		}
147 		finally {
148 			if (hangAroundFor == 0) {
149 				sentFirst = new CountDownLatch(1);
150 			}
151 			else {
152 				okToRun = false;
153 			}
154 			// tell the receiver to we're done
155 			doneProcessing.countDown();
156 		}
157 		assertThat(firstReceived.await(5, TimeUnit.SECONDS)).isTrue();
158 		assertThat(new String(finalMessage.getPayload())).isEqualTo(testingIpText);
159 	}
160 
161 	public int getReceiverPort() {
162 		return receiverPort;
163 	}
164 
165 	/**
166 	 * Instantiate the receiving context
167 	 */
168 	@SuppressWarnings("unchecked")
169 	@Override
170 	public void run() {
171 		@SuppressWarnings("resource")
172 		AbstractApplicationContext ctx = new ClassPathXmlApplicationContext(
173 				"testIp-in-context.xml", UdpUnicastEndToEndTests.class);
174 		UnicastReceivingChannelAdapter inbound = ctx.getBean(UnicastReceivingChannelAdapter.class);
175 		int n = 0;
176 		try {
177 			while (!inbound.isListening()) {
178 				Thread.sleep(100);
179 				if (n++ > 100) {
180 					throw new RuntimeException("Failed to start listening");
181 				}
182 			}
183 		}
184 		catch (RuntimeException e) {
185 			throw e;
186 		}
187 		catch (InterruptedException e) {
188 			Thread.currentThread().interrupt();
189 			throw new RuntimeException("interrupted");
190 		}
191 		this.receiverPort = inbound.getPort();
192 		while (okToRun) {
193 			try {
194 				readyToReceive.countDown();
195 				sentFirst.await();
196 			}
197 			catch (InterruptedException e) {
198 				e.printStackTrace();
199 			}
200 			QueueChannel channel = ctx.getBean("udpOutChannel", QueueChannel.class);
201 			finalMessage = (Message<byte[]>) channel.receive();
202 			MessageHistory history = MessageHistory.read(finalMessage);
203 			Properties componentHistoryRecord = TestUtils.locateComponentInHistory(history, "udpReceiver", 0);
204 			assertThat(componentHistoryRecord).isNotNull();
205 			assertThat(componentHistoryRecord.get("type")).isEqualTo("ip:udp-inbound-channel-adapter");
206 			firstReceived.countDown();
207 			try {
208 				doneProcessing.await();
209 			}
210 			catch (InterruptedException e) {
211 				e.printStackTrace();
212 			}
213 		}
214 		ctx.stop();
215 		ctx.close();
216 	}

219 	public static void main(String[] args) throws Exception {
220 		hangAroundFor = 120000;
221 		new UdpUnicastEndToEndTests().runIt();
222 		new UdpUnicastEndToEndTests().tesUdpOutboundChannelAdapterWithinChain();
223 	}
