17  package com.google.javascript.refactoring;
18  
19  import static com.google.common.base.Preconditions.checkArgument;
20  
21  import com.google.common.base.Preconditions;
22  import com.google.common.base.Splitter;
23  import com.google.common.base.Strings;
24  import com.google.javascript.jscomp.CommandLineRunner;
25  import com.google.javascript.jscomp.CompilerOptions;
26  import com.google.javascript.jscomp.ErrorManager;
27  import com.google.javascript.jscomp.TypeMatchingStrategy;
28  import java.io.File;
29  import java.io.IOException;
30  import java.util.ArrayList;
31  import java.util.HashSet;
32  import java.util.List;
33  import java.util.Set;
34  import java.util.TreeSet;
35  import java.util.logging.Level;
36  import java.util.logging.Logger;
37  import org.kohsuke.args4j.Argument;
38  import org.kohsuke.args4j.CmdLineParser;
39  import org.kohsuke.args4j.Option;
40  import org.kohsuke.args4j.spi.BooleanOptionHandler;
41  
42  /**
43   * Main binary that drives a RefasterJS refactoring.
44   */
45  final class RefasterJs {
46  
47    @Option(name = "--help",
48          hidden = true,
49          handler = BooleanOptionHandler.class,
50          usage = "Show instructions for how to use RefasterJS")
51    private boolean displayHelp = false;
52  
53    @Option(
54        name = "--inputs",
55        usage = "List of input files for the refactoring. You may also use glob patterns to "
56            + "match files. For example, use --js='**.js' --js='!**_test.js' "
57            + "to recursively include all js files that do not end in _test.js")
58    private List<String> inputs = new ArrayList<>();
59  
60    @Option(name = "--externs", usage = "List of externs files to use in the compilation.")
61    private List<String> externs = new ArrayList<>();
62  
63    @Option(
64        name = "--refasterjs_template",
65        usage = "Location of the JS file to use as the RefasterJS template.")
66    private String refasterJsTemplate = null;
67  
68    @Option(name = "--env",
69        usage = "Which set of externs to include. Defaults to BROWSER.")
70    private CompilerOptions.Environment environment =
71        CompilerOptions.Environment.BROWSER;
72  
73    @Option(name = "--type_matching",
74      usage = "Which type matching strategy to use. Defaults to SUBTYPES.")
75    private TypeMatchingStrategy typeMatchingStrategy = TypeMatchingStrategy.SUBTYPES;
76  
77    @Option(name = "--dry_run",
78        usage = "Use this to display what changes would be made without applying the changes.")
79    private boolean dryRun = false;
80  
81    @Option(name = "--verbose", usage = "Use this to print verbose statements from RefasterJS.")
82    private boolean verbose = false;
83  
84    @Argument
85    private List<String> arguments = new ArrayList<>();
86  
87    private void doMain(String[] args) throws Exception {
88      CmdLineParser parser = new CmdLineParser(this);
89      parser.parseArgument(args);
90      if (args.length < 1 || displayHelp) {
91        CmdLineParser p = new CmdLineParser(this);
92        p.printUsage(System.out);
93        return;
94      }
95      checkArgument(
96          !Strings.isNullOrEmpty(refasterJsTemplate), "--refasterjs_template must be provided");
97      List<String> fileInputs = getInputs();
98      checkArgument(
99          !fileInputs.isEmpty(), "At least one input must be provided in the --inputs flag.");
100     for (String input : fileInputs) {
101       Preconditions.checkArgument(
102           new File(input).exists(), "Input file %s does not exist.", input);
103     }
104 
105     if (!verbose) {
106       // This is done here instead of using the Compiler#setLoggingLevel function since the
107       // Compiler is created and then run inside of RefactoringDriver.
108       Logger errorManagerLogger = Logger.getLogger("com.google.javascript.jscomp");
109       errorManagerLogger.setLevel(Level.OFF);
110     }
111 
112     RefasterJsScanner scanner = new RefasterJsScanner();
113     scanner.setTypeMatchingStrategy(typeMatchingStrategy);
114     scanner.loadRefasterJsTemplate(refasterJsTemplate);
115     CompilerOptions options = new CompilerOptions();
116     options.setEnvironment(environment);
117     RefactoringDriver driver =
118         new RefactoringDriver.Builder()
119             .addExterns(CommandLineRunner.getBuiltinExterns(environment))
120             .addExternsFromFile(getExterns())
121             .addInputsFromFile(fileInputs)
122             .build();
123     System.out.println("Compiling JavaScript code and searching for suggested fixes.");
124     // TODO(bangert): allow picking a non-default choice in RefasterJS, e.g. via a switch.
125     List<SuggestedFix> fixes = driver.drive(scanner);
126 
127     if (!verbose) {
128       // When running in quiet mode, the Compiler's error manager will not have printed
129       // this information itself.
130       ErrorManager errorManager = driver.getCompiler().getErrorManager();
131       System.out.println("Compiler results: " + errorManager.getErrorCount()
132           + " errors and " + errorManager.getWarningCount() + " warnings.");
133     }
134     System.out.println("Found " + fixes.size() + " suggested fixes.");
135     if (dryRun) {
136       if (!fixes.isEmpty()) {
137         System.out.println("SuggestedFixes: " + fixes);
138       }
139     } else {
140       Set<String> affectedFiles = new TreeSet<>();
141       for (SuggestedFix fix : fixes) {
142         affectedFiles.addAll(fix.getReplacements().keySet());
143       }
144       System.out.println("Modifying affected files: " + affectedFiles);
145       ApplySuggestedFixes.applySuggestedFixesToFiles(fixes);
146     }
147   }
148 
149   private List<String> getInputs() throws IOException {
150     Set<String> patterns = new HashSet<>();
151     // The args4j library can't handle multiple files provided within the same flag option,
152     // like --inputs=file1.js,file2.js so handle that here.
153     Splitter commaSplitter = Splitter.on(',');
154     for (String input : inputs) {
155       patterns.addAll(commaSplitter.splitToList(input));
156     }
157     patterns.addAll(arguments);
158     return CommandLineRunner.findJsFiles(patterns);
159   }
160 
161   private List<String> getExterns() throws IOException {
162     Set<String> patterns = new HashSet<>();
163     // The args4j library can't handle multiple files provided within the same flag option,
164     // like --externs=file1.js,file2.js so handle that here.
165     Splitter commaSplitter = Splitter.on(',');
166     for (String extern : externs) {
167       patterns.addAll(commaSplitter.splitToList(extern));
168     }
169     return CommandLineRunner.findJsFiles(patterns);
170   }
171 
172   public static void main(String[] args) throws Exception {
173     new RefasterJs().doMain(args);
174   }
175 }
