34  /**
35   * PerformanceCompareStressTest is useful to get an idea on relative performance between Infinispan
36   * in local or clustered mode against a RAMDirectory or FSDirectory. To be reliable set a long
37   * DURATION_MS and a number of threads similar to the use case you're interested in: results might
38   * vary on the number of threads because of the lock differences. This is not meant as a benchmark
39   * but used to detect regressions.
40   *
41   * This requires Lucene > 2.9.1 or Lucene > 3.0.0 because of
42   * https://issues.apache.org/jira/browse/LUCENE-2095
43   *
44   * @author Sanne Grinovero
45   * @since 4.0
46   */

155    private void setNetworkDelay(int delay) throws Exception {
156       for (int i=0; i<NUM_NODES; i++) {
157          EmbeddedCacheManager cm = cacheManagers.get(i);
158          //Any cache will do:
159          TestingUtil.setDelayForCache(cm.getCache(), delay, delay);
160       }
161       System.out.println("Simulating network packet delay of: " + delay);
162    }

164    private ThreadFactory createThreadFactory() {
165       return new ThreadFactory() {
166          private AtomicInteger atomicInteger = new AtomicInteger(0);
167 
168          @Override
169          public Thread newThread(Runnable runnable) {
170             Thread thread = new Thread(runnable);
171             thread.setName("File-deleter-" + atomicInteger.incrementAndGet());
172             return thread;
173          }
174       };
175    }
176 
177    private Executor createDeleteExecutor() {
178       return new ThreadPoolExecutor(
179             0,
180             ASYNC_DELETES_POOL_SIZE,
181             60L,
182             TimeUnit.SECONDS,
183             new SynchronousQueue<Runnable>(),
184             createThreadFactory(),
185             new ThreadPoolExecutor.CallerRunsPolicy());
186    }

188    private Directory buildDirectoryFromNode(int node) {
189       EmbeddedCacheManager cm = cacheManagers.get(node);
190       BuildContext context = DirectoryBuilder
191             .newDirectoryInstance(cm.getCache("index_metadata"), cm.getCache("index_data"), cm.getCache("index_locks"), indexName)
192             .writeFileListAsynchronously(ASYNC_METADATA_WRITES)
193             .chunkSize(CHUNK_SIZE);
194       if (ASYNC_DELETES) {
195          context.deleteOperationsExecutor(createDeleteExecutor());
196       }
197       return context.create();
198    }

200    @Test(enabled=false)//to prevent invocations from some versions of TestNG
201    private static void stressTestDirectory(Directory dirWriter, Directory dirReaders, String testLabel, long durationMs, Properties results, String currentMethod) throws InterruptedException, IOException {
202       SharedState state = new SharedState(DICTIONARY_SIZE);
203       CacheTestSupport.initializeDirectory(dirWriter);
204       ExecutorService e = Executors.newFixedThreadPool(READER_THREADS + WRITER_THREADS);
205       for (int i = 0; i < READER_THREADS; i++) {
206          e.execute(new LuceneReaderThread(dirReaders, state));
207       }
208       for (int i = 0; i < WRITER_THREADS; i++) {
209          e.execute(INDEX_EXCLUSIVE ? new LuceneWriterExclusiveThread(dirWriter, state) : new LuceneWriterThread(dirWriter, state));
210       }
211       e.shutdown();
212       System.out.println("Started test: " + testLabel);
213       state.startWaitingThreads();
214       Thread.sleep(durationMs);
215       long searchesCount = state.incrementIndexSearchesCount(0);
216       long writerTaskCount = state.incrementIndexWriterTaskCount(0);
217       state.quit();
218       boolean terminatedCorrectly = e.awaitTermination(20, TimeUnit.SECONDS);
219       AssertJUnit.assertTrue(terminatedCorrectly);
220       System.out.println("Test " + testLabel + " run in " + durationMs + "ms:\n\tSearches: " + searchesCount + "\n\t" + "Writes: "
221                + writerTaskCount);
222       if (results != null) {
223          results.setProperty(currentMethod + ".label", testLabel);
224          results.setProperty(currentMethod + ".searches", Long.toString(searchesCount));
225          results.setProperty(currentMethod + ".writes", Long.toString(writerTaskCount));
226       }
227    }

229    @BeforeMethod
230    public void beforeTest() throws IOException {
231       for (int i = 0; i < NUM_NODES; i++) {
232          DefaultCacheManager node = new DefaultCacheManager(CONFIGURATION);
233          node.start();
234          //Start all its caches:
235          node.getCache("index_metadata").start();
236          node.getCache("index_data").start();
237          node.getCache("index_locks").start();
238          cacheManagers.put(i, node);
239       }
240    }

244       for (EmbeddedCacheManager node : cacheManagers.values()) {
245          TestingUtil.killCacheManagers(node);
246       }
247       TestingUtil.recursiveFileRemove(indexName);
248    }

256    /**
257     * It's much better to compare performance out of the scope of TestNG by
258     * running this directly as TestNG enables assertions.
259     *
260     * Select which tests to run:
261     * -Dlucene.profiling.tests=profileInfinispanLocalDirectory,profileTestInfinispanDirectoryWithNetworkDelayZero
262     *
263     * Suggested test switches:
264     * -Xmx4G -Xms4G -XX:MaxPermSize=128M -XX:+HeapDumpOnOutOfMemoryError -Xss512k -XX:HeapDumpPath=/tmp/java_heap -Djava.net.preferIPv4Stack=true -Djgroups.bind_addr=127.0.0.1 -XX:+UseLargePages -XX:LargePageSizeInBytes=2m
265     *
266     * With detailed GC logging:
267     * -Xmx4G -Xms4G -XX:MaxPermSize=32M -XX:+HeapDumpOnOutOfMemoryError -Xss256k -XX:HeapDumpPath=/tmp/java_heap -Djava.net.preferIPv4Stack=true -Djgroups.bind_addr=127.0.0.1 -XX:+UseLargePages -XX:LargePageSizeInBytes=2m -XX:+UseLargePages -XX:LargePageSizeInBytes=2m -Xloggc:gc-full.log -XX:+PrintGCDetails -XX:+PrintTenuringDistribution -XX:+PrintGCApplicationStoppedTime
268     *
269     * To enable flight recorder:
270     * -XX:+UnlockCommercialFeatures -XX:+FlightRecorder
271     */
272    public static void main(String[] args) throws Exception {
273       String[] testMethods = System.getProperty("lucene.profiling.tests",
274             "profileTestRAMDirectory,profileTestFSDirectory,profileInfinispanLocalDirectory,profileTestInfinispanDirectoryWithNetworkDelayZero").split(",");
275       PerformanceCompareStressTest test = new PerformanceCompareStressTest();
276       test.durationMs = new Long(System.getProperty("lucene.profiling.duration", String.valueOf(DEFAULT_DURATION_MS)));
277       String outputFile = System.getProperty("lucene.profiling.output");
278       test.results = outputFile == null ? null : new Properties();
279       for (String testMethod : testMethods) {
280          try {
281             test.currentMethod = testMethod;
282             Method m = PerformanceCompareStressTest.class.getMethod(testMethod);
283             test.beforeTest();
284             try {
285                m.invoke(test);
286             } finally {
287                test.afterTest();
288             }
289          } catch (NoSuchMethodException e) {
290             System.out.println("Couldn't find method " + testMethod);
291             System.exit(1);
292          }
293       }
294       if (test.results != null && !test.results.isEmpty()) {
295          System.out.println("Writing results to " + outputFile + " ...");
296          TestingUtil.outputPropertiesToXML(outputFile, test.results);
297       }
298    }
