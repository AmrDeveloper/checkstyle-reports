244 	@Override
245 	public String getLimitString(String sql, int offset, int limit) {
246 		if ( offset == 0 ) {
247 			return sql + " fetch first " + limit + " rows only";
248 		}
249 		StringBuilder pagingSelect = new StringBuilder( sql.length() + 200 )
250 				.append(
251 						"select * from ( select inner2_.*, rownumber() over(order by order of inner2_) as rownumber_ from ( "
252 				)
253 				.append( sql )  //nest the main query in an outer select
254 				.append( " fetch first " )
255 				.append( limit )
256 				.append( " rows only ) as inner2_ ) as inner1_ where rownumber_ > " )
257 				.append( offset )
258 				.append( " order by rownumber_" );
259 		return pagingSelect.toString();
260 	}

295 	@Override
296 	public String getSelectClauseNullString(int sqlType) {
297 		String literal;
298 		switch ( sqlType ) {
299 			case Types.VARCHAR:
300 			case Types.CHAR:
301 				literal = "'x'";
302 				break;
303 			case Types.DATE:
304 				literal = "'2000-1-1'";
305 				break;
306 			case Types.TIMESTAMP:
307 				literal = "'2000-1-1 00:00:00'";
308 				break;
309 			case Types.TIME:
310 				literal = "'00:00:00'";
311 				break;
312 			default:
313 				literal = "0";
314 		}
315 		return "nullif(" + literal + ',' + literal + ')';
316 	}

345 	public ResultSet getResultSet(CallableStatement ps) throws SQLException {
346 		boolean isResultSet = ps.execute();
347 		// This assumes you will want to ignore any update counts 
348 		while ( !isResultSet && ps.getUpdateCount() != -1 ) {
349 			isResultSet = ps.getMoreResults();
350 		}
351 		ResultSet rs = ps.getResultSet();
352 		// You may still have other ResultSets or update counts left to process here 
353 		// but you can't do it now or the ResultSet you just got will be closed 
354 		return rs;
355 	}

445 	@Override
446 	public SQLExceptionConversionDelegate buildSQLExceptionConversionDelegate() {
447 		return new SQLExceptionConversionDelegate() {
448 			@Override
449 			public JDBCException convert(SQLException sqlException, String message, String sql) {
450 				final String sqlState = JdbcExceptionHelper.extractSqlState( sqlException );
451 				final int errorCode = JdbcExceptionHelper.extractErrorCode( sqlException );
452 
453 				if( -952 == errorCode && "57014".equals( sqlState )){
454 					throw new LockTimeoutException( message, sqlException, sql );
455 				}
456 				return null;
457 			}
458 		};
459 	}
