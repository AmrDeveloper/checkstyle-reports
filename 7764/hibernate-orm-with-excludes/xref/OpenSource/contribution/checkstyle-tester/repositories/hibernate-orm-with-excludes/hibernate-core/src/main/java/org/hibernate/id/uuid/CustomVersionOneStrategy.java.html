24  package org.hibernate.id.uuid;
25  import java.util.UUID;
26  
27  import org.hibernate.engine.spi.SessionImplementor;
28  import org.hibernate.id.UUIDGenerationStrategy;
29  import org.hibernate.internal.util.BytesHelper;
30  
31  /**
32   * Applies a version 1 (time-based) generation strategy (using ip address rather than mac address) but applies them in a
33   * different layout.  The strategy is very similar to the legacy {@link org.hibernate.id.UUIDHexGenerator} id generator
34   * but uses a RFC 4122 compliant layout (variant 2).
35   * <p/>
36   * NOTE : Can be a bottle neck due to the need to synchronize in order to increment an
37   * internal count as part of the algorithm.
38   *
39   * @author Steve Ebersole
40   */
41  public class CustomVersionOneStrategy implements UUIDGenerationStrategy {
42  	public int getGeneratedVersion() {
43  		return 1;
44  	}
45  
46  	private final long mostSignificantBits;
47  
48  	public CustomVersionOneStrategy() {
49  		// generate the "most significant bits" ~~~~~~~~~~~~~~~~~~~~~~~~~~~~
50  		byte[] hiBits = new byte[8];
51  		// use address as first 32 bits (8 * 4 bytes)
52  		System.arraycopy( Helper.getAddressBytes(), 0, hiBits, 0, 4 );
53  		// use the "jvm identifier" as the next 32 bits
54  		System.arraycopy( Helper.getJvmIdentifierBytes(), 0, hiBits, 4, 4 );
55  		// set the version (rfc term) appropriately
56  		hiBits[6] &= 0x0f;
57  		hiBits[6] |= 0x10;
58  
59  		mostSignificantBits = BytesHelper.asLong( hiBits );
60  	}
61  
62  	public UUID generateUUID(SessionImplementor session) {
63  		long leastSignificantBits = generateLeastSignificantBits( System.currentTimeMillis() );
64  		return new UUID( mostSignificantBits, leastSignificantBits );
65  	}
66  
67  	public long getMostSignificantBits() {
68  		return mostSignificantBits;
69  	}
70  
71  	public static long generateLeastSignificantBits(long seed) {
72  		byte[] loBits = new byte[8];
73  
74  		short hiTime = (short) ( seed >>> 32 );
75  		int loTime = (int) seed;
76  		System.arraycopy( BytesHelper.fromShort( hiTime ), 0, loBits, 0, 2 );
77  		System.arraycopy( BytesHelper.fromInt( loTime ), 0, loBits, 2, 4 );
78  		System.arraycopy( Helper.getCountBytes(), 0, loBits, 6, 2 );
79  		loBits[0] &= 0x3f;
80  		loBits[0] |= ((byte)2 << (byte)6);
81  
82  		return BytesHelper.asLong( loBits );
83  	}
84  
85  	public static void main(String[] args) {
86  		CustomVersionOneStrategy strategy = new CustomVersionOneStrategy();
87  
88  		for ( int i = 0; i < 1000; i++ ) {
89  			System.out.println( "Generation # " + i + " ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~" );
90  			byte[] loBits = new byte[8];
91  
92  			long sysTime = System.currentTimeMillis();
93  			short hiTime = (short) ( System.currentTimeMillis() >>> 32 );
94  			int loTime = (int) sysTime;
95  			System.arraycopy( BytesHelper.fromShort( hiTime ), 0, loBits, 0, 2 );
96  			System.arraycopy( BytesHelper.fromInt( loTime ), 0, loBits, 2, 4 );
97  			System.arraycopy( Helper.getCountBytes(), 0, loBits, 6, 2 );
98  
99  			System.out.println( "    before bit setting ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~" );
100 			System.out.println( "       loBits[0] : " + BytesHelper.toBinaryString( loBits[0] ) );
101 			System.out.println( "             lsb : " + BytesHelper.toBinaryString( BytesHelper.asLong( loBits ) ) );
102 			System.out.println( "    ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~" );
103 
104 			loBits[0] &= 0x3f;
105 			loBits[0] |= ((byte)2 << (byte)6);
106 
107 			System.out.println( "    after bit setting ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~" );
108 			System.out.println( "       loBits[0] : " + BytesHelper.toBinaryString( loBits[0] ) );
109 			long leastSignificantBits = BytesHelper.asLong( loBits );
110 			System.out.println( "             lsb : " + BytesHelper.toBinaryString( leastSignificantBits ) );
111 			System.out.println( "    ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~" );
112 
113 
114 			UUID uuid = new UUID( strategy.mostSignificantBits, leastSignificantBits );
115 			System.out.println( "  uuid : " + uuid.toString() );
116 			System.out.println( "  variant : " + uuid.variant() );
117 			System.out.println( "  version : " + uuid.version() );
118 			if ( uuid.variant() != 2 ) {
119 				throw new RuntimeException( "bad variant" );
120 			}
121 			System.out.println( "~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~" );
122 		}
123 	}
124 }
