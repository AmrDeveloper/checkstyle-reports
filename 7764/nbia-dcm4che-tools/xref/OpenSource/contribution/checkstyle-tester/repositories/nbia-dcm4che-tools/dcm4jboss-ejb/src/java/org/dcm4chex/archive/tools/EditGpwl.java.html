68  public class EditGpwl {
69  
70      private static ResourceBundle messages =
71          ResourceBundle.getBundle(EditGpwl.class.getName());
72  
73      private final static LongOpt[] LONG_OPTS =
74          new LongOpt[] {
75              new LongOpt("url", LongOpt.REQUIRED_ARGUMENT, null, 'u'),
76              new LongOpt("help", LongOpt.NO_ARGUMENT, null, 'h'),
77              new LongOpt("version", LongOpt.NO_ARGUMENT, null, 'v'),
78              };
79  
80      private String ejbProviderURL = "jnp://localhost:1099";
81  
82      public static void main(String[] args) {
83           Getopt g = new Getopt("gpwlitem.jar", args, "af:r:u:hv", LONG_OPTS);
84  
85          EditGpwl gpwl = new EditGpwl();
86          int c;
87          int cmd = 0;
88          String iuid = null;
89          String fpath = null;
90          while ((c = g.getopt()) != -1) {
91              switch (c) {
92                  case 'a' :
93                      cmd = c;
94                      break;
95                  case 'r' :
96                      cmd = c;
97                      iuid = g.getOptarg();
98                      break;
99                  case 'f' :
100                     fpath = g.getOptarg();
101                     break;
102                 case 'u' :
103                     gpwl.ejbProviderURL = g.getOptarg();
104                     break;
105                 case 'v' :
106                     exit(messages.getString("version"), false);
107                 case 'h' :
108                     exit(messages.getString("usage"), false);
109                 case '?' :
110                     exit(null, true);
111                     break;
112             }
113         }
114         if (cmd == 0) {
115             exit(messages.getString("missing"), true);
116         }
117         try {
118             switch (cmd) {
119                 case 'a' :
120                     iuid = gpwl.add(loadGPWLItem(fpath));
121                     System.out.println(iuid);
122                     break;
123                 case 'r' :
124                     gpwl.remove(iuid);
125                     }
126         } catch (Throwable t) {
127             t.printStackTrace();
128             System.exit(1);
129         }
130     }
131 
132     private String add(Dataset ds) throws Exception {
133         GPWLManager mgr = getGPWLManager();
134         try {
135             return mgr.addWorklistItem(ds);
136         } finally {
137             try {
138                 mgr.remove();
139             } catch (Exception ignore) {
140             }
141         }
142     }
143 
144     private Dataset remove(String uid) throws Exception {
145         GPWLManager mgr = getGPWLManager();
146         try {
147             return mgr.removeWorklistItem(uid);
148         } finally {
149             try {
150                 mgr.remove();
151             } catch (Exception ignore) {
152             }
153         }
154     }
155 
156     private static Dataset loadGPWLItem(String fpath) throws SAXException, IOException {
157         if (isXML(fpath)) {
158             InputStream is = fpath == null ? System.in : new FileInputStream(fpath);
159             try {
160                 return DatasetUtils.fromXML(is);            
161             } finally {
162                 if (fpath != null) {
163                     try {
164                         is.close();
165                     } catch (IOException ignore) {
166                     }
167                 }
168             }
169         } else {
170             Dataset ds = DcmObjectFactory.getInstance().newDataset();
171             ds.readFile(new File(fpath), null, -1);
172             return ds;
173         }
174     }
175 
176     private static boolean isXML(String fpath) {
177         return fpath == null || fpath.toUpperCase().endsWith(".XML");
178     }
179 
180     private Hashtable makeEnv() {
181         Hashtable env = new Hashtable();
182         env.put(
183                 "java.naming.factory.initial",
184         "org.jnp.interfaces.NamingContextFactory");
185         env.put(
186                 "java.naming.factory.url.pkgs",
187         "org.jboss.naming:org.jnp.interfaces");
188         env.put("java.naming.provider.url", ejbProviderURL);
189         return env;
190     }
191 
192     private GPWLManager getGPWLManager() throws Exception {
193         Context ctx = new InitialContext(makeEnv());
194         GPWLManagerHome home = (GPWLManagerHome) ctx.lookup(GPWLManagerHome.JNDI_NAME);
195         ctx.close();
196         return home.create();
197     }
198     
199     private static void exit(String prompt, boolean error) {
200         if (prompt != null) {
201             System.err.println(prompt);
202         }
203         if (error) {
204             System.err.println(messages.getString("usage"));
205         }
206         System.exit(1);
207     }
