67  public class VerifyTar {
68  	private static Logger log = LoggerFactory.getLogger(VerifyTar.class);
69  
70      private static final int BUF_SIZE = 8192;
71      private static final String USAGE = "Usage: java -jar verifytar.jar [-p<num>] <file or directory path>[..]\n\n"
72              + " -p<num>  Strip the smallest prefix containing <num> leading slashes from each\n"
73              + "          file name prompted to stdout.";

75  
76      public static Map<String, byte[]> verify(File file, byte[] buf)
77              throws IOException, VerifyTarException {
78          FileInputStream in = new FileInputStream(file);
79          try {
80              return verify(in, file.toString(), buf);
81          } finally {
82              in.close();
83          }
84      }
85      
86      public static Map<String, byte[]> verify(InputStream in, String tarname, byte[] buf)
87      throws IOException, VerifyTarException {
88          return verify(in, tarname, buf, null);
89      }
90  
91      public static Map<String, byte[]> verify(InputStream in, String tarname, byte[] buf, ArrayList<String> objectNames)
92      throws IOException, VerifyTarException {
93          TarInputStream tar = new TarInputStream(in);
94          try {
95              log.debug("Verify tar file: {}",tarname);
96              TarEntry entry = tar.getNextEntry();
97              if (entry == null)
98                  throw new VerifyTarException("No entries in " + tarname);
99              String entryName = entry.getName();
100             if (!"MD5SUM".equals(entryName))
101                 throw new VerifyTarException("Missing MD5SUM entry in "
102                         + tarname);
103             BufferedReader dis = new BufferedReader(new InputStreamReader(tar));
104             
105             HashMap<String, byte[]> md5sums = new HashMap<String, byte[]>();
106             String line;
107             while ((line = dis.readLine()) != null) {
108                 char[] c = line.toCharArray();
109                 byte[] md5sum = new byte[16];
110                 for (int i = 0, j = 0; i < md5sum.length; i++, j++, j++) {
111                     md5sum[i] = (byte) ((fromHexDigit(c[j]) << 4)
112                             | fromHexDigit(c[j + 1]));
113                 }
114                 md5sums.put(line.substring(34), md5sum);
115             }
116             Map<String, byte[]> entries = new HashMap<String, byte[]>(md5sums.size());
117             entries.putAll(md5sums);
118             MessageDigest digest;
119             try {
120                 digest = MessageDigest.getInstance("MD5");
121             } catch (NoSuchAlgorithmException e) {
122                 throw new RuntimeException(e);
123             }
124             while ((entry = tar.getNextEntry()) != null) {
125                 entryName = entry.getName();
126                 log.debug("START: Check MD5 of entry: {}",entryName);
127                 if(objectNames != null && !objectNames.remove(entryName))
128                     throw new VerifyTarException( "TAR " + tarname + " contains entry: "
129                             + entryName + " not in file list");
130                 byte[] md5sum = (byte[]) md5sums.remove(entryName);
131                 if (md5sum == null)
132                     throw new VerifyTarException("Unexpected TAR entry: "
133                             + entryName + " in " + tarname);
134                 digest.reset();
135                 in = new DigestInputStream(tar, digest);
136                 while (in.read(buf) > 0)
137                     ;
138                 if (!Arrays.equals(digest.digest(), md5sum)) {
139                     throw new VerifyTarException(
140                             "Failed MD5 check of TAR entry: " + entryName
141                             + " in " + tarname);
142                 }
143                 log.debug("DONE: Check MD5 of entry: {}",entryName);
144             }
145             if (!md5sums.isEmpty())
146                 throw new VerifyTarException("Missing TAR entries: "
147                         + md5sums.keySet() + " in " + tarname);
148             if (objectNames != null && !objectNames.isEmpty())
149                 throw new VerifyTarException("Missing TAR entries from object list: "
150                         + objectNames.toString() + " in " + tarname);
151             return entries;
152         } finally {
153             tar.close();
154         }
155     }
156 
157     public static int fromHexDigit(char c) {
158         return c - ((c <= '9') ? '0' : (((c <= 'F') ? 'A' : 'a') - 10));
159     }
160 
161     public static void main(String[] args) {
162         if (args.length == 0) {
163             System.out.println(USAGE);
164             System.exit(-1);
165         }
166         int off = 0;
167         int strip = 0;
168         if (args[0].startsWith("-p")) {
169             try {
170                 strip = Integer.parseInt(args[0].substring(2));
171                 off = 1;
172             } catch (NumberFormatException e) {
173                 System.out.println(USAGE);
174                 System.exit(-1);
175             }
176         }
177         int errors = 0;
178         byte[] buf = new byte[BUF_SIZE];
179         for (int i = off; i < args.length; i++) {
180             try {
181                 errors += VerifyTar.verify(new File(args[i]), strip, buf);
182             } catch (FileNotFoundException e) {
183                 System.err.println(e.getMessage());
184                 System.exit(-2);
185             }
186         }
187         System.exit(errors);
188     }
189 
190     private static int verify(File file, int strip, byte[] buf) 
191             throws FileNotFoundException {
192         int errors = 0;
193         if (file.isDirectory()) {
194             String[] ss = file.list();
195             for (int i = 0; i < ss.length; i++) {
196                 errors += verify(new File(file, ss[i]), strip, buf);
197             }
198         } else {
199              String tarname = file.getPath();
200 
201             try {
202                 int pos = 0;
203                 while (strip-- > 0) {
204                     pos = tarname.indexOf(File.separatorChar, pos);
205                     if (pos == -1)
206                         break;
207                     pos++;
208                 }
209                 if (pos != -1) {
210                     log.info(tarname.substring(pos) + ' ');
211                 }
212                 verify(file, buf);
213                 log.info("ok");
214             } catch (Exception e) {
215                 errors = 1;
216                 log.error("Failed to create substring", e.getMessage());
217             }
218         }
219         return errors;
220     }
