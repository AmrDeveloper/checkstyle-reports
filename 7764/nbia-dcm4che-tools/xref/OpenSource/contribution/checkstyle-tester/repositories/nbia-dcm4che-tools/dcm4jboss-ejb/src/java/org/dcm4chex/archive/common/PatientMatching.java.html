92      public PatientMatching(String s) {
93          this(indexOf(s, PID), indexOf(s, ISSUER), s);
94      }

96      private PatientMatching(int pid, int issuer, String s) {
97          boolean trust = s.charAt(s.length()-1) == ']';
98          int alt = indexOfAlt(s);
99          if (alt != -1) {
100             altDemographicsMatch = 
101                     new PatientMatching(pid, issuer, s.substring(alt+1));
102             s = s.substring(0, alt);
103         }
104         int familyName = indexOf(s, FAMILYNAME);
105         int givenName = indexOf(s, GIVENNAME);
106         int middleName = indexOf(s, MIDDLENAME);
107         int namePrefix = indexOf(s, NAMEPREFIX);
108         int nameSuffix = indexOf(s, NAMESUFFIX);
109         int birthdate = indexOf(s, BIRTHDATE);
110         int sex = indexOf(s, SEX);
111         if (pid == -1 || issuer == -1
112                 || initialMatch(s, pid, PID)
113                 || initialMatch(s, issuer, ISSUER)
114                 || initialMatch(s, birthdate, BIRTHDATE)
115                 || initialMatch(s, sex, SEX)) {
116             throw new IllegalArgumentException(s);
117         }
118         familyNameMustMatch = familyName != -1;
119         givenNameMustMatch = givenName != -1;
120         middleNameMustMatch = middleName != -1;
121         namePrefixMustMatch = namePrefix != -1;
122         nameSuffixMustMatch = nameSuffix != -1;
123         birthDateMustMatch = birthdate != -1;
124         sexMustMatch = sex != -1;
125         unknownPatientIDAlwaysMatch = unknownAlwaysMatch(s, pid, PID, false);
126         unknownIssuerAlwaysMatch = unknownAlwaysMatch(s, issuer, ISSUER, false);
127         familyNameInitialMatch = initialMatch(s, familyName, FAMILYNAME);
128         unknownFamilyNameAlwaysMatch = 
129                 unknownAlwaysMatch(s, familyName, FAMILYNAME, familyNameInitialMatch);
130         givenNameInitialMatch = initialMatch(s, givenName, GIVENNAME);
131         unknownGivenNameAlwaysMatch =
132                 unknownAlwaysMatch(s, givenName, GIVENNAME, givenNameInitialMatch);
133         middleNameInitialMatch = initialMatch(s, middleName, MIDDLENAME);
134         unknownMiddleNameAlwaysMatch =
135                 unknownAlwaysMatch(s, middleName, MIDDLENAME, middleNameInitialMatch);
136         namePrefixInitialMatch = initialMatch(s, namePrefix, NAMEPREFIX);
137         unknownNamePrefixAlwaysMatch =
138                 unknownAlwaysMatch(s, namePrefix, NAMEPREFIX, namePrefixInitialMatch);
139         nameSuffixInitialMatch = initialMatch(s, nameSuffix, NAMESUFFIX);
140         unknownNameSuffixAlwaysMatch =
141                 unknownAlwaysMatch(s, nameSuffix, NAMESUFFIX, nameSuffixInitialMatch);
142         unknownBirthDateAlwaysMatch =
143                 unknownAlwaysMatch(s, birthdate, BIRTHDATE, false);
144         unknownSexAlwaysMatch =
145                 unknownAlwaysMatch(s, sex, SEX, false);
146         ignore = ignorePattern(s);
147         trustPatientIDWithIssuer = trust
148                 || !familyNameMustMatch
149                 && !givenNameMustMatch && !middleNameMustMatch
150                 && !namePrefixMustMatch && !nameSuffixMustMatch
151                 && !birthDateMustMatch && !sexMustMatch;
152         if (unknownPatientIDAlwaysMatch && !familyNameMustMatch) {
153             throw new IllegalArgumentException(s);
154         }
155     }
156 
157     private int indexOfAlt(String s) {
158         int ignore = s.indexOf(IGNORE);
159         int alt = -1;
160         while ((alt = s.indexOf('|', alt+1)) != -1)
161             // check if '|' not part of ignore("<regex>")
162             if (ignore == -1 || ignore > alt
163                     || s.indexOf("\")", ignore+2) < alt)
164                 return alt;
165         return -1;
166     }

184     private boolean isUnknownPersonNameAlwaysMatch() {
185         return unknownFamilyNameAlwaysMatch && unknownGivenNameAlwaysMatch
186                 && unknownMiddleNameAlwaysMatch && unknownNamePrefixAlwaysMatch
187                 && unknownNameSuffixAlwaysMatch;
188     }
189 
190     private boolean initialMatch(String s, int index, String substr) {
191         return index != -1 && s.startsWith(INITIAL, index + substr.length());
192     }
193 
194     private boolean unknownAlwaysMatch(String s, int index, String substr,
195             boolean initialMatch) {
196         if (index == -1)
197             return true;
198         int after = index + substr.length();
199         if (initialMatch)
200             after += INITIAL.length();
201         return after < s.length() && s.charAt(after) == '?';
202     }
203 
204     private Pattern ignorePattern(String s) {
205         int index = s.indexOf(IGNORE);
206         if (index == -1)
207             return null;
208         int after = index + IGNORE.length();
209         int begin = after+2;
210         int end;
211         if (!s.startsWith("(\"", after)
212                 || (end = s.indexOf("\")", begin)) == -1)
213                 throw new IllegalArgumentException(s);
214         String regex = s.substring(begin, end);
215         return Pattern.compile(regex);
216     }
217 
218     private static int indexOf(String str, String substr) {
219         int index = str.indexOf(substr);
220         if (index != -1) {
221             int after;
222             if (index > 0 
223                     && " ,[".indexOf(str.charAt(index-1)) == -1
224                     || (after = index + substr.length()) < str.length() 
225                     && " ,]?(".indexOf(str.charAt(after)) == -1) {
226                 throw new IllegalArgumentException(str);
227             }
228         }
229         return index;
230     }

236     public StringBuilder toStringBuilder(StringBuilder sb) {
237         sb.append(PID);
238         if (unknownPatientIDAlwaysMatch) {
239             sb.append('?');
240         }
241         sb.append(',').append(ISSUER);
242         if (unknownIssuerAlwaysMatch) {
243             sb.append('?');
244         }
245         if (familyNameMustMatch || givenNameMustMatch || middleNameMustMatch
246                 || namePrefixMustMatch || nameSuffixMustMatch 
247                 || birthDateMustMatch || sexMustMatch) {
248             sb.append(',');
249             if (trustPatientIDWithIssuer) {
250                 sb.append('[');
251             }
252             appendDemographicsMatch(sb);
253             for (PatientMatching m = altDemographicsMatch; m != null;
254                     m = m.altDemographicsMatch)
255                 m.appendDemographicsMatch(sb.append('|'));
256             if (trustPatientIDWithIssuer) {
257                 sb.append(']');
258             }
259         }
260         return sb;
261     }
262 
263     private void appendDemographicsMatch(StringBuilder sb) {
264         int count = 0;
265         if (ignore != null) {
266             count++;
267             sb.append(IGNORE).append("(\"").append(ignore).append("\")");
268         }
269         if (familyNameMustMatch) {
270             if (count++ > 0) {
271                 sb.append(',');
272             }
273             sb.append(FAMILYNAME);
274             if (familyNameInitialMatch) {
275                 sb.append(INITIAL);
276             }
277             if (unknownFamilyNameAlwaysMatch) {
278                 sb.append('?');
279             }
280         }
281         if (givenNameMustMatch) {
282             if (count++ > 0) {
283                 sb.append(',');
284             }
285             sb.append(GIVENNAME);
286             if (givenNameInitialMatch) {
287                 sb.append(INITIAL);
288             }
289             if (unknownGivenNameAlwaysMatch) {
290                 sb.append('?');
291             }
292         }
293         if (middleNameMustMatch) {
294             if (count++ > 0) {
295                 sb.append(',');
296             }
297             sb.append(MIDDLENAME);
298             if (middleNameInitialMatch) {
299                 sb.append(INITIAL);
300             }
301             if (unknownMiddleNameAlwaysMatch) {
302                 sb.append('?');
303             }
304         }
305         if (namePrefixMustMatch) {
306             if (count++ > 0) {
307                 sb.append(',');
308             }
309             sb.append(NAMEPREFIX);
310             if (namePrefixInitialMatch) {
311                 sb.append(INITIAL);
312             }
313             if (unknownNamePrefixAlwaysMatch) {
314                 sb.append('?');
315             }
316         }
317         if (nameSuffixMustMatch) {
318             if (count++ > 0) {
319                 sb.append(',');
320             }
321             sb.append(NAMESUFFIX);
322             if (nameSuffixInitialMatch) {
323                 sb.append(INITIAL);
324             }
325             if (unknownNameSuffixAlwaysMatch) {
326                 sb.append('?');
327             }
328         }
329         if (birthDateMustMatch) {
330             if (count > 0) {
331                 sb.append(',');
332             }
333             sb.append(BIRTHDATE);
334             if (unknownBirthDateAlwaysMatch) {
335                 sb.append('?');
336             }
337         }
338         if (sexMustMatch) {
339             if (count > 0) {
340                 sb.append(',');
341             }
342             sb.append(SEX);
343             if (unknownSexAlwaysMatch) {
344                 sb.append('?');
345             }
346         }
347     }

349     public List<Pattern> compilePNPatterns(String familyName, String givenName,
350             String middleName, String namePrefix, String nameSuffix) {
351         List<Pattern> list = new LinkedList<Pattern>();
352         for (PatientMatching m = this; m != null; m = m.altDemographicsMatch )
353             list.add(m.compilePNPattern(familyName, givenName, middleName,
354                     namePrefix, nameSuffix));
355         return list ;
356     }
357 
358     public Pattern compilePNPattern(String familyName, String givenName,
359             String middleName, String namePrefix, String nameSuffix) {
360         if (allMatchesFor(familyName, givenName, middleName, namePrefix,
361                 nameSuffix)) {
362             return null;
363         }
364         boolean appendNameSuffix =
365                 nameSuffixMustMatch && nameSuffix != null;
366         boolean appendNamePrefix = appendNameSuffix
367                 || namePrefixMustMatch && namePrefix != null;
368         boolean appendMiddleName = appendNamePrefix
369                 || middleNameMustMatch && middleName != null;
370         boolean appendGivenName = appendMiddleName
371                 || givenNameMustMatch && givenName != null;
372         boolean appendFamilyName = appendGivenName
373                 || familyNameMustMatch && familyName != null;
374         StringBuilder regex = new StringBuilder();
375         if (appendFamilyName) {
376             appendRegex(regex, ignore(familyName), familyNameMustMatch,
377                     familyNameInitialMatch, unknownFamilyNameAlwaysMatch);
378             regex.append("\\^");
379             if (appendGivenName) {
380                 appendRegex(regex, ignore(givenName), givenNameMustMatch,
381                         givenNameInitialMatch, unknownGivenNameAlwaysMatch);
382                 regex.append("\\^");
383                 if (appendMiddleName) {
384                     appendRegex(regex, ignore(middleName), middleNameMustMatch,
385                             middleNameInitialMatch, unknownMiddleNameAlwaysMatch);
386                     regex.append("\\^");
387                     if (appendNamePrefix) {
388                         appendRegex(regex, ignore(namePrefix), namePrefixMustMatch,
389                                 namePrefixInitialMatch, unknownNamePrefixAlwaysMatch);
390                         regex.append("\\^");
391                         if (appendNameSuffix) {
392                             appendRegex(regex, ignore(nameSuffix), nameSuffixMustMatch,
393                                     nameSuffixInitialMatch, unknownNameSuffixAlwaysMatch);
394                         }
395                     }
396                 }
397             }
398         }
399         if (!appendNameSuffix)
400             regex.append(".*");
401         return Pattern.compile(regex.toString());
402     }
403 
404     public String ignore(String s) {
405         return (ignore == null ||  s == null) ? s
406                 : ignore.matcher(s).replaceAll("");
407     }
408 
409     private static void appendRegex(StringBuilder regex, String value,
410             boolean mustMatch, boolean initialMatch,
411             boolean unknownAlwaysMatch) {
412         if (!mustMatch || value == null || value.length() == 0) {
413             regex.append("[^\\^]*");
414         } else if (initialMatch) {
415             regex.append(unknownAlwaysMatch ? "(\\Q" : "\\Q")
416                 .append(value.charAt(0))
417                 .append(unknownAlwaysMatch ? "\\E[^\\^]*)?" : "\\E[^\\^]*");
418         } else {
419             regex.append(unknownAlwaysMatch ? "(\\Q" : "\\Q")
420                  .append(value)
421                  .append(unknownAlwaysMatch ? "\\E)?" : "\\E");
422         }
423     }

425     public boolean noMatchesFor(String pid, String issuer, String familyName,
426             String givenName, String middleName, String namePrefix,
427             String nameSuffix, String birthdate, String sex) {
428         return !unknownPatientIDAlwaysMatch && pid == null
429                 || !unknownIssuerAlwaysMatch && issuer == null
430                 || !(trustPatientIDWithIssuer && pid != null && issuer != null)
431                 && noMatchesFor(familyName, givenName, middleName, namePrefix,
432                         nameSuffix, birthdate, sex);
433     }
434 
435     public boolean noMatchesFor(String familyName, String givenName,
436             String middleName, String namePrefix, String nameSuffix,
437             String birthdate, String sex) {
438         return (!unknownFamilyNameAlwaysMatch && familyName == null
439                 || !unknownGivenNameAlwaysMatch && givenName == null
440                 || !unknownMiddleNameAlwaysMatch && middleName == null
441                 || !unknownNamePrefixAlwaysMatch && namePrefix == null
442                 || !unknownNameSuffixAlwaysMatch && nameSuffix == null
443                 || !unknownBirthDateAlwaysMatch && birthdate == null
444                 || !unknownSexAlwaysMatch && sex == null)
445                 && (altDemographicsMatch == null 
446                 || altDemographicsMatch.noMatchesFor(familyName, givenName,
447                         middleName, namePrefix, nameSuffix, birthdate, sex));
448     }
449 
450     private boolean allMatchesFor(String familyName, String givenName,
451             String middleName, String namePrefix, String nameSuffix) {
452         return (!familyNameMustMatch 
453                         || unknownFamilyNameAlwaysMatch && familyName == null)
454             && (!givenNameMustMatch
455                         || unknownGivenNameAlwaysMatch && givenName == null)
456             && (!middleNameMustMatch
457                         || unknownMiddleNameAlwaysMatch && middleName == null)
458             && (!namePrefixMustMatch
459                         || unknownNamePrefixAlwaysMatch && namePrefix == null)
460             && (!nameSuffixMustMatch
461                         || unknownNameSuffixAlwaysMatch && nameSuffix == null);
462     }
463 
464     public boolean allMatchesFor(String familyName, String givenName,
465             String middleName, String namePrefix, String nameSuffix,
466             String birthdate, String sex) {
467         return allMatchesFor(familyName, givenName, middleName, namePrefix, nameSuffix)
468                 && (!birthDateMustMatch
469                         || unknownBirthDateAlwaysMatch && birthdate == null)
470                 && (!sexMustMatch
471                         || unknownSexAlwaysMatch && sex == null)
472                 || altDemographicsMatch != null 
473                 && altDemographicsMatch.allMatchesFor(familyName, givenName,
474                         middleName, namePrefix, nameSuffix, birthdate, sex);
475     }

477     public boolean matches(String pn, String birthdate, String sex,
478             Iterator<Pattern> pnPatternIter, String birthdate2, String sex2) {
479         Pattern pnPattern = pnPatternIter.next();
480         return (pnPattern == null 
481                         || (pn == null 
482                                 ? isUnknownPersonNameAlwaysMatch()
483                                 : pnPattern.matcher(ignore(pn)).matches()))
484                 && (!birthDateMustMatch 
485                         || (birthdate == null || birthdate2 == null)
486                                 ? unknownBirthDateAlwaysMatch
487                                 : birthdate.equals(birthdate2))
488                 && (!sexMustMatch 
489                         || ((sex == null || sex2 == null)
490                                 ? unknownSexAlwaysMatch
491                                 : sex.equals(sex2)))
492                 || altDemographicsMatch != null
493                 && altDemographicsMatch.matches(pn, birthdate, sex,
494                         pnPatternIter, birthdate2, sex2);
495     }
