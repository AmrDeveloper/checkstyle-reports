39  package org.dcm4chex.archive.tools;
40  
41  import java.security.MessageDigest;
42  
43  import org.dcm4che.util.Base64;
44  
45  /**
46   * @author Gunter Zeilinger <gunterze@gmail.com>
47   * @version $Id: Pwd2Hash.java 3122 2007-02-15 17:35:26Z gunterze $
48   * @since Feb 15, 2007
49   */
50  public class Pwd2Hash {
51  
52      public static void main(String[] args) {
53          if ( args.length < 1 ) {
54              System.out.println("USAGE: Pwd2Hash <readable password>");
55          } else {
56              try {
57                  String hash = Base64.byteArrayToBase64(createHash(args[0]));
58                  System.out.println("Hashed password:"+hash);
59              } catch (Exception x) {
60                  System.out.println("Cant create SHA1 hash"+x.getMessage());
61                  x.printStackTrace();
62              }
63          }
64      }
65  
66      public static final byte[] createHash(String value) throws Exception
67      {
68             final MessageDigest digest = MessageDigest.getInstance("SHA");
69             byte[] hashBytes = digest.digest((value).getBytes("UTF-8"));
70             return hashBytes;
71      }
72      
73  }
