92  /**
93   * Diagnostic tool that turns a {@code JCTree} (javac) based tree into a hierarchical dump that should make
94   * it easy to analyse the exact structure of the AST.
95   */
96  public class JcTreePrinter {
97  	private final StringBuilder output = new StringBuilder();
98  	private final boolean includePositions;
99  	private final boolean includeObjectRefs;
100 	private int indent;
101 	private String rel;
102 	private Map<JCTree, Integer> endPosTable;
103 	private boolean modsOfEnum;
104 	private final Map<Object, Integer> visited = new MapMaker().weakKeys().makeMap();
105 	private int objectCounter = 0;
106 	
107 	private static final Method GET_TAG_METHOD;
108 	private static final Field TAG_FIELD;
109 	
110 	//TODO Adopt the reflective printer principle used in the EcjTreePrinter. For example, we don't currently know if the type ref is shared or unique amongst
111 	// JCVarDecls that all come from the same line: int[] x, y;
112 	
113 	static {
114 		Method m = null;
115 		Field f = null;
116 		try {
117 			m = JCTree.class.getDeclaredMethod("getTag");
118 		} catch (NoSuchMethodException e) {
119 			try {
120 				f = JCTree.class.getDeclaredField("tag");
121 			} catch (NoSuchFieldException e1) {
122 				e1.printStackTrace();
123 			}
124 		}
125 		GET_TAG_METHOD = m;
126 		TAG_FIELD = f;
127 	}
128 
129 	static int getTag(JCTree tree) {
130 		if (GET_TAG_METHOD != null) {
131 			try {
132 				return (Integer) GET_TAG_METHOD.invoke(tree);
133 			} catch (IllegalAccessException e) {
134 				throw new RuntimeException(e);
135 			} catch (InvocationTargetException e) {
136 				throw new RuntimeException(e.getCause());
137 			}
138 		}
139 		try {
140 			return TAG_FIELD.getInt(tree);
141 		} catch (IllegalAccessException e) {
142 			throw new RuntimeException(e);
143 		}
144 	}

164 	public static void main(String[] args) throws IOException {
165 		if (args.length == 0) {
166 			System.out.println("Usage: Supply a file name to print.");
167 			return;
168 		}
169 		Context context = new Context();
170 		
171 		Options.instance(context).put(OptionName.ENCODING, "UTF-8");
172 		
173 		JavaCompiler compiler = new JavaCompiler(context);
174 		compiler.genEndPos = true;
175 		compiler.keepComments = true;
176 		
177 		@SuppressWarnings("deprecation") JCCompilationUnit cu = compiler.parse(args[0]);
178 		JcTreePrinter printer = new JcTreePrinter(true);
179 		printer.visit(cu);
180 		System.out.println(printer);
181 	}

183 	private void printNode(JCTree tree) {
184 		if (tree == null) {
185 			printNode("NULL");
186 		} else {
187 			String suffix = "";
188 			int objId;
189 			Integer backRef = visited.get(tree);
190 			if (backRef == null) {
191 				objId = ++objectCounter;
192 				visited.put(tree, objId);
193 			} else {
194 				objId = backRef;
195 			}
196 			
197 			if (includePositions) {
198 				Integer endPos_ = null;
199 				if (endPosTable != null) endPos_ = endPosTable.get(tree);
200 				int endPos = endPos_ == null ? tree.getEndPosition(endPosTable) : endPos_;
201 				int startPos = tree.pos;
202 				if (tree instanceof JCTypeApply || tree instanceof JCWildcard || tree instanceof JCTypeParameter) {
203 					// Javac itself actually has bugs in generating the right endpos. To make sure our tests that compare end pos don't fail,
204 					// as we do set the end pos at the right place, we overwrite it with magic value -2 which means: javac screws this up.
205 					endPos = -2;
206 				}
207 				
208 				// When there are no modifiers but an internal flag is set, javac screws up,
209 				// and sets start to the beginning of the node, and end to the last non-whitespace thing before it;
210 				// yes - that would make for negatively sized modifier nodes.
211 				if (tree instanceof JCModifiers && endPos-tree.pos <= 0) {
212 					startPos = -1;
213 					endPos = -1;
214 				}
215 				
216 				// The end value of an annotation is end-of-annotation-type in older javac, and end-of-parenslist in new javac.
217 				if (tree instanceof JCAnnotation || tree instanceof JCModifiers) {
218 					endPos = -2;
219 				}
220 				
221 				// Modifiers of enums never get their position set, but we do set the position.
222 				if (modsOfEnum) {
223 					startPos = -1;
224 					endPos = -1;
225 					modsOfEnum = false;
226 				}
227 				
228 				//In rare conditions, the end pos table is filled with a silly value, but start is -1.
229 				if (startPos == -1 && endPos >= 0) endPos = -1;
230 				
231 				/*Javac bug: sometimes the startpos of a binary expression is set to the wrong operator.*/ {
232 					if (tree instanceof JCBinary && ((JCBinary)tree).rhs instanceof JCBinary) {
233 						if (getTag(tree) != getTag(((JCBinary)tree).rhs)) {
234 							startPos = -2;
235 						}
236 					}
237 					
238 					if (tree instanceof JCBinary && ((JCBinary)tree).rhs instanceof JCInstanceOf) {
239 						startPos = -2;
240 					}
241 				}
242 				
243 				/* Javac bug: The end position of a "super.FOO" node which is itself the LHS of a select-like
244 				   concept (Select, Method Invocation, etcetera) are set to right after the dot following
245 				   it. This doesn't happen with 'this' or anything other than super. */ {
246 					if (tree instanceof JCMethodInvocation) {
247 						JCMethodInvocation invoke = (JCMethodInvocation) tree;
248 						if (invoke.meth instanceof JCFieldAccess && ((JCFieldAccess) invoke.meth).selected instanceof JCIdent) {
249 							JCIdent selected = (JCIdent) ((JCFieldAccess) invoke.meth).selected;
250 							if (selected.name.toString().equals("super")) endPos = -2;
251 						}
252 					}
253 					
254 					if (tree instanceof JCFieldAccess && ((JCFieldAccess) tree).selected instanceof JCIdent) {
255 						JCIdent selected = (JCIdent) ((JCFieldAccess) tree).selected;
256 						if (selected.name.toString().equals("super")) endPos = -2;
257 					}
258 				}
259 				
260 				/* Javac bug: the 'JCAssign' starts at a dot (if present) in the expression instead of at the start of it, which is weird
261 				   and inconsistent. */ {
262 					if (tree instanceof JCAssign && ((JCAssign) tree).rhs instanceof JCFieldAccess) {
263 						startPos = -2;
264 					}
265 				}
266 				suffix += String.format("(%d-%d)", startPos, endPos);
267 			}
268 			
269 			if (includeObjectRefs) {
270 				suffix += String.format("(id: %d%s)", objId, backRef != null ? " BACKREF" : "");
271 			}
272 			
273 			printNode(String.format("%s%s", tree.getClass().getSimpleName(), suffix));
274 		}
275 	}

277 	private void printNode(String nodeKind) {
278 		printIndent();
279 		if (rel != null)
280 			output.append(rel).append(": ");
281 		rel = null;
282 		output.append("[").append(nodeKind).append("]\n");
283 		indent++;
284 	}
285 	
286 	private void printIndent() {
287 		for (int i = 0; i < indent; i++) {
288 			output.append("\t");
289 		}
290 	}
291 	
292 	private void property(String rel, Object val) {
293 		printIndent();
294 		if (rel != null)
295 			output.append(rel).append(": ");
296 		if (val instanceof JCTree)
297 			output.append("!!JCTree-AS-PROP!!");
298 		if (val == null) {
299 			output.append("[NULL]\n");
300 		} else {
301 			String content;
302 			if (val instanceof String) {
303 				content = new StringLiteral().astValue(val.toString()).rawValue();
304 			} else {
305 				content = String.valueOf(val);
306 			}
307 			output.append("[").append(val.getClass().getSimpleName()).append(" ").append(content).append("]\n");
308 		}
309 	}

311 	private void child(String rel, JCTree node) {
312 		this.rel = rel;
313 		if (node != null) {
314 			node.accept(visitor);
315 		} else {
316 			printNode("NULL");
317 			indent--;
318 		}
319 	}
320 	
321 	private void children(String rel, List<? extends JCTree> nodes) {
322 		this.rel = rel;
323 		
324 		if (nodes == null) {
325 			;
326 			printNode("LISTNULL");
327 			indent--;
328 		} else if (nodes.isEmpty()) {
329 			printNode("LISTEMPTY");
330 			indent--;
331 		} else {
332 			int i = 0;
333 			for (JCTree node : nodes) {
334 				child(String.format("%s[%d]", rel, i++), node);
335 			}
336 		}
337 	}

339 	public void visit(JCTree tree) {
340 		tree.accept(visitor);
341 	}

343 	private final JCTree.Visitor visitor = new JCTree.Visitor() {
344 		@Override public void visitTopLevel(JCCompilationUnit tree) {
345 			printNode(tree);
346 			endPosTable = tree.endPositions;
347 			child("pid", tree.pid);
348 			children("defs", tree.defs);
349 			indent--;
350 		}
351 		
352 		@Override public void visitImport(JCImport tree) {
353 			printNode(tree);
354 			property("staticImport", tree.staticImport);
355 			child("qualid", tree.qualid);
356 			indent--;
357 		}
358 		
359 		@Override public void visitClassDef(JCClassDecl tree) {
360 			printNode(tree);
361 			modsOfEnum = (tree.mods != null && (tree.mods.flags & Flags.ENUM) != 0);
362 			child("mods", tree.mods);
363 			property("name", tree.name);
364 			children("typarams", tree.typarams);
365 			child("extends", tree.extending);
366 			children("implementing", tree.implementing);
367 			children("defs", tree.defs);
368 			indent--;
369 		}
370 		
371 		@Override public void visitMethodDef(JCMethodDecl tree) {
372 			printNode(tree);
373 			property("name", tree.name);
374 			child("mods", tree.mods);
375 			children("typarams", tree.typarams);
376 			children("params", tree.params);
377 			children("thrown", tree.thrown);
378 			child("default", tree.defaultValue);
379 			child("body", tree.body);
380 			indent--;
381 		}
382 		
383 		@Override public void visitVarDef(JCVariableDecl tree) {
384 			printNode(tree);
385 			child("mods", tree.mods);
386 			child("vartype", tree.vartype);
387 			property("name", tree.name);
388 			child("init", tree.init);
389 			indent--;
390 		}
391 		
392 		@Override public void visitSkip(JCSkip tree) {
393 			printNode(tree);
394 			indent--;
395 		}
396 		
397 		@Override public void visitBlock(JCBlock tree) {
398 			printNode(tree);
399 			property("flags", "0x" + Long.toString(tree.flags, 0x10));
400 			children("stats", tree.stats);
401 			indent--;
402 		}

404 		@Override public void visitDoLoop(JCDoWhileLoop tree) {
405 			printNode(tree);
406 			child("body", tree.body);
407 			child("cond", tree.cond);
408 			indent--;
409 		}

416 		}

570 		public String operatorName(int tag) {
571 			switch (tag) {
572 			case JCTree.POS:
573 				return "+";
574 			case JCTree.NEG:
575 				return "-";
576 			case JCTree.NOT:
577 				return "!";
578 			case JCTree.COMPL:
579 				return "~";
580 			case JCTree.PREINC:
581 				return "++X";
582 			case JCTree.PREDEC:
583 				return "--X";
584 			case JCTree.POSTINC:
585 				return "X++";
586 			case JCTree.POSTDEC:
587 				return "X--";
588 			case JCTree.NULLCHK:
589 				return "<*nullchk*>";
590 			case JCTree.OR:
591 				return "||";
592 			case JCTree.AND:
593 				return "&&";
594 			case JCTree.EQ:
595 				return "==";
596 			case JCTree.NE:
597 				return "!=";
598 			case JCTree.LT:
599 				return "<";
600 			case JCTree.GT:
601 				return ">";

776 		@Override public void visitWildcard(JCWildcard tree) {
777 			printNode(tree);
778 			Object o;
779 			
780 			// In some javacs (older ones), JCWildcard.kind is a BoundKind, which is an enum. In newer ones its a TypeBoundKind which is a JCTree, i.e. has positions.
781 			try {
782 				o = tree.getClass().getField("kind").get(tree);
783 			} catch (Exception e) {
784 				throw new RuntimeException("There's no field at all named 'kind' in JCWildcard? This is not a javac I understand.", e);
785 			}
786 			
787 			if (o instanceof JCTree) {
788 				child("kind", (JCTree)o);
789 			} else if (o instanceof BoundKind) {
790 				property("kind", String.valueOf(o));
791 			}
792 			child("inner", tree.inner);
793 			indent--;
794 		}
