3   /**
4    * 
5    * Mediator encapsulates how a set of objects (PartyMember) interact. Instead of
6    * referring to each other directly they use a mediator (Party) interface.
7    * 
8    */
9   public class App {
10  
11  	public static void main(String[] args) {
12  		
13  		// create party and members
14  		Party party = new PartyImpl();
15  		Hobbit hobbit = new Hobbit();
16  		Wizard wizard = new Wizard();
17  		Rogue rogue = new Rogue();
18  		Hunter hunter = new Hunter();
19  
20  		// add party members
21  		party.addMember(hobbit);
22  		party.addMember(wizard);
23  		party.addMember(rogue);
24  		party.addMember(hunter);
25  
26  		// perform actions -> the other party members
27  		// are notified by the party
28  		hobbit.act(Action.ENEMY);
29  		wizard.act(Action.TALE);
30  		rogue.act(Action.GOLD);
31  		hunter.act(Action.HUNT);
32  	}
33  }
