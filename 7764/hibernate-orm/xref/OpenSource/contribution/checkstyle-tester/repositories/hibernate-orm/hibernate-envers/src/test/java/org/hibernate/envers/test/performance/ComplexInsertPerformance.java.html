25  package org.hibernate.envers.test.performance;
26  
27  import java.io.IOException;
28  import java.util.Date;
29  import java.util.HashSet;
30  import javax.persistence.EntityManager;
31  
32  import org.junit.Ignore;
33  
34  import org.hibernate.ejb.Ejb3Configuration;
35  import org.hibernate.envers.test.performance.complex.ChildEntity1;
36  import org.hibernate.envers.test.performance.complex.ChildEntity2;
37  import org.hibernate.envers.test.performance.complex.RootEntity;
38  
39  /**
40   * @author Adam Warski (adam at warski dot org)
41   */
42  @Ignore
43  public class ComplexInsertPerformance extends AbstractPerformanceTest {
44      public void configure(Ejb3Configuration cfg) {
45          cfg.addAnnotatedClass(RootEntity.class);
46          cfg.addAnnotatedClass(ChildEntity1.class);
47          cfg.addAnnotatedClass(ChildEntity2.class);
48      }
49  
50      private final static int NUMBER_INSERTS = 1000;
51  
52      private long idCounter = 0;
53  
54      private ChildEntity2 createChildEntity2() {
55          ChildEntity2 ce = new ChildEntity2();
56          ce.setId(idCounter++);
57          ce.setNumber(12345678);
58          ce.setData("some data, not really meaningful");
59          ce.setStrings(new HashSet<String>());
60          ce.getStrings().add("aaa");
61          ce.getStrings().add("bbb");
62          ce.getStrings().add("ccc");
63  
64          return ce;
65      }
66  
67      private ChildEntity1 createChildEntity1() {
68          ChildEntity1 ce = new ChildEntity1();
69          ce.setId(idCounter++);
70          ce.setData1("xxx");
71          ce.setData2("yyy");
72          ce.setChild1(createChildEntity2());
73          ce.setChild2(createChildEntity2());
74  
75          return ce;
76      }
77  
78      protected void doTest() {
79          for (int i=0; i<NUMBER_INSERTS; i++) {
80              newEntityManager();
81              EntityManager entityManager = getEntityManager();
82  
83              entityManager.getTransaction().begin();
84  
85              RootEntity re = new RootEntity();
86              re.setId(idCounter++);
87              re.setData1("data1");
88              re.setData2("data2");
89              re.setDate1(new Date());
90              re.setNumber1(123);
91              re.setNumber2(456);
92              re.setChild1(createChildEntity1());
93              re.setChild2(createChildEntity1());
94              re.setChild3(createChildEntity1());
95  
96              start();
97              entityManager.persist(re);            
98              entityManager.getTransaction().commit();
99              stop();
100         }
101     }
102 
103     public static void main(String[] args) throws IOException {
104         ComplexInsertPerformance insertsPerformance = new ComplexInsertPerformance();
105         insertsPerformance.test(3);
106     }
107 }
