51      private String convertString(String s, String def) {
52          if ("NULL".equals(s)) { return null; }
53          if ("".equals(s)) { return def; }
54          return s;
55      }

57      private int convertStringToInteger(String s, int def) {
58          if ("".equals(s)) { return def; }
59          try {
60              return Integer.parseInt(s);
61          } catch (NumberFormatException e) {
62              System.err.println("Invalid number, returning 0.");
63              return 0;
64          }
65      }

67      private void printPerson(StringBuilder sb, Person p) {
68          sb.append("id = ").append(p.getId()).append(", name = ").append(p.getName())
69                  .append(", surname = ").append(p.getSurname());
70  
71          Address a = p.getAddress();
72          if (a != null) {
73              sb.append(", address = <").append(a.getId()).append("> ").append(a.getStreetName()).append(" ")
74                      .append(a.getHouseNumber()).append("/").append(a.getFlatNumber());
75          }
76      }

78      @SuppressWarnings({"unchecked"})
79      private void printPersons(StringBuilder sb) {
80          List<Person> persons = entityManager.createQuery(
81                  "select p from Person p order by p.id").getResultList();
82  
83          sb.append("Persons:\n");
84          for (Person p : persons) {
85              printPerson(sb, p);
86              sb.append("\n");
87          }
88      }

90      private void printPersonHistory(StringBuilder sb, int personId) {
91          AuditReader reader = AuditReaderFactory.get(entityManager);
92  
93          List personHistory = reader.createQuery()
94                  .forRevisionsOfEntity(Person.class, false, true)
95                  .add(AuditEntity.id().eq(personId))
96                  .getResultList();
97  
98          if (personHistory.size() == 0) {
99              sb.append("A person with id ").append(personId).append(" does not exist.\n");
100         } else {
101             for (Object historyObj : personHistory) {
102                 Object[] history = (Object[]) historyObj;
103                 DefaultRevisionEntity revision = (DefaultRevisionEntity) history[1];
104                 sb.append("revision = ").append(revision.getId()).append(", ");
105                 printPerson(sb, (Person) history[0]);
106                 sb.append(" (").append(revision.getRevisionDate()).append(")\n");
107             }
108         }
109     }

122     private void readAndSetAddress(Scanner scanner, Person p) {
123         Address old = p.getAddress();
124 
125         String input = scanner.nextLine();
126         if ("NULL".equals(input)) {
127             p.setAddress(null);
128             if (old != null) {
129                 old.getPersons().remove(p);
130             }
131         } else if ("".equals(input)) {
132         } else {
133             try {
134                 Integer id = Integer.valueOf(input);
135 
136                 Address a = entityManager.find(Address.class, id);
137 
138                 if (a == null) {
139                     System.err.println("Unknown address id, setting to NULL.");
140                     p.setAddress(null);
141                     if (old != null) {
142                         old.getPersons().remove(p);
143                     }
144                 } else {
145                     p.setAddress(a);
146 
147                     a.getPersons().add(p);
148 
149                     if (old != null) {
150                         old.getPersons().remove(p);
151                     }
152                 }
153             } catch (NumberFormatException e) {
154                 System.err.println("Invalid address id, setting to NULL.");
155                 p.setAddress(null);
156                 if (old != null) {
157                     old.getPersons().remove(p);
158                 }
159             }
160         }
161     }

164         Person p = new Person();

166         out.print("Person name (NULL for null): ");

178     private void readModifyPerson(PrintStream out, Scanner scanner, int personId) {
179         Person current = entityManager.find(Person.class, personId);
180 
181         if (current == null) {
182             out.println("Person with id " + personId + " does not exist.");
183             return;
184         }
185 
186         out.print("Person name (NULL for null, enter for no change, current - " + current.getName() + "): ");
187         current.setName(convertString(scanner.nextLine(), current.getName()));
188 
189         out.print("Person surname (NULL for null, enter for no change, current - " + current.getSurname() + "): ");
190         current.setSurname(convertString(scanner.nextLine(), current.getSurname()));
191 
192         out.print("Person address id (NULL for null, enter for no change, current - " +
193                 (current.getAddress() == null ? "NULL" : current.getAddress().getId()) + "): ");
194         readAndSetAddress(scanner, current);
195     }

197     private void printAddress(StringBuilder sb, Address a) {
198         sb.append("id = ").append(a.getId()).append(", streetName = ").append(a.getStreetName())
199                 .append(", houseNumber = ").append(a.getHouseNumber())
200                 .append(", flatNumber = ").append(a.getFlatNumber())
201                 .append(", persons = (");
202 
203         Iterator<Person> iter = a.getPersons().iterator();
204         while (iter.hasNext()) {
205             Person p = iter.next();
206             sb.append("<").append(p.getId()).append("> ").append(p.getName()).append(" ").append(p.getSurname());
207             if (iter.hasNext()) {
208                 sb.append(", ");
209             }
210         }
211 
212         sb.append(")");
213     }

294     private void start() {
295         Scanner scanner = new Scanner(System.in);
296         PrintStream out = System.out;
297 
298         while (true) {
299             out.println("-----------------------------------------------");
300             out.println("1 - list persons             5 - list addresses");
301             out.println("2 - list person history      6 - list addresses history");
302             out.println("3 - new person               7 - new address");
303             out.println("4 - modify person            8 - modify address");
304             out.println("9 - get person at revision  10 - get address at revision");
305             out.println("                             0 - end");
306 
307             try {
308                 int choice = scanner.nextInt();
309 
310                 scanner.nextLine();
311 
312                 entityManager.getTransaction().begin();
313 
314                 StringBuilder sb;
315                 int personId;
316                 int addressId;
317                 int revision;
318                 switch (choice) {
319                     case 1:
320                         sb = new StringBuilder();
321                         printPersons(sb);
322                         out.println(sb.toString());
323                         break;
324                     case 2:
325                         out.print("Person id: ");
326                         personId = scanner.nextInt(); scanner.nextLine();
327                         sb = new StringBuilder();
328                         printPersonHistory(sb, personId);
329                         out.println(sb.toString());
330                         break;
331                     case 3:
332                         Person p = readNewPerson(out, scanner);
333                         entityManager.persist(p);
334                         break;
335                     case 4:
336                         out.print("Person id: ");
337                         personId = scanner.nextInt(); scanner.nextLine();
338                         readModifyPerson(out, scanner, personId);
339                         break;
340                     case 5:
341                         sb = new StringBuilder();
342                         printAddresses(sb);
343                         out.println(sb.toString());
344                         break;
345                     case 6:
346                         out.print("Address id: ");
347                         addressId = scanner.nextInt(); scanner.nextLine();
348                         sb = new StringBuilder();
349                         printAddressHistory(sb, addressId);
350                         out.println(sb.toString());
351                         break;
352                     case 7:
353                         Address a = readNewAddress(out, scanner);
354                         entityManager.persist(a);
355                         break;
356                     case 8:
357                         out.print("Address id: ");
358                         addressId = scanner.nextInt(); scanner.nextLine();
359                         readModifyAddress(out, scanner, addressId);
360                         break;
361                     case 9:
362                         out.print("Person id: ");
363                         personId = scanner.nextInt(); scanner.nextLine();
364                         out.print("Revision number: ");
365                         revision = scanner.nextInt(); scanner.nextLine();
366                         if (revision <= 0) {
367                             System.out.println("Revision must be greater then 0!");
368                             continue;
369                         }
370                         sb = new StringBuilder();
371                         printPersonAtRevision(sb, personId, revision);
372                         out.println(sb.toString());
373                         break;
374                     case 10:
375                         out.print("Address id: ");
376                         addressId = scanner.nextInt(); scanner.nextLine();
377                         out.print("Revision number: ");
378                         revision = scanner.nextInt(); scanner.nextLine();
379                         if (revision <= 0) {
380                             System.out.println("Revision must be greater then 0!");
381                             continue;
382                         }
383                         sb = new StringBuilder();
384                         printAddressAtRevision(sb, addressId, revision);
385                         out.println(sb.toString());
386                         break;
387 
388                     case 0:
389                         return;
390                 }
391             } catch (InputMismatchException e) {
392                 // continuing
393             } finally {
394                 entityManager.getTransaction().commit();
395             }
396         }
397     }

399     private boolean hasData() {
400         return (((Long) entityManager.createQuery("select count(a) from Address a").getSingleResult()) +
401                 ((Long) entityManager.createQuery("select count(p) from Person p").getSingleResult())) > 0;
402     }

453     public static void main(String[] args) {
454         String userDbFile = System.getProperty("java.io.tmpdir") + File.separator + "_versions_demo.db";
455 
456         Map<String, String> configurationOverrides = new HashMap<String, String>();
457         EntityManagerFactory emf = Persistence.createEntityManagerFactory("ConsolePU", configurationOverrides);
458         EntityManager entityManager = emf.createEntityManager();
459 
460         TestConsole console = new TestConsole(entityManager);
461 
462         System.out.println("");
463         System.out.println("Welcome to EntityVersions demo!");
464         System.out.println("HSQLDB database file location: " + userDbFile);
465 
466         console.populateTestData();
467         console.start();
468 
469         entityManager.close();
470         emf.close();
471     }
