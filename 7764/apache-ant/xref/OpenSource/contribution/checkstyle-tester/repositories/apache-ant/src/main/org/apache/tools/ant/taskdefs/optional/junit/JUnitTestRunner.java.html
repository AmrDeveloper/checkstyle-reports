53  /**
54   * Simple Testrunner for JUnit that runs all tests of a testsuite.
55   *
56   * <p>This TestRunner expects a name of a TestCase class as its
57   * argument. If this class provides a static suite() method it will be
58   * called and the resulting Test will be run. So, the signature should be
59   * <pre><code>
60   *     public static junit.framework.Test suite()
61   * </code></pre>
62   *
63   * <p> If no such method exists, all public methods starting with
64   * "test" and taking no argument will be run.
65   *
66   * <p> Summary output is generated at the end.
67   *
68   * @since Ant 1.2
69   */
70

320     private void setupIOStreams(ByteArrayOutputStream o,
321                                 ByteArrayOutputStream e) {
322         systemOut = new PrintStream(o);
323         systemError = new PrintStream(e);
324 
325         if (forked) {
326             if (!outputToFormatters) {
327                 if (!showOutput) {
328                     savedOut = System.out;
329                     savedErr = System.err;
330                     System.setOut(createEmptyStream());
331                     System.setErr(createEmptyStream());
332                 }
333             } else {
334                 savedOut = System.out;
335                 savedErr = System.err;
336                 if (!showOutput) {
337                     System.setOut(systemOut);
338                     System.setErr(systemError);
339                 } else {
340                     System.setOut(createTeePrint(savedOut, systemOut));
341                     System.setErr(createTeePrint(savedErr, systemError));
342                 }
343                 perm = null;
344             }
345         } else {
346             if (perm != null) {
347                 perm.setSecurityManager();
348             }
349         }
350     }

352     /**
353      * Run the test.
354      */
355     public void run() {
356         res = new IgnoredTestResult();
357         res.addListener(wrapListener(this));
358         final int size = formatters.size();
359         for (int i = 0; i < size; i++) {
360             res.addListener(wrapListener((TestListener) formatters.elementAt(i)));
361         }
362 
363         ByteArrayOutputStream errStrm = new ByteArrayOutputStream();
364         ByteArrayOutputStream outStrm = new ByteArrayOutputStream();
365 
366         setupIOStreams(outStrm, errStrm);
367 
368         Test suite = null;
369         Throwable exception = null;
370         boolean startTestSuiteSuccess = false;
371 
372         try {
373 
374             try {
375                 Class testClass = null;
376                 if (loader == null) {
377                     testClass = Class.forName(junitTest.getName());
378                 } else {
379                     testClass = Class.forName(junitTest.getName(), true,
380                                               loader);
381                 }
382 
383                 final boolean testMethodsSpecified = (methods != null);
384 
385                 // check for a static suite method first, even when using
386                 // JUnit 4
387                 Method suiteMethod = null;
388                 if (!testMethodsSpecified) {
389                 try {
390                     // check if there is a suite method
391                     suiteMethod = testClass.getMethod("suite", new Class[0]);
392                 } catch (NoSuchMethodException e) {
393                     // no appropriate suite method found. We don't report any
394                     // error here since it might be perfectly normal.
395                 }
396                 }
397 
398                 if (suiteMethod != null) {
399                     // if there is a suite method available, then try
400                     // to extract the suite from it. If there is an error
401                     // here it will be caught below and reported.
402                     suite = (Test) suiteMethod.invoke(null, new Object[0]);
403 
404                 } else {
405                     Class junit4TestAdapterClass = null;
406                     Class junit4TestAdapterCacheClass = null;
407                     boolean useSingleMethodAdapter = false;
408 
409                     if (junit.framework.TestCase.class.isAssignableFrom(testClass)) {
410                         // Do not use JUnit 4 API for running JUnit 3.x
411                         // tests - it is not able to run individual test
412                         // methods.
413                         //
414                         // Technical details:
415                         // org.junit.runner.Request.method(Class, String).getRunner()
416                         // would return a runner which always executes all
417                         // test methods. The reason is that the Runner would be
418                         // an instance of class
419                         // org.junit.internal.runners.OldTestClassRunner
420                         // that does not implement interface Filterable - so it
421                         // is unable to filter out test methods not matching
422                         // the requested name.
423                     } else {
424                     // Check for JDK 5 first. Will *not* help on JDK 1.4
425                     // if only junit-4.0.jar in CP because in that case
426                     // linkage of whole task will already have failed! But
427                     // will help if CP has junit-3.8.2.jar:junit-4.0.jar.
428 
429                     // In that case first C.fN will fail with CNFE and we
430                     // will avoid UnsupportedClassVersionError.
431 
432                     try {
433                         Class.forName("java.lang.annotation.Annotation");
434                         junit4TestAdapterCacheClass = Class.forName("org.apache.tools.ant.taskdefs.optional.junit.CustomJUnit4TestAdapterCache");
435                         if (loader == null) {
436                             junit4TestAdapterClass =
437                                 Class.forName(JUNIT_4_TEST_ADAPTER);
438                             if (testMethodsSpecified) {
439                                 /*
440                                  * We cannot try to load the JUnit4TestAdapter
441                                  * before trying to load JUnit4TestMethodAdapter
442                                  * because it might fail with
443                                  * NoClassDefFoundException, instead of plain
444                                  * ClassNotFoundException.
445                                  */
446                                 junit4TestAdapterClass = Class.forName(
447                                     "org.apache.tools.ant.taskdefs.optional.junit.JUnit4TestMethodAdapter");
448                                 useSingleMethodAdapter = true;
449                             }
450                         } else {
451                             junit4TestAdapterClass =
452                                 Class.forName(JUNIT_4_TEST_ADAPTER,
453                                               true, loader);
454                             if (testMethodsSpecified) {
455                                 junit4TestAdapterClass =
456                                     Class.forName(
457                                         "org.apache.tools.ant.taskdefs.optional.junit.JUnit4TestMethodAdapter",
458                                         true, loader);
459                                 useSingleMethodAdapter = true;
460                             }
461                         }
462                     } catch (ClassNotFoundException e) {
463                         // OK, fall back to JUnit 3.
464                     }
465                     }
466                     junit4 = junit4TestAdapterClass != null;
467 
468                     if (junitTest.isSkipNonTests()) {
469                        if (!containsTests( testClass, junit4)) {
470                            return;
471                        }
472                     }
473 
474 
475                     if (junit4) {
476                         // Let's use it!
477                         Class[] formalParams;
478                         Object[] actualParams;
479                         if (useSingleMethodAdapter) {
480                             formalParams = new Class[] {Class.class, String[].class};
481                             actualParams = new Object[] {testClass, methods};
482                         } else {
483                             formalParams = new Class[] {Class.class, Class.forName("junit.framework.JUnit4TestAdapterCache")};
484                             actualParams = new Object[] {testClass, junit4TestAdapterCacheClass.getMethod("getInstance").invoke(null)};
485                         }
486                         suite =
487                             (Test) junit4TestAdapterClass
488                             .getConstructor(formalParams).
489                             newInstance(actualParams);
490                     } else {
491                         // Use JUnit 3.
492 
493                         // try to extract a test suite automatically this
494                         // will generate warnings if the class is no
495                         // suitable Test
496                         if (!testMethodsSpecified) {
497                             suite = new TestSuite(testClass);
498                         } else if (methods.length == 1) {
499                             suite = TestSuite.createTest(testClass, methods[0]);
500                         } else {
501                             TestSuite testSuite = new TestSuite(testClass.getName());
502                             for (int i = 0; i < methods.length; i++) {
503                                 testSuite.addTest(
504                                     TestSuite.createTest(testClass, methods[i]));
505                             }
506                             suite = testSuite;
507                         }
508                     }
509 
510                 }
511 
512             } catch (Throwable e) {
513                 retCode = ERRORS;
514                 exception = e;
515             }
516 
517             long start = System.currentTimeMillis();
518 
519             fireStartTestSuite();
520             startTestSuiteSuccess = true;
521             if (exception != null) { // had an exception constructing suite
522                 final int formatterSize = formatters.size();
523                 for (int i = 0; i < formatterSize; i++) {
524                     ((TestListener) formatters.elementAt(i))
525                         .addError(null, exception);
526                 }
527                 junitTest.setCounts(1, 0, 1, 0);
528                 junitTest.setRunTime(0);
529             } else {
530                 try {
531                     logTestListenerEvent("tests to run: " + suite.countTestCases());
532                     suite.run(res);
533                 } finally {
534                     if (junit4 ||
535                         suite.getClass().getName().equals(JUNIT_4_TEST_ADAPTER)) {
536                         int[] cnts = findJUnit4FailureErrorCount(res);
537                         junitTest.setCounts(res.runCount() + res.ignoredCount(), cnts[0], cnts[1], res.ignoredCount() + res.skippedCount());
538                     } else {
539                         junitTest.setCounts(res.runCount() + res.ignoredCount(), res.failureCount(),
540                                 res.errorCount(), res.ignoredCount() + res.skippedCount());
541                     }
542                     junitTest.setRunTime(System.currentTimeMillis() - start);
543                 }
544             }
545         } finally {
546             if (perm != null) {
547                 perm.restoreSecurityManager();
548             }
549             if (savedOut != null) {
550                 System.setOut(savedOut);
551             }
552             if (savedErr != null) {
553                 System.setErr(savedErr);
554             }
555 
556             systemError.close();
557             systemError = null;
558             systemOut.close();
559             systemOut = null;
560             if (startTestSuiteSuccess) {
561                 String out, err;
562                 try {
563                     out = new String(outStrm.toByteArray());
564                 } catch (OutOfMemoryError ex) {
565                     out = "out of memory on output stream";
566                 }
567                 try {
568                     err = new String(errStrm.toByteArray());
569                 } catch (OutOfMemoryError ex) {
570                     err = "out of memory on error stream";
571                 }
572                 sendOutAndErr(out, err);
573             }
574         }
575         fireEndTestSuite();
576 
577         // junitTest has the correct counts for JUnit4, while res doesn't
578         if (retCode != SUCCESS || junitTest.errorCount() != 0) {
579             retCode = ERRORS;
580         } else if (junitTest.failureCount() != 0) {
581             retCode = FAILURES;
582         }
583     }

585     private static boolean containsTests(Class<?> testClass, boolean isJUnit4) {
586         Class testAnnotation = null;
587         Class suiteAnnotation = null;
588         Class runWithAnnotation = null;
589 
590         try {
591             testAnnotation = Class.forName("org.junit.Test");
592         } catch (ClassNotFoundException e) {
593             if (isJUnit4) {
594                 // odd - we think we're JUnit4 but don't support the test annotation. We therefore can't have any tests!
595                 return false;
596             }
597             // else... we're a JUnit3 test and don't need the annotation
598         }
599 
600         try {
601             suiteAnnotation = Class.forName("org.junit.Suite.SuiteClasses");
602         } catch(ClassNotFoundException ex) {
603             // ignore - we don't have this annotation so make sure we don't check for it
604         }
605         try {
606             runWithAnnotation = Class.forName("org.junit.runner.RunWith");
607         } catch(ClassNotFoundException ex) {
608             // also ignore as this annotation doesn't exist so tests can't use it
609         }
610 
611 
612         if (!isJUnit4 && !TestCase.class.isAssignableFrom(testClass)) {
613             //a test we think is JUnit3 but does not extend TestCase. Can't really be a test.
614             return false;
615         }
616 
617         // check if we have any inner classes that contain suitable test methods
618         for (Class<?> innerClass : testClass.getDeclaredClasses()) {
619             if (containsTests(innerClass, isJUnit4) || containsTests(innerClass, !isJUnit4)) {
620                 return true;
621             }
622         }
623 
624         if (Modifier.isAbstract(testClass.getModifiers()) || Modifier.isInterface(testClass.getModifiers())) {
625             // can't instantiate class and no inner classes are tests either
626             return false;
627         }
628 
629         if (isJUnit4) {
630              if (suiteAnnotation != null && testClass.getAnnotation(suiteAnnotation) != null) {
631                 // class is marked as a suite. Let JUnit try and work its magic on it.
632                 return true;
633              }
634             if (runWithAnnotation != null && testClass.getAnnotation(runWithAnnotation) != null) {
635                 /* Class is marked with @RunWith. If this class is badly written (no test methods, multiple
636                  * constructors, private constructor etc) then the class is automatically run and fails in the
637                  * IDEs I've tried... so I'm happy handing the class to JUnit to try and run, and let JUnit
638                  * report a failure if a bad test case is provided. Trying to do anything else is likely to
639                  * result in us filtering out cases that could be valid for future versions of JUnit so would
640                  * just increase future maintenance work.
641                  */
642                 return true;
643             }
644         }
645 
646         for (Method m : testClass.getMethods()) {
647             if (isJUnit4) {
648                 // check if suspected JUnit4 classes have methods with @Test annotation
649                 if (m.getAnnotation(testAnnotation) != null) {
650                     return true;
651                 }
652             } else {
653                 // check if JUnit3 class have public or protected no-args methods starting with names starting with test
654                 if (m.getName().startsWith("test") && m.getParameterTypes().length == 0
655                         && (Modifier.isProtected(m.getModifiers()) || Modifier.isPublic(m.getModifiers()))) {
656                     return true;
657                 }
658             }
659             // check if JUnit3 or JUnit4 test have a public or protected, static,
660             // no-args 'suite' method
661             if (m.getName().equals("suite") && m.getParameterTypes().length == 0
662                     && (Modifier.isProtected(m.getModifiers()) || Modifier.isPublic(m.getModifiers()))
663                     && Modifier.isStatic(m.getModifiers())) {
664                 return true;
665             }
666         }
667 
668         // no test methods found
669         return false;
670     }

682      * Interface TestListener.
683      *
684      * <p>A new Test is started.
685      * @param t the test.

858     /**
859      * Entry point for standalone (forked) mode.
860      *
861      * Parameters: testcaseclassname plus parameters in the format
862      * key=value, none of which is required.
863      *
864      * <table cols="4" border="1">
865      * <tr><th>key</th><th>description</th><th>default value</th></tr>
866      *
867      * <tr><td>haltOnError</td><td>halt test on
868      * errors?</td><td>false</td></tr>
869      *
870      * <tr><td>haltOnFailure</td><td>halt test on
871      * failures?</td><td>false</td></tr>
872      *
873      * <tr><td>formatter</td><td>A JUnitResultFormatter given as
874      * classname,filename. If filename is omitted, System.out is
875      * assumed.</td><td>none</td></tr>
876      *
877      * <tr><td>showoutput</td><td>send output to System.err/.out as
878      * well as to the formatters?</td><td>false</td></tr>
879      *
880      * <tr><td>logtestlistenerevents</td><td>log TestListener events to
881      * System.out.</td><td>false</td></tr>
882      *
883      * <tr><td>methods</td><td>Comma-separated list of names of individual
884      * test methods to execute.
885      * </td><td>null</td></tr>
886      *
887      * </table>
888      * @param args the command line arguments.
889      * @throws IOException on error.
890      */
891     public static void main(String[] args) throws IOException {
892         String[] methods = null;
893         boolean haltError = false;
894         boolean haltFail = false;
895         boolean stackfilter = true;
896         Properties props = new Properties();
897         boolean showOut = false;
898         boolean outputToFormat = true;
899         boolean logFailedTests = true;
900         boolean logTestListenerEvents = false;
901         boolean skipNonTests = false;
902         int antThreadID = 0; /* Ant id of thread running this unit test, 0 in single-threaded mode */
903 
904         if (args.length == 0) {
905             System.err.println("required argument TestClassName missing");
906             System.exit(ERRORS);
907         }
908 
909         if (args[0].startsWith(Constants.TESTSFILE)) {
910             multipleTests = true;
911             args[0] = args[0].substring(Constants.TESTSFILE.length());
912         }
913 
914         for (int i = 1; i < args.length; i++) {
915             if (args[i].startsWith(Constants.METHOD_NAMES)) {
916                 try {
917                     String methodsList = args[i].substring(Constants.METHOD_NAMES.length());
918                     methods = JUnitTest.parseTestMethodNamesList(methodsList);
919                 } catch (IllegalArgumentException ex) {
920                     System.err.println("Invalid specification of test method names: " + args[i]);
921                     System.exit(ERRORS);
922                 }
923             } else if (args[i].startsWith(Constants.HALT_ON_ERROR)) {
924                 haltError = Project.toBoolean(args[i].substring(Constants.HALT_ON_ERROR.length()));
925             } else if (args[i].startsWith(Constants.HALT_ON_FAILURE)) {
926                 haltFail = Project.toBoolean(args[i].substring(Constants.HALT_ON_FAILURE.length()));
927             } else if (args[i].startsWith(Constants.FILTERTRACE)) {
928                 stackfilter = Project.toBoolean(args[i].substring(Constants.FILTERTRACE.length()));
929             } else if (args[i].startsWith(Constants.CRASHFILE)) {
930                 crashFile = args[i].substring(Constants.CRASHFILE.length());
931                 registerTestCase(Constants.BEFORE_FIRST_TEST);
932             } else if (args[i].startsWith(Constants.FORMATTER)) {
933                 try {
934                     createAndStoreFormatter(args[i].substring(Constants.FORMATTER.length()));
935                 } catch (BuildException be) {
936                     System.err.println(be.getMessage());
937                     System.exit(ERRORS);
938                 }
939             } else if (args[i].startsWith(Constants.PROPSFILE)) {
940                 FileInputStream in = new FileInputStream(args[i]
941                                                          .substring(Constants.PROPSFILE.length()));
942                 props.load(in);
943                 in.close();
944             } else if (args[i].startsWith(Constants.SHOWOUTPUT)) {
945                 showOut = Project.toBoolean(args[i].substring(Constants.SHOWOUTPUT.length()));
946             } else if (args[i].startsWith(Constants.LOGTESTLISTENEREVENTS)) {
947                 logTestListenerEvents = Project.toBoolean(
948                     args[i].substring(Constants.LOGTESTLISTENEREVENTS.length()));
949             } else if (args[i].startsWith(Constants.OUTPUT_TO_FORMATTERS)) {
950                 outputToFormat = Project.toBoolean(
951                     args[i].substring(Constants.OUTPUT_TO_FORMATTERS.length()));
952             } else if (args[i].startsWith(Constants.LOG_FAILED_TESTS)) {
953                 logFailedTests = Project.toBoolean(
954                     args[i].substring(Constants.LOG_FAILED_TESTS.length()));
955             } else if (args[i].startsWith(Constants.SKIP_NON_TESTS)) {
956                 skipNonTests = Project.toBoolean(
957                     args[i].substring(Constants.SKIP_NON_TESTS.length()));
958             } else if (args[i].startsWith(Constants.THREADID)) {
959                 antThreadID = Integer.parseInt( args[i].substring(Constants.THREADID.length()) );
960             }
961         }
962 
963         // Add/overlay system properties on the properties from the Ant project
964         Hashtable p = System.getProperties();
965         for (Enumeration e = p.keys(); e.hasMoreElements();) {
966             Object key = e.nextElement();
967             props.put(key, p.get(key));
968         }
969 
970         int returnCode = SUCCESS;
971         if (multipleTests) {
972             try {
973                 java.io.BufferedReader reader =
974                     new java.io.BufferedReader(new java.io.FileReader(args[0]));
975                 String testCaseName;
976                 String[] testMethodNames;
977                 int code = 0;
978                 boolean errorOccurred = false;
979                 boolean failureOccurred = false;
980                 String line = null;
981                 while ((line = reader.readLine()) != null) {
982                     StringTokenizer st = new StringTokenizer(line, ",");
983                     String testListSpec = st.nextToken();
984                     int colonIndex = testListSpec.indexOf(':');
985                     if (colonIndex == -1) {
986                         testCaseName = testListSpec;
987                         testMethodNames = null;
988                     } else {
989                         testCaseName = testListSpec.substring(0, colonIndex);
990                         testMethodNames = JUnitTest.parseTestMethodNamesList(
991                                                     testListSpec
992                                                     .substring(colonIndex + 1)
993                                                     .replace('+', ','));
994                     }
995                     JUnitTest t = new JUnitTest(testCaseName);
996                     t.setTodir(new File(st.nextToken()));
997                     t.setOutfile(st.nextToken());
998                     t.setProperties(props);
999                     t.setSkipNonTests(skipNonTests);
1000                     t.setThread(antThreadID);
1001                     code = launch(t, testMethodNames, haltError, stackfilter, haltFail,
1002                                   showOut, outputToFormat,
1003                                   logTestListenerEvents);
1004                     errorOccurred = (code == ERRORS);
1005                     failureOccurred = (code != SUCCESS);
1006                     if (errorOccurred || failureOccurred) {
1007                         if ((errorOccurred && haltError)
1008                             || (failureOccurred && haltFail)) {
1009                             registerNonCrash();
1010                             System.exit(code);
1011                         } else {
1012                             if (code > returnCode) {
1013                                 returnCode = code;
1014                             }
1015                             if (logFailedTests) {
1016                                 System.out.println("TEST " + t.getName()
1017                                                    + " FAILED");
1018                             }
1019                         }
1020                     }
1021                 }
1022             } catch (IOException e) {
1023                 e.printStackTrace();
1024             }
1025         } else {
1026             JUnitTest t = new JUnitTest(args[0]);
1027             t.setThread(antThreadID);
1028             t.setProperties(props);
1029             t.setSkipNonTests(skipNonTests);
1030             returnCode = launch(
1031                 t, methods, haltError, stackfilter, haltFail,
1032                 showOut, outputToFormat, logTestListenerEvents);
1033         }
1034 
1035         registerNonCrash();
1036         System.exit(returnCode);
1037     }

1086     /**
1087      * Line format is: formatter=<classname>(,<pathname>)?
1088      */
1089     private static void createAndStoreFormatter(String line)
1090         throws BuildException {
1091         FormatterElement fe = new FormatterElement();
1092         int pos = line.indexOf(',');
1093         if (pos == -1) {
1094             fe.setClassname(line);
1095             fe.setUseFile(false);
1096         } else {
1097             fe.setClassname(line.substring(0, pos));
1098             fe.setUseFile(true);
1099             if (!multipleTests) {
1100                 fe.setOutfile(new File(line.substring(pos + 1)));
1101             } else {
1102                 int fName = line.indexOf(IGNORED_FILE_NAME);
1103                 if (fName > -1) {
1104                     fe.setExtension(line
1105                                     .substring(fName
1106                                                + IGNORED_FILE_NAME.length()));
1107                 }
1108             }
1109         }
1110         fromCmdLine.addElement(fe);
1111     }

1124     /**
1125      * Filters stack frames from internal JUnit and Ant classes
1126      * @param stack the stack trace to filter.
1127      * @return the filtered stack.
1128      */
1129     public static String filterStack(String stack) {
1130         if (!filtertrace) {
1131             return stack;
1132         }
1133         StringWriter sw = new StringWriter();
1134         BufferedWriter pw = new BufferedWriter(sw);
1135         StringReader sr = new StringReader(stack);
1136         BufferedReader br = new BufferedReader(sr);
1137 
1138         String line;
1139         try {
1140             boolean firstLine = true;
1141             while ((line = br.readLine()) != null) {
1142                 if (firstLine || !filterLine(line)) {
1143                     pw.write(line);
1144                     pw.newLine();
1145                 }
1146                 firstLine = false;
1147             }
1148         } catch (Exception e) {
1149             return stack; // return the stack unfiltered
1150         } finally {
1151             FileUtils.close(pw);
1152         }
1153         return sw.toString();
1154     }

1217     /**
1218      * Modifies a TestListener when running JUnit 4: treats AssertionFailedError
1219      * as a failure not an error.
1220      *
1221      * @since Ant 1.7
1222      */
1223     private TestListenerWrapper wrapListener(final TestListener testListener) {
1224         return new TestListenerWrapper(testListener) {
1225             public void addError(Test test, Throwable t) {
1226                 if (junit4 && t instanceof AssertionFailedError) {
1227                     // JUnit 4 does not distinguish between errors and failures
1228                     // even in the JUnit 3 adapter.
1229                     // So we need to help it a bit to retain compatibility for JUnit 3 tests.
1230                     testListener.addFailure(test, (AssertionFailedError) t);
1231                 } else if (junit4 && t instanceof  AssertionError) {
1232                     // Not strictly necessary but probably desirable.
1233                     // JUnit 4-specific test GUIs will show just "failures".
1234                     // But Ant's output shows "failures" vs. "errors".
1235                     // We would prefer to show "failure" for things that logically are.
1236                     String msg = t.getMessage();
1237                     AssertionFailedError failure = msg != null
1238                         ? new AssertionFailedError(msg) : new AssertionFailedError();
1239                     failure.setStackTrace(t.getStackTrace());
1240                     testListener.addFailure(test, failure);
1241                 } else {
1242                     testListener.addError(test, t);
1243                 }
1244             }
1245             public void addFailure(Test test, AssertionFailedError t) {
1246                 testListener.addFailure(test, t);
1247             }
1248             public void addFailure(Test test, Throwable t) { // pre-3.4
1249                 if (t instanceof AssertionFailedError) {
1250                     testListener.addFailure(test, (AssertionFailedError) t);
1251                 } else {
1252                     testListener.addError(test, t);
1253                 }
1254             }
1255             public void endTest(Test test) {
1256                 testListener.endTest(test);
1257             }
1258             public void startTest(Test test) {
1259                 testListener.startTest(test);
1260             }
1261         };
1262     }

1264     /**
1265      * Use instead of TestResult.get{Failure,Error}Count on JUnit 4,
1266      * since the adapter claims that all failures are errors.
1267      * @since Ant 1.7
1268      */
1269     private int[] findJUnit4FailureErrorCount(TestResult result) {
1270         int failures = 0;
1271         int errors = 0;
1272         Enumeration e = result.failures();
1273         while (e.hasMoreElements()) {
1274             e.nextElement();
1275             failures++;
1276         }
1277         e = result.errors();
1278         while (e.hasMoreElements()) {
1279             Throwable t = ((TestFailure) e.nextElement()).thrownException();
1280             if (t instanceof AssertionFailedError
1281                 || t instanceof AssertionError) {
1282                 failures++;
1283             } else {
1284                 errors++;
1285             }
1286         }
1287         return new int[] {failures, errors};
1288     }
