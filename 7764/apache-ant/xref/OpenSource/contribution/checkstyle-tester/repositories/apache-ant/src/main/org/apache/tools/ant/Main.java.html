201     /**
202      * Start Ant
203      * @param args command line args
204      * @param additionalUserProperties properties to set beyond those that
205      *        may be specified on the args list
206      * @param coreLoader - not used
207      *
208      * @since Ant 1.6
209      */
210     public void startAnt(String[] args, Properties additionalUserProperties,
211                          ClassLoader coreLoader) {
212 
213         try {
214             processArgs(args);
215         } catch (Throwable exc) {
216             handleLogfile();
217             printMessage(exc);
218             exit(1);
219             return;
220         }
221 
222         if (additionalUserProperties != null) {
223             for (Enumeration<?> e = additionalUserProperties.keys();
224                     e.hasMoreElements();) {
225                 String key = (String) e.nextElement();
226                 String property = additionalUserProperties.getProperty(key);
227                 definedProps.put(key, property);
228             }
229         }
230 
231         // expect the worst
232         int exitCode = 1;
233         try {
234             try {
235                 runBuild(coreLoader);
236                 exitCode = 0;
237             } catch (ExitStatusException ese) {
238                 exitCode = ese.getStatus();
239                 if (exitCode != 0) {
240                     throw ese;
241                 }
242             }
243         } catch (BuildException be) {
244             if (err != System.err) {
245                 printMessage(be);
246             }
247         } catch (Throwable exc) {
248             exc.printStackTrace();
249             printMessage(exc);
250         } finally {
251             handleLogfile();
252         }
253         exit(exitCode);
254     }

311     /**
312      * Process command line arguments.
313      * When ant is started from Launcher, launcher-only arguments do not get
314      * passed through to this routine.
315      *
316      * @param args the command line arguments.
317      *
318      * @since Ant 1.6
319      */
320     private void processArgs(String[] args) {
321         String searchForThis = null;
322         boolean searchForFile = false;
323         PrintStream logTo = null;
324 
325         // cycle through given args
326 
327         boolean justPrintUsage = false;
328         boolean justPrintVersion = false;
329         boolean justPrintDiagnostics = false;
330 
331         ArgumentProcessorRegistry processorRegistry = ArgumentProcessorRegistry.getInstance();
332         
333         for (int i = 0; i < args.length; i++) {
334             String arg = args[i];
335 
336             if (arg.equals("-help") || arg.equals("-h")) {
337                 justPrintUsage = true;
338             } else if (arg.equals("-version")) {
339                 justPrintVersion = true;
340             } else if (arg.equals("-diagnostics")) {
341                 justPrintDiagnostics = true;
342             } else if (arg.equals("-quiet") || arg.equals("-q")) {
343                 msgOutputLevel = Project.MSG_WARN;
344             } else if (arg.equals("-verbose") || arg.equals("-v")) {
345                 msgOutputLevel = Project.MSG_VERBOSE;
346             } else if (arg.equals("-debug") || arg.equals("-d")) {
347                 msgOutputLevel = Project.MSG_DEBUG;
348             } else if (arg.equals("-silent") || arg.equals("-S")) {
349                 silent = true;
350             } else if (arg.equals("-noinput")) {
351                 allowInput = false;
352             } else if (arg.equals("-logfile") || arg.equals("-l")) {
353                 try {
354                     File logFile = new File(args[i + 1]);
355                     i++;
356                     logTo = new PrintStream(new FileOutputStream(logFile));
357                     isLogFileUsed = true;
358                 } catch (IOException ioe) {
359                     String msg = "Cannot write on the specified log file. "
360                         + "Make sure the path exists and you have write "
361                         + "permissions.";
362                     throw new BuildException(msg);
363                 } catch (ArrayIndexOutOfBoundsException aioobe) {
364                     String msg = "You must specify a log file when "
365                         + "using the -log argument";
366                     throw new BuildException(msg);
367                 }
368             } else if (arg.equals("-buildfile") || arg.equals("-file")
369                        || arg.equals("-f")) {
370                 i = handleArgBuildFile(args, i);
371             } else if (arg.equals("-listener")) {
372                 i = handleArgListener(args, i);
373             } else if (arg.startsWith("-D")) {
374                 i = handleArgDefine(args, i);
375             } else if (arg.equals("-logger")) {
376                 i = handleArgLogger(args, i);
377             } else if (arg.equals("-inputhandler")) {
378                 i = handleArgInputHandler(args, i);
379             } else if (arg.equals("-emacs") || arg.equals("-e")) {
380                 emacsMode = true;
381             } else if (arg.equals("-projecthelp") || arg.equals("-p")) {
382                 // set the flag to display the targets and quit
383                 projectHelp = true;
384             } else if (arg.equals("-find") || arg.equals("-s")) {
385                 searchForFile = true;
386                 // eat up next arg if present, default to build.xml
387                 if (i < args.length - 1) {
388                     searchForThis = args[++i];
389                 }
390             } else if (arg.startsWith("-propertyfile")) {
391                 i = handleArgPropertyFile(args, i);
392             } else if (arg.equals("-k") || arg.equals("-keep-going")) {
393                 keepGoingMode = true;
394             } else if (arg.equals("-nice")) {
395                 i = handleArgNice(args, i);
396             } else if (LAUNCH_COMMANDS.contains(arg)) {
397                 //catch script/ant mismatch with a meaningful message
398                 //we could ignore it, but there are likely to be other
399                 //version problems, so we stamp down on the configuration now
400                 String msg = "Ant's Main method is being handed "
401                         + "an option " + arg + " that is only for the launcher class."
402                         + "\nThis can be caused by a version mismatch between "
403                         + "the ant script/.bat file and Ant itself.";
404                 throw new BuildException(msg);
405             } else if (arg.equals("-autoproxy")) {
406                 proxy = true;
407             } else if (arg.startsWith("-")) {
408                 boolean processed = false;
409                 for (ArgumentProcessor processor : processorRegistry.getProcessors()) {
410                     int newI = processor.readArguments(args, i);
411                     if (newI != -1) {
412                         List<String> extraArgs = extraArguments.get(processor.getClass());
413                         if (extraArgs == null) {
414                             extraArgs = new ArrayList<String>();
415                             extraArguments.put(processor.getClass(), extraArgs);
416                         }
417                         for (; i < newI && i < args.length; i++) {
418                             extraArgs.add(args[i]);
419                         }
420                         processed = true;
421                         break;
422                     }
423                 }
424                 if (!processed) {
425                     // we don't have any more args to recognize!
426                     String msg = "Unknown argument: " + arg;
427                     System.err.println(msg);
428                     printUsage();
429                     throw new BuildException("");
430                 }
431             } else {
432                 // if it's no other arg, it may be the target
433                 targets.addElement(arg);
434             }
435         }
436 
437         if (msgOutputLevel >= Project.MSG_VERBOSE || justPrintVersion) {
438             printVersion(msgOutputLevel);
439         }
440 
441         if (justPrintUsage || justPrintVersion || justPrintDiagnostics) {
442             if (justPrintUsage) {
443                 printUsage();
444             }
445             if (justPrintDiagnostics) {
446                 Diagnostics.doReport(System.out, msgOutputLevel);
447             }
448             return;
449         }
450 
451         // if buildFile was not specified on the command line,
452         if (buildFile == null) {
453             // but -find then search for it
454             if (searchForFile) {
455                 if (searchForThis != null) {
456                     buildFile = findBuildFile(System.getProperty("user.dir"), searchForThis);
457                     if (buildFile == null) {
458                         throw new BuildException("Could not locate a build file!");
459                     }
460                 } else {
461                     // no search file specified: so search an existing default file
462                     Iterator<ProjectHelper> it = ProjectHelperRepository.getInstance().getHelpers();
463                     do {
464                         ProjectHelper helper = it.next();
465                         searchForThis = helper.getDefaultBuildFile();
466                         if (msgOutputLevel >= Project.MSG_VERBOSE) {
467                             System.out.println("Searching the default build file: " + searchForThis);
468                         }
469                         buildFile = findBuildFile(System.getProperty("user.dir"), searchForThis);
470                     } while (buildFile == null && it.hasNext());
471                     if (buildFile == null) {
472                         throw new BuildException("Could not locate a build file!");
473                     }
474                 }
475             } else {
476                 // no build file specified: so search an existing default file
477                 Iterator<ProjectHelper> it = ProjectHelperRepository.getInstance().getHelpers();
478                 do {
479                     ProjectHelper helper = it.next();
480                     buildFile = new File(helper.getDefaultBuildFile());
481                     if (msgOutputLevel >= Project.MSG_VERBOSE) {
482                         System.out.println("Trying the default build file: " + buildFile);
483                     }
484                 } while (!buildFile.exists() && it.hasNext());
485             }
486         }
487 
488         // make sure buildfile exists
489         if (!buildFile.exists()) {
490             System.out.println("Buildfile: " + buildFile + " does not exist!");
491             throw new BuildException("Build failed");
492         }
493 
494         if (buildFile.isDirectory()) {
495             File whatYouMeant = new File(buildFile, "build.xml");
496             if (whatYouMeant.isFile()) {
497                 buildFile = whatYouMeant;
498             } else {
499                 System.out.println("What? Buildfile: " + buildFile + " is a dir!");
500                 throw new BuildException("Build failed");
501             }
502         }
503 
504         // Normalize buildFile for re-import detection
505         buildFile =
506             FileUtils.getFileUtils().normalize(buildFile.getAbsolutePath());
507 
508         // Load the property files specified by -propertyfile
509         loadPropertyFiles();
510 
511         if (msgOutputLevel >= Project.MSG_INFO) {
512             System.out.println("Buildfile: " + buildFile);
513         }
514 
515         if (logTo != null) {
516             out = logTo;
517             err = logTo;
518             System.setOut(out);
519             System.setErr(err);
520         }
521         readyToRun = true;
522     }

553     /** Handler -D argument */
554     private int handleArgDefine(String[] args, int argPos) {
555         /* Interestingly enough, we get to here when a user
556          * uses -Dname=value. However, in some cases, the OS
557          * goes ahead and parses this out to args
558          *   {"-Dname", "value"}
559          * so instead of parsing on "=", we just make the "-D"
560          * characters go away and skip one argument forward.
561          *
562          * I don't know how to predict when the JDK is going
563          * to help or not, so we simply look for the equals sign.
564          */
565         String arg = args[argPos];
566         String name = arg.substring(2, arg.length());
567         String value = null;
568         int posEq = name.indexOf("=");
569         if (posEq > 0) {
570             value = name.substring(posEq + 1);
571             name = name.substring(0, posEq);
572         } else if (argPos < args.length - 1) {
573             value = args[++argPos];
574         } else {
575             throw new BuildException("Missing value for property "
576                                      + name);
577         }
578         definedProps.put(name, value);
579         return argPos;
580     }

625     /** Handle the -nice argument. */
626     private int handleArgNice(String[] args, int pos) {
627         try {
628             threadPriority = Integer.decode(args[++pos]);
629         } catch (ArrayIndexOutOfBoundsException aioobe) {
630             throw new BuildException(
631                 "You must supply a niceness value (1-10)"
632                 + " after the -nice option");
633         } catch (NumberFormatException e) {
634             throw new BuildException("Unrecognized niceness value: "
635                                      + args[pos]);
636         }
637 
638         if (threadPriority.intValue() < Thread.MIN_PRIORITY
639             || threadPriority.intValue() > Thread.MAX_PRIORITY) {
640             throw new BuildException(
641                 "Niceness value is out of the range 1-10");
642         }
643         return pos;
644     }

650     /** Load the property files specified by -propertyfile */
651     private void loadPropertyFiles() {
652         for (String filename : propertyFiles) {
653             Properties props = new Properties();
654             FileInputStream fis = null;
655             try {
656                 fis = new FileInputStream(filename);
657                 props.load(fis);
658             } catch (IOException e) {
659                 System.out.println("Could not load property file "
660                                    + filename + ": " + e.getMessage());
661             } finally {
662                 FileUtils.close(fis);
663             }
664 
665             // ensure that -D properties take precedence
666             Enumeration<?> propertyNames = props.propertyNames();
667             while (propertyNames.hasMoreElements()) {
668                 String name = (String) propertyNames.nextElement();
669                 if (definedProps.getProperty(name) == null) {
670                     definedProps.put(name, props.getProperty(name));
671                 }
672             }
673         }
674     }

695     /**
696      * Search parent directories for the build file.
697      * <p>
698      * Takes the given target as a suffix to append to each
699      * parent directory in search of a build file.  Once the
700      * root of the file-system has been reached <code>null</code>
701      * is returned.
702      *
703      * @param start  Leaf directory of search.
704      *               Must not be <code>null</code>.
705      * @param suffix  Suffix filename to look for in parents.
706      *                Must not be <code>null</code>.
707      *
708      * @return A handle to the build file if one is found, <code>null</code> if not
709      */
710     private File findBuildFile(String start, String suffix) {
711         if (msgOutputLevel >= Project.MSG_INFO) {
712             System.out.println("Searching for " + suffix + " ...");
713         }
714 
715         File parent = new File(new File(start).getAbsolutePath());
716         File file = new File(parent, suffix);
717 
718         // check if the target file exists in the current directory
719         while (!file.exists()) {
720             // change to parent directory
721             parent = getParentFile(parent);
722 
723             // if parent is null, then we are at the root of the fs,
724             // complain that we can't find the build file.
725             if (parent == null) {
726                 return null;
727             }
728 
729             // refresh our file handle

736     /**
737      * Executes the build. If the constructor for this instance failed
738      * (e.g. returned after issuing a warning), this method returns
739      * immediately.
740      *
741      * @param coreLoader The classloader to use to find core classes.
742      *                   May be <code>null</code>, in which case the
743      *                   system classloader is used.
744      *
745      * @exception BuildException if the build fails
746      */
747     private void runBuild(ClassLoader coreLoader) throws BuildException {
748 
749         if (!readyToRun) {
750             return;
751         }
752 
753         ArgumentProcessorRegistry processorRegistry = ArgumentProcessorRegistry.getInstance();
754 
755         for (ArgumentProcessor processor : processorRegistry.getProcessors()) {
756             List<String> extraArgs = extraArguments.get(processor.getClass());
757             if (extraArgs != null) {
758                 if (processor.handleArg(extraArgs)) {
759                     return;
760                 }
761             }
762         }
763 
764         final Project project = new Project();
765         project.setCoreLoader(coreLoader);
766 
767         Throwable error = null;
768 
769         try {
770             addBuildListeners(project);
771             addInputHandler(project);
772 
773             PrintStream savedErr = System.err;
774             PrintStream savedOut = System.out;
775             InputStream savedIn = System.in;
776 
777             // use a system manager that prevents from System.exit()
778             SecurityManager oldsm = null;
779             oldsm = System.getSecurityManager();
780 
781                 //SecurityManager can not be installed here for backwards
782                 //compatibility reasons (PD). Needs to be loaded prior to
783                 //ant class if we are going to implement it.
784                 //System.setSecurityManager(new NoExitSecurityManager());
785             try {
786                 if (allowInput) {
787                     project.setDefaultInputStream(System.in);
788                 }
789                 System.setIn(new DemuxInputStream(project));
790                 System.setOut(new PrintStream(new DemuxOutputStream(project, false)));
791                 System.setErr(new PrintStream(new DemuxOutputStream(project, true)));
792 
793 
794                 if (!projectHelp) {
795                     project.fireBuildStarted();
796                 }
797 
798                 // set the thread priorities
799                 if (threadPriority != null) {
800                     try {
801                         project.log("Setting Ant's thread priority to "
802                                 + threadPriority, Project.MSG_VERBOSE);
803                         Thread.currentThread().setPriority(threadPriority.intValue());
804                     } catch (SecurityException swallowed) {
805                         //we cannot set the priority here.
806                         project.log("A security manager refused to set the -nice value");
807                     }
808                 }
809 
810                 setProperties(project);
811 
812                 project.setKeepGoingMode(keepGoingMode);
813                 if (proxy) {
814                     //proxy setup if enabled
815                     ProxySetup proxySetup = new ProxySetup(project);
816                     proxySetup.enableProxies();
817                 }
818 
819                 for (ArgumentProcessor processor : processorRegistry.getProcessors()) {
820                     List<String> extraArgs = extraArguments.get(processor.getClass());
821                     if (extraArgs != null) {
822                         processor.prepareConfigure(project, extraArgs);
823                     }
824                 }
825 
826                 ProjectHelper.configureProject(project, buildFile);
827 
828                 for (ArgumentProcessor processor : processorRegistry.getProcessors()) {
829                     List<String> extraArgs = extraArguments.get(processor.getClass());
830                     if (extraArgs != null) {
831                         if (processor.handleArg(project, extraArgs)) {
832                             return;
833                         }
834                     }
835                 }
836 
837                 if (projectHelp) {
838                     printDescription(project);
839                     printTargets(project, msgOutputLevel > Project.MSG_INFO,
840                             msgOutputLevel > Project.MSG_VERBOSE);
841                     return;
842                 }
843 
844                 // make sure that we have a target to execute
845                 if (targets.size() == 0) {
846                     if (project.getDefaultTarget() != null) {
847                         targets.addElement(project.getDefaultTarget());
848                     }
849                 }
850 
851                 project.executeTargets(targets);
852             } finally {
853                 // put back the original security manager
854                 //The following will never eval to true. (PD)
855                 if (oldsm != null) {
856                     System.setSecurityManager(oldsm);
857                 }
858 
859                 System.setOut(savedOut);
860                 System.setErr(savedErr);
861                 System.setIn(savedIn);
862             }
863         } catch (RuntimeException exc) {
864             error = exc;
865             throw exc;
866         } catch (Error e) {
867             error = e;
868             throw e;
869         } finally {
870             if (!projectHelp) {
871                 try {
872                     project.fireBuildFinished(error);
873                 } catch (Throwable t) {
874                     // yes, I know it is bad style to catch Throwable,
875                     // but if we don't, we lose valuable information
876                     System.err.println("Caught an exception while logging the"
877                                        + " end of the build.  Exception was:");
878                     t.printStackTrace();
879                     if (error != null) {
880                         System.err.println("There has been an error prior to"
881                                            + " that:");
882                         error.printStackTrace();
883                     }
884                     throw new BuildException(t);
885                 }
886             } else if (error != null) {
887                 project.log(error.toString(), Project.MSG_ERR);
888             }
889         }
890     }

1135     /**
1136      * Targets in imported files with a project name
1137      * and not overloaded by the main build file will
1138      * be in the target map twice. This method
1139      * removes the duplicate target.
1140      * @param targets the targets to filter.
1141      * @return the filtered targets.
1142      */
1143     private static Map<String, Target> removeDuplicateTargets(Map<String, Target> targets) {
1144         Map<Location, Target> locationMap = new HashMap<Location, Target>();
1145         for (Entry<String, Target> entry : targets.entrySet()) {
1146             String name = entry.getKey();
1147             Target target = entry.getValue();
1148             Target otherTarget = locationMap.get(target.getLocation());
1149             // Place this entry in the location map if
1150             //  a) location is not in the map
1151             //  b) location is in map, but its name is longer
1152             //     (an imported target will have a name. prefix)
1153             if (otherTarget == null
1154                 || otherTarget.getName().length() > name.length()) {
1155                 locationMap.put(
1156                     target.getLocation(), target); // Smallest name wins
1157             }
1158         }
1159         Map<String, Target> ret = new HashMap<String, Target>();
1160         for (Target target : locationMap.values()) {
1161             ret.put(target.getName(), target);
1162         }
1163         return ret;
1164     }

1166     /**
1167      * Prints a list of all targets in the specified project to
1168      * <code>System.out</code>, optionally including subtargets.
1169      *
1170      * @param project The project to display a description of.
1171      *                Must not be <code>null</code>.
1172      * @param printSubTargets Whether or not subtarget names should also be
1173      *                        printed.
1174      */
1175     private static void printTargets(Project project, boolean printSubTargets,
1176             boolean printDependencies) {
1177         // find the target with the longest name
1178         int maxLength = 0;
1179         Map<String, Target> ptargets = removeDuplicateTargets(project.getTargets());
1180         // split the targets in top-level and sub-targets depending
1181         // on the presence of a description
1182         Vector<String> topNames = new Vector<String>();
1183         Vector<String> topDescriptions = new Vector<String>();
1184         Vector<Enumeration<String>> topDependencies = new Vector<Enumeration<String>>();
1185         Vector<String> subNames = new Vector<String>();
1186         Vector<Enumeration<String>> subDependencies = new Vector<Enumeration<String>>();
1187 
1188         for (Target currentTarget : ptargets.values()) {
1189             String targetName = currentTarget.getName();
1190             if (targetName.equals("")) {
1191                 continue;
1192             }
1193             String targetDescription = currentTarget.getDescription();
1194             // maintain a sorted list of targets
1195             if (targetDescription == null) {
1196                 int pos = findTargetPosition(subNames, targetName);
1197                 subNames.insertElementAt(targetName, pos);
1198                 if (printDependencies) {
1199                     subDependencies.insertElementAt(currentTarget.getDependencies(), pos);
1200                 }
1201             } else {
1202                 int pos = findTargetPosition(topNames, targetName);
1203                 topNames.insertElementAt(targetName, pos);
1204                 topDescriptions.insertElementAt(targetDescription, pos);
1205                 if (targetName.length() > maxLength) {
1206                     maxLength = targetName.length();
1207                 }
1208                 if (printDependencies) {
1209                     topDependencies.insertElementAt(currentTarget.getDependencies(), pos);
1210                 }
1211             }
1212         }
1213 
1214         printTargets(project, topNames, topDescriptions, topDependencies,
1215                 "Main targets:", maxLength);
1216         //if there were no main targets, we list all subtargets
1217         //as it means nothing has a description
1218         if (topNames.size() == 0) {
1219             printSubTargets = true;
1220         }
1221         if (printSubTargets) {
1222             printTargets(project, subNames, null, subDependencies, "Other targets:", 0);
1223         }
1224 
1225         String defaultTarget = project.getDefaultTarget();
1226         if (defaultTarget != null && !"".equals(defaultTarget)) {
1227             // shouldn't need to check but...
1228             project.log("Default target: " + defaultTarget);
1229         }
1230     }

1232     /**
1233      * Searches for the correct place to insert a name into a list so as
1234      * to keep the list sorted alphabetically.
1235      *
1236      * @param names The current list of names. Must not be <code>null</code>.
1237      * @param name  The name to find a place for.
1238      *              Must not be <code>null</code>.
1239      *
1240      * @return the correct place in the list for the given name
1241      */
1242     private static int findTargetPosition(Vector<String> names, String name) {
1243         final int size = names.size();
1244         int res = size;
1245         for (int i = 0; i < size && res == size; i++) {
1246             if (name.compareTo(names.elementAt(i)) < 0) {
1247                 res = i;
1248             }
1249         }
1250         return res;
1251     }

1253     /**
1254      * Writes a formatted list of target names to <code>System.out</code>
1255      * with an optional description.
1256      *
1257      *
1258      * @param project the project instance.
1259      * @param names The names to be printed.
1260      *              Must not be <code>null</code>.
1261      * @param descriptions The associated target descriptions.
1262      *                     May be <code>null</code>, in which case
1263      *                     no descriptions are displayed.
1264      *                     If non-<code>null</code>, this should have
1265      *                     as many elements as <code>names</code>.
1266      * @param topDependencies The list of dependencies for each target.
1267      *                        The dependencies are listed as a non null
1268      *                        enumeration of String.
1269      * @param heading The heading to display.
1270      *                Should not be <code>null</code>.
1271      * @param maxlen The maximum length of the names of the targets.
1272      *               If descriptions are given, they are padded to this
1273      *               position so they line up (so long as the names really
1274      *               <i>are</i> shorter than this).
1275      */
1276     private static void printTargets(Project project, Vector<String> names,
1277                                      Vector<String> descriptions, Vector<Enumeration<String>> dependencies,
1278                                      String heading,
1279                                      int maxlen) {
1280         // now, start printing the targets and their descriptions
1281         String lSep = System.getProperty("line.separator");
1282         // got a bit annoyed that I couldn't find a pad function
1283         String spaces = "    ";
1284         while (spaces.length() <= maxlen) {
1285             spaces += spaces;
1286         }
1287         StringBuilder msg = new StringBuilder();
1288         msg.append(heading + lSep + lSep);
1289         final int size = names.size();
1290         for (int i = 0; i < size; i++) {
1291             msg.append(" ");
1292             msg.append(names.elementAt(i));
1293             if (descriptions != null) {
1294                 msg.append(
1295                     spaces.substring(0, maxlen - names.elementAt(i).length() + 2));
1296                 msg.append(descriptions.elementAt(i));
1297             }
1298             msg.append(lSep);
1299             if (!dependencies.isEmpty()) {
1300                 Enumeration<String> deps = dependencies.elementAt(i);
1301                 if (deps.hasMoreElements()) {
1302                     msg.append("   depends on: ");
1303                     while (deps.hasMoreElements()) {
1304                         msg.append(deps.nextElement());
1305                         if (deps.hasMoreElements()) {
1306                             msg.append(", ");
1307                         }
1308                     }
1309                     msg.append(lSep);
1310                 }
1311             }
1312         }
1313         project.log(msg.toString(), Project.MSG_WARN);
1314     }
