20  package edu.umd.cs.findbugs.xml;
21  
22  import java.util.Iterator;
23  import java.util.List;
24  
25  import org.dom4j.Attribute;
26  import org.dom4j.Document;
27  import org.dom4j.Element;
28  import org.dom4j.Node;
29  import org.dom4j.io.SAXReader;
30  
31  
32  /**
33   * Find nodes in a dom4j tree that match a particular XPath expression. The
34   * main() driver prints out information about matching nodes in an XML document.
35   *
36   * <p>
37   * For example, to find the list of non-disabled detectors in a FindBugs plugin
38   * descriptor, you can use the expression <blockquote>
39   * <code>/FindbugsPlugin/Detector[boolean(@disabled)=false()]/@class</code>
40   * </blockquote>
41   *
42   * @author David Hovemeyer
43   */
44  public abstract class XPathFind {
45      private final Document document;
46  
47      public XPathFind(Document document) {
48          this.document = document;
49      }
50  
51      public void find(String xpath) {
52          List<Node> nodes =  XMLUtil.selectNodes(document, xpath);
53          for (Node node : nodes) {
54              match(node);
55          }
56      }
57  
58      protected abstract void match(Node node);
59  
60      public static void main(String[] argv) throws Exception {
61          if (argv.length != 2) {
62              System.err.println("Usage: " + XPathFind.class.getName() + ": <filename> <xpath expression>");
63              System.exit(1);
64          }
65  
66          String fileName = argv[0];
67          String xpath = argv[1];
68  
69          SAXReader reader = new SAXReader();
70          Document document = reader.read(fileName);
71  
72          XPathFind finder = new XPathFind(document) {
73              @Override
74              protected void match(Node node) {
75                  // System.out.println(node.toString());
76                  if (node instanceof Element) {
77                      Element element = (Element) node;
78                      System.out.println("Element: " + element.getQualifiedName());
79                      System.out.println("\tText: " + element.getText());
80                      System.out.println("\tAttributes:");
81                      for (Iterator<?> i = element.attributeIterator(); i.hasNext();) {
82                          Attribute attribute = (Attribute) i.next();
83                          System.out.println("\t\t" + attribute.getName() + "=" + attribute.getValue());
84                      }
85                  } else if (node instanceof Attribute) {
86                      Attribute attribute = (Attribute) node;
87                      System.out.println("Attribute: " + attribute.getName() + "=" + attribute.getValue());
88                  }
89              }
90          };
91  
92          finder.find(xpath);
93      }
94  }
