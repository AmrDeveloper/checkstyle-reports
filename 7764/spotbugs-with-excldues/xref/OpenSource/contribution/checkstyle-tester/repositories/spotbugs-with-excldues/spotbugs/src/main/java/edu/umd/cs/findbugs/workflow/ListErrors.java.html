20  package edu.umd.cs.findbugs.workflow;
21  
22  import edu.umd.cs.findbugs.AnalysisError;
23  import edu.umd.cs.findbugs.FindBugs;
24  import edu.umd.cs.findbugs.SortedBugCollection;
25  
26  /**
27   * List the analysis errors in a bug collection
28   *
29   * @author Bill Pugh
30   */
31  public class ListErrors {
32      public static void main(String[] args) throws Exception {
33          if (args.length != 1) {
34              System.out.println("Usage: " + ListErrors.class.getName() + " <bug collection>");
35              System.exit(1);
36          }
37          FindBugs.setNoAnalysis();
38          SortedBugCollection bugCollection = new SortedBugCollection();
39          bugCollection.readXML(args[0]);
40          for (AnalysisError e : bugCollection.getErrors()) {
41              String msg = e.getExceptionMessage();
42              if (msg != null && msg.trim().length() > 0) {
43                  System.out.println(e.getMessage() + " : " + msg);
44              } else {
45                  System.out.println(e.getMessage());
46              }
47  
48          }
49  
50      }
51  }
