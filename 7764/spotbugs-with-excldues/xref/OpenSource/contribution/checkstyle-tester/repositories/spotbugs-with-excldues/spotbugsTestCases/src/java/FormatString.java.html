1   import java.util.Date;
2   
3   import edu.umd.cs.findbugs.annotations.ExpectWarning;
4   import edu.umd.cs.findbugs.annotations.NoWarning;
5   
6   public class FormatString {
7       @ExpectWarning("FS")
8       public static void main(String args[]) {
9           falsePositive();
10  
11          java.util.Formatter f = new java.util.Formatter();
12          System.out.println(f.format("%d %s %f %d %f", 8, "boy", 7.7, 9, 9.9, 10)); // warning
13                                                                                     // !
14          System.out.println(f.format("%d %s %f %d %f %d", 8, "boy", 7.7, 9, 9.9, 10, 11)); // warning
15                                                                                            // !
16          System.out.println(String.format("%d %n", 1, 2)); // warning!
17  
18          System.out.printf("%s\n", new int[] { 1, 2 });
19          System.out.printf("%b\n", "false");
20          System.out.printf("%d\n", "foo");
21          System.out.printf("%d% \n", 42);
22          System.out.printf("%d\n", new int[] { 1, 2 });
23          System.out.printf("%s\n", (Object) new String[] { "a", "b" });
24  
25          varargsMethod((Object[]) args);
26      }
27  
28      @NoWarning("FS")
29      public static void falsePositive() {
30          System.out.printf("last updated %1$tY-%1$tm-%1$te %1$tH:%1$tM:%1$tS.%1$tL, threshold %2$d min.%n", new Date(), 20);
31          java.util.Formatter f = new java.util.Formatter();
32          System.out.println(String.format("%d %s", 8, "boy")); // no warning
33          System.out.println(f.format("%d %s %f %d %f %d", 8, "boy", 7.7, 9, 9.9, 10)); // no
34                                                                                        // warning
35          System.out.printf("%d", 7); // no warning
36          System.out.format("%d", 8); // no warning
37          System.out.println(f.format("%d %%%%%s %% %f %d %%%f %%%d", 8, "boy", 7.7, 9, 9.9, 10)); // no
38                                                                                                   // warning
39          System.out.println(String.format("%d %n", 1)); // no warning (%n does
40                                                         // not consume an
41                                                         // argument)
42  
43      }
44  
45      private static void varargsMethod(Object... args) {
46          if (args[0] instanceof String) {
47              System.out.println((String) args[0]);
48          }
49      }
50  }
