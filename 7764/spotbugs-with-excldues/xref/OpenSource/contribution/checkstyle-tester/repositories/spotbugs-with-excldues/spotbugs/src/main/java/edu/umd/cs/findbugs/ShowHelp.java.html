19  package edu.umd.cs.findbugs;

21  import java.util.TreeSet;
22  
23  /**
24   * Show command line help.
25   *
26   * @author David Hovemeyer
27   */
28  public class ShowHelp {
29  
30      public static void main(String[] args) {
31          System.out.println("SpotBugs version " + Version.VERSION_STRING + ", " + Version.WEBSITE);
32  
33          DetectorFactoryCollection.instance();
34          System.out.println("Command line options");
35  
36          TreeSet<FindBugsMain> cmds = new TreeSet<>();
37          for(Plugin p : Plugin.getAllPlugins()) {
38              for(FindBugsMain m : p.getAllFindBugsMain()) {
39                  cmds.add(m);
40              }
41          }
42          for(FindBugsMain m : cmds) {
43              System.out.printf("fb %-12s %-12s %s%n", m.cmd, m.kind, m.description);
44          }
45  
46          //        System.out.println();
47          //        System.out.println("GUI Options:");
48          //        FindBugsCommandLine guiCmd = new FindBugsCommandLine(true) {
49          //        };
50          //        guiCmd.printUsage(System.out);
51          //        System.out.println();
52          //        System.out.println("TextUI Options:");
53          //        FindBugs.showCommandLineOptions();
54          System.out.println();
55          showGeneralOptions();
56  
57      }
58  
59      public static void showSynopsis() {
60          System.out.println("Usage: findbugs [general options] [gui options]");
61      }
62  
63      public static void showGeneralOptions() {
64  
65          System.out.println("General options:");
66          System.out.println("  -jvmArgs args    Pass args to JVM");
67          System.out.println("  -maxHeap size    Maximum Java heap size in megabytes (default=768)");
68          System.out.println("  -javahome <dir>  Specify location of JRE");
69  
70      }
71  }
