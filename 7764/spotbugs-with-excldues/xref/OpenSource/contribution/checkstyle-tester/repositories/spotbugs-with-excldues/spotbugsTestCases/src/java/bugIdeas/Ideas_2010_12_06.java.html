18  public class Ideas_2010_12_06 extends HttpServlet {
19  
20      @Override
21  	@ExpectWarning("PT_RELATIVE_PATH_TRAVERSAL")
22      public void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
23          response.setContentType("text/plain");
24          PrintWriter out = response.getWriter();
25          String path = request.getParameter("path");
26          BufferedReader r = new BufferedReader(new FileReader("data/" + path));
27          while (true) {
28              String txt = r.readLine();
29              if (txt == null)
30                  break;
31              out.println(txt);
32          }
33          out.close();
34          r.close();
35      }
36  
37      @Override
38  	@ExpectWarning("PT_ABSOLUTE_PATH_TRAVERSAL")
39      public void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
40          response.setContentType("text/plain");
41          PrintWriter out = response.getWriter();
42          String path = request.getParameter("path");
43          BufferedReader r = new BufferedReader(new FileReader(path));
44          while (true) {
45              String txt = r.readLine();
46              if (txt == null)
47                  break;
48              out.println(txt);
49          }
50          out.close();
51          r.close();
52      }
