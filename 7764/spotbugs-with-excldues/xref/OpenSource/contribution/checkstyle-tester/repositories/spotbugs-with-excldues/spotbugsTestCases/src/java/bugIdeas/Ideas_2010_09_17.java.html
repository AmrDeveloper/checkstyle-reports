8       private void resize(int size) {
9           synchronized (dataBuffer) {
10              System.out.println("Entering synchronized block (resizing to: " + size + ")");
11              bufferSize = size;
12              if (dataBuffer.length < size) {
13                  dataBuffer = new int[size]; // danger modifying mutex!!!!
14              }
15              try {
16                  Thread.sleep(3000);
17              } catch (InterruptedException e) {
18                  System.out.println("Sleep was interrupted");
19              }
20              System.out.println("Exiting synchronized block");
21          }
22      }
