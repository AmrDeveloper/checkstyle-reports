1   import java.io.BufferedInputStream;
2   import java.io.ByteArrayInputStream;
3   
4   public class SkipBufferedInputStream {
5       public static void main(String[] args) throws Exception {
6           byte[] buffer = new byte[100];
7           ByteArrayInputStream bais = new ByteArrayInputStream(buffer);
8           BufferedInputStream bis = new BufferedInputStream(bais, 50);
9   
10          byte[] smallBuf = new byte[10];
11  
12          bis.read(smallBuf);
13          long available = bis.available();
14          int request = 50;
15          long s = bis.skip(request);
16          if (s < available && s < request) {
17              System.out.println("Skipped fewer bytes than requested and fewer bytes than available");
18              System.out.println("Available: " + available);
19              System.out.println("Requested: " + request);
20              System.out.println("Skipped: " + s);
21          }
22      }
23  }
