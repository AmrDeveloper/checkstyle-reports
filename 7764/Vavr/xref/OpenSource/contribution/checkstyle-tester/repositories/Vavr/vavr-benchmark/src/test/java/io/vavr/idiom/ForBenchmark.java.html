9   import io.vavr.JmhRunner;
10  import io.vavr.collection.Array;
11  import org.junit.Test;
12  import org.openjdk.jmh.annotations.Benchmark;
13  import org.openjdk.jmh.annotations.Scope;
14  import org.openjdk.jmh.annotations.Setup;
15  import org.openjdk.jmh.annotations.State;
16  
17  import java.util.ArrayList;
18  import java.util.Arrays;
19  import java.util.List;
20  import java.util.function.BiFunction;
21  
22  import static java.util.stream.Collectors.toList;
23  import static io.vavr.API.*;
24  import static io.vavr.JmhRunner.getRandomValues;
25  
26  /**
27   * Benchmark for nested loops vs Vavr's For().yield comprehensions.
28   *
29   * @see For2
30   */
31  public class ForBenchmark {
32      static final Array<Class<?>> CLASSES = Array(
33              For.class
34      );
35  
36      @Test
37      public void testAsserts() {
38          JmhRunner.runDebugWithAsserts(CLASSES);
39      }
40  
41      public static void main(String... args) {
42          JmhRunner.runNormalNoAsserts(CLASSES);
43      }
44  
45      @State(Scope.Benchmark)
46      public static class Base {
47          int CONTAINER_SIZE = 1000;
48          int AGGREGATE;
49          List<Integer> ELEMENTS;
50  
51          final BiFunction<Integer, Integer, Integer> AGGREGATOR = (i, j) -> i ^ j;
52  
53          @Setup
54          public void setup() {
55              ELEMENTS = Arrays.asList(getRandomValues(CONTAINER_SIZE, 0, true));
56  
57              AGGREGATE = 0;
58              for (Integer i : ELEMENTS) {
59                  for (Integer j : ELEMENTS) {
60                      AGGREGATE += AGGREGATOR.apply(i, j);
61                  }
62              }
63  
64          }
65      }
66  
67      public static class For extends Base {
68          @Benchmark
69          public Object java_for() {
70              final List<Integer> result = new ArrayList<>(CONTAINER_SIZE * CONTAINER_SIZE);
71              for (Integer i : ELEMENTS) {
72                  for (Integer j : ELEMENTS) {
73                      result.add(AGGREGATOR.apply(i, j));
74                  }
75              }
76  
77              assert Array(result).sum().intValue() == AGGREGATE;
78              return result;
79          }
80  
81          @Benchmark
82          public Object vavr_for() {
83              final List<Integer> result = For(ELEMENTS, ELEMENTS).yield(AGGREGATOR).collect(toList());
84              assert Array(result).sum().intValue() == AGGREGATE;
85              return result;
86          }
87      }
88  
89  }
