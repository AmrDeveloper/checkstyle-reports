7   package io.vavr.idiom;
8   
9   import io.vavr.JmhRunner;
10  import io.vavr.collection.Array;
11  import org.junit.Test;
12  import org.openjdk.jmh.annotations.Benchmark;
13  import org.openjdk.jmh.annotations.Scope;
14  import org.openjdk.jmh.annotations.Setup;
15  import org.openjdk.jmh.annotations.State;
16  import org.openjdk.jmh.infra.Blackhole;
17  
18  import java.util.Random;
19  
20  import static io.vavr.API.*;
21  
22  /**
23   * Benchmark for nested loops vs Vavr's For().yield comprehensions.
24   *
25   * @see For2
26   */
27  public class PatternMatchingBenchmark {
28      
29      static final Array<Class<?>> CLASSES = Array(
30              MatchVsSwitchIntValues.class
31      );
32  
33      @Test
34      public void testAsserts() {
35          JmhRunner.runDebugWithAsserts(CLASSES);
36      }
37  
38      public static void main(String... args) {
39          JmhRunner.runNormalNoAsserts(CLASSES);
40      }
41  
42      @State(Scope.Benchmark)
43      public static class MatchVsSwitchIntValues {
44  
45          int[] VALUES;
46  
47          @Setup
48          public void setup() {
49              final int INSTANCES = 1000;
50              final int CASES = 5;
51              VALUES = Array.range(0, INSTANCES).map(i -> new Random(0).nextInt(CASES)).toJavaStream().mapToInt(i -> i).toArray();
52          }
53          
54          @Benchmark
55          public void java_switch(Blackhole bh) {
56              for (int i : VALUES) {
57                  final String result;
58                  switch (i) {
59                      case 0:
60                          result = "0";
61                          break;
62                      case 1:
63                          result = "1";
64                          break;
65                      case 2:
66                          result = "2";
67                          break;
68                      case 3:
69                          result = "3";
70                          break;
71                      default:
72                          result = "4";
73                          break;
74                  }
75  
76                  assert String.valueOf(i).equals(result);
77                  bh.consume(result);
78              }
79          }
80  
81          @Benchmark
82          public void vavr_match(Blackhole bh) {
83              for (int i : VALUES) {
84                  final String result = Match(i).of(
85                          Case($(0), "0"),
86                          Case($(1), "1"),
87                          Case($(2), "2"),
88                          Case($(3), "3"),
89                          Case($(), "4")
90                  );
91                  assert String.valueOf(i).equals(result);
92                  bh.consume(result);
93              }
94          }
95      }
96  }
