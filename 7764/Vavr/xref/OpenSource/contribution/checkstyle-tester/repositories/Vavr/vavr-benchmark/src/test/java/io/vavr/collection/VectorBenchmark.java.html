61      @State(Scope.Benchmark)
62      public static class Base {
63          @Param({ "10", "100", "1026" })
64          public int CONTAINER_SIZE;
65  
66          int EXPECTED_AGGREGATE;
67          Integer[] ELEMENTS;
68          int[] INT_ELEMENTS;
69          int[] RANDOMIZED_INDICES;
70  
71          /* Only use this for non-mutating operations */
72          java.util.ArrayList<Integer> javaMutable;
73  
74          fj.data.Seq<Integer> fjavaPersistent;
75          org.pcollections.PVector<Integer> pCollectionsPersistent;
76          org.eclipse.collections.api.list.ImmutableList<Integer> eCollectionsPersistent;

78          scala.collection.immutable.Vector<Integer> scalaPersistent;

81          io.vavr.collection.Vector<Byte> vavrPersistentByte;

83          @Setup
84          public void setup() {
85              final Random random = new Random(0);
86              ELEMENTS = getRandomValues(CONTAINER_SIZE, false, random);
87              INT_ELEMENTS = ArrayType.asPrimitives(int.class, Array.of(ELEMENTS));
88              RANDOMIZED_INDICES = shuffle(Array.range(0, CONTAINER_SIZE).toJavaStream().mapToInt(Integer::intValue).toArray(), random);
89  
90              EXPECTED_AGGREGATE = Array.of(ELEMENTS).reduce(JmhRunner::aggregate);
91  
92              javaMutable = create(java.util.ArrayList::new, asList(ELEMENTS), v -> areEqual(v, asList(ELEMENTS)));
93              fjavaPersistent = create(fj.data.Seq::fromJavaList, javaMutable, v -> areEqual(v, javaMutable));
94              pCollectionsPersistent = create(org.pcollections.TreePVector::from, javaMutable, v -> areEqual(v, javaMutable));
95              eCollectionsPersistent = create(org.eclipse.collections.impl.factory.Lists.immutable::ofAll, javaMutable, v -> areEqual(v, javaMutable));
96              clojurePersistent = create(clojure.lang.PersistentVector::create, javaMutable, v -> areEqual(v, javaMutable));
97              scalaPersistent = create(v -> (scala.collection.immutable.Vector<Integer>) scala.collection.immutable.Vector$.MODULE$.apply(asScalaBuffer(v)), javaMutable, v -> areEqual(asJavaCollection(v), javaMutable));
98              vavrPersistent = create(io.vavr.collection.Vector::ofAll, javaMutable, v -> areEqual(v, javaMutable));
99              vavrPersistentInt = create(v -> io.vavr.collection.Vector.ofAll(INT_ELEMENTS), javaMutable, v -> areEqual(v, javaMutable) && (v.trie.type.type() == int.class));
100 
101             final byte[] BYTE_ELEMENTS = new byte[CONTAINER_SIZE];
102             random.nextBytes(BYTE_ELEMENTS);
103             vavrPersistentByte = create(v -> io.vavr.collection.Vector.ofAll(BYTE_ELEMENTS), javaMutable, v -> areEqual(v, Array.ofAll(BYTE_ELEMENTS)) && (v.trie.type.type() == byte.class));
104         }
