19  /**
20   * Benchmark for Tuple[2,4,8] vs an array.
21   */
22  @State(Scope.Benchmark)
23  public class TupleBenchmark {
24      static final Array<Class<?>> CLASSES = Array(
25              Tuple2Benchmark.class,
26              Tuple4Benchmark.class,
27              Tuple8Benchmark.class
28      );
29  
30      @Test
31      public void testAsserts() {
32          JmhRunner.runDebugWithAsserts(CLASSES);
33      }
34  
35      public static void main(String... args) {
36          JmhRunner.runNormalNoAsserts(CLASSES);
37      }
38  
39      public static class Tuple2Benchmark {
40          @Benchmark
41          public Object java_tuple() { return new Integer[] { 0, 1 }; }
42  
43          @Benchmark
44          public Object vavr_tuple() { return Tuple.of(0, 1); }
45      }
46  
47      public static class Tuple4Benchmark {
48          @Benchmark
49          public Object java_tuple() { return new Integer[] { 0, 1, 2 }; }
50  
51          @Benchmark
52          public Object vavr_tuple() { return Tuple.of(0, 1, 2, 3); }
53      }
54  
55      public static class Tuple8Benchmark {
56          @Benchmark
57          public Object java_tuple() { return new Integer[] { 0, 1, 2, 3, 4, 5, 6, 7 }; }
58  
59          @Benchmark
60          public Object vavr_tuple() { return Tuple.of(0, 1, 2, 3, 4, 5, 6, 7); }
61      }
