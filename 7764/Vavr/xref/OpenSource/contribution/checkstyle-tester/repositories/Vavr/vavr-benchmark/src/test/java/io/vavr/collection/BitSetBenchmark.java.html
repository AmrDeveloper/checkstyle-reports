7   package io.vavr.collection;
8   
9   import io.vavr.JmhRunner;
10  import org.junit.Test;
11  import org.openjdk.jmh.annotations.*;
12  
13  import static io.vavr.JmhRunner.create;
14  import static io.vavr.JmhRunner.getRandomValues;
15  import static io.vavr.collection.Collections.areEqual;
16  import static scala.collection.JavaConverters.asJavaCollection;
17  import static scala.collection.JavaConverters.asScalaBuffer;
18  
19  public class BitSetBenchmark {
20      static final Array<Class<?>> CLASSES = Array.of(
21              AddAll.class,
22              Iterate.class
23      );
24  
25      @Test
26      public void testAsserts() {
27          JmhRunner.runDebugWithAsserts(CLASSES);
28      }
29  
30      public static void main(String... args) {
31          JmhRunner.runNormalNoAsserts(CLASSES);
32      }
33  
34      @State(Scope.Benchmark)
35      public static class Base {
36          @Param({ "10", "100", "1000" })
37          public int CONTAINER_SIZE;
38  
39          int EXPECTED_AGGREGATE;
40          int[] ELEMENTS;
41          TreeSet<Integer> DISTINCT;
42  
43          scala.collection.immutable.BitSet scalaPersistent;
44          io.vavr.collection.BitSet<Integer> vavrPersistent;
45  
46          @Setup
47          @SuppressWarnings("RedundantCast")
48          public void setup() {
49              final Integer[] values = getRandomValues(CONTAINER_SIZE, 0, true);
50              ELEMENTS = new int[CONTAINER_SIZE];
51              for (int i = 0; i < CONTAINER_SIZE; i++) {
52                  ELEMENTS[i] = values[i];
53              }
54  
55              DISTINCT = TreeSet.ofAll(ELEMENTS);
56              EXPECTED_AGGREGATE = DISTINCT.reduce(JmhRunner::aggregate);
57  
58              scalaPersistent = create(v -> (scala.collection.immutable.BitSet) scala.collection.immutable.BitSet$.MODULE$.apply(asScalaBuffer(v)), DISTINCT.toJavaList(), v -> areEqual(asJavaCollection(v), DISTINCT));
59              vavrPersistent = create(io.vavr.collection.BitSet::ofAll, ELEMENTS, ELEMENTS.length, v -> areEqual(v, DISTINCT));
60          }
61      }
62  
63      public static class AddAll extends Base {
64          @Benchmark
65          public Object scala_persistent() {
66              scala.collection.immutable.BitSet values = new scala.collection.immutable.BitSet.BitSet1(0L);
67              for (int element : ELEMENTS) {
68                  values = values.$plus(element);
69              }
70              assert ((scala.collection.immutable.SortedSet) values).equals(scalaPersistent);
71              return values;
72          }
73  
74          @Benchmark
75          public Object vavr_persistent() {
76              io.vavr.collection.Set<Integer> values = io.vavr.collection.BitSet.empty();
77              for (Integer element : ELEMENTS) {
78                  values = values.add(element);
79              }
80              assert values.equals(vavrPersistent);
81              return values;
82          }
83      }
84  
85      @SuppressWarnings("ForLoopReplaceableByForEach")
86      public static class Iterate extends Base {
87          @Benchmark
88          public int scala_persistent() {
89              int aggregate = 0;
90              for (final scala.collection.Iterator<Object> iterator = scalaPersistent.iterator(); iterator.hasNext(); ) {
91                  aggregate ^= (Integer) iterator.next();
92              }
93              assert aggregate == EXPECTED_AGGREGATE;
94              return aggregate;
95          }
96  
97          @Benchmark
98          public int vavr_persistent() {
99              int aggregate = 0;
100             for (final io.vavr.collection.Iterator<Integer> iterator = vavrPersistent.iterator(); iterator.hasNext(); ) {
101                 aggregate ^= iterator.next();
102             }
103             assert aggregate == EXPECTED_AGGREGATE;
104             return aggregate;
105         }
106     }
