219     private void previousButtonActionPerformed(java.awt.event.ActionEvent evt) {// GEN-FIRST:event_previousButtonActionPerformed
220         if (step > MIN_STEP) {
221             setStep(step - 1);
222         }
223         progressModel.removeAllElements();
224         foundModel.removeAllElements();
225     }// GEN-LAST:event_previousButtonActionPerformed

227     private void nextButtonActionPerformed(java.awt.event.ActionEvent evt) {// GEN-FIRST:event_nextButtonActionPerformed
228 
229         discover = new Thread() {
230             @Override
231             public void run() {
232                 IErrorLogger errorLogger = new IErrorLogger() {
233                     @Override
234                     public void reportMissingClass(ClassNotFoundException ex) {
235                         String className = ClassNotFoundExceptionParser.getMissingClassName(ex);
236                         if (className != null) {
237                             logError("Missing class: " + className);
238                         } else {
239                             logError("Missing class: " + ex);
240                         }
241                     }
242 
243                     @Override
244                     public void reportMissingClass(ClassDescriptor classDescriptor) {
245                         logError("Missing class: " + classDescriptor.toDottedClassName());
246                     }
247 
248                     @Override
249                     public void logError(String message) {
250                         System.err.println("Error: " + message);
251                     }
252 
253                     @Override
254                     public void logError(String message, Throwable e) {
255                         logError(message + ": " + e.getMessage());
256                     }
257 
258                     @Override
259                     public void reportSkippedAnalysis(MethodDescriptor method) {
260                         logError("Skipped analysis of method " + method.toString());
261                     }
262                 };
263 
264                 DiscoverSourceDirectories.Progress progress = new DiscoverSourceDirectories.Progress() {
265 
266                     @Override
267                     public void startRecursiveDirectorySearch() {
268                         progressModel.addElement("Scanning directories...");
269                     }
270 
271                     @Override
272                     public void doneRecursiveDirectorySearch() {
273                     }
274 
275                     @Override
276                     public void startScanningArchives(int numArchivesToScan) {
277                         progressModel.addElement("Scanning " + numArchivesToScan + " archives..");
278                     }
279 
280                     @Override
281                     public void doneScanningArchives() {
282                     }
283 
284                     @Override
285                     public void startScanningClasses(int numClassesToScan) {
286                         progressModel.addElement("Scanning " + numClassesToScan + " classes...");
287                     }
288 
289                     @Override
290                     public void finishClass() {
291                     }
292 
293                     @Override
294                     public void doneScanningClasses() {
295                     }
296 
297                     @Override
298                     public void finishArchive() {
299                     }
300 
301                     @Override
302                     public void startArchive(String name) {
303                     }
304                 };
305                 DiscoverSourceDirectories discoverSourceDirectories = new DiscoverSourceDirectories();
306                 discoverSourceDirectories.setProject(project);
307                 discoverSourceDirectories.setRootSourceDirectory(sourceRootBox.getText());
308                 discoverSourceDirectories.setErrorLogger(errorLogger);
309                 discoverSourceDirectories.setProgress(progress);
310 
311                 try {
312                     discoverSourceDirectories.execute();
313                 } catch (CheckedAnalysisException e1) {
314                     // TODO Auto-generated catch block
315                     e1.printStackTrace();
316                 } catch (IOException e1) {
317                     // TODO Auto-generated catch block
318                     e1.printStackTrace();
319                 } catch (InterruptedException e1) {
320                     // TODO Auto-generated catch block
321                     e1.printStackTrace();
322                 }
323 
324                 jList1.setModel(foundModel);
325 
326                 for (String srcDir : discoverSourceDirectories.getDiscoveredSourceDirectoryList()) {
327                     foundModel.addElement(srcDir);
328                 }
329 
330                 if (step < MAX_STEP) {
331                     setStep(step + 1);
332                 }
333             }
334         };
335         discover.start();
336     }// GEN-LAST:event_nextButtonActionPerformed

338     private void finshButtonActionPerformed(java.awt.event.ActionEvent evt) {// GEN-FIRST:event_finshButtonActionPerformed
339         if (parentGUI != null) {
340             parentGUI.setSourceDirecs(foundModel);
341         }
342         if (discover != null && discover.isAlive()) {
343             discover.stop();
344         }
345         dispose();
346     }// GEN-LAST:event_finshButtonActionPerformed

421     private void setStep(int step) {
422         if (step < MIN_STEP || step > MAX_STEP) {
423             throw new IllegalArgumentException("Invalid step " + step);
424         }
425         this.step = step;
426         previousButton.setEnabled(step != MIN_STEP);
427         nextButton.setEnabled(step != MAX_STEP && !"".equals(sourceRootBox.getText()));
428 
429         CardLayout cards = (CardLayout) contentPanel.getLayout();
430         cards.show(contentPanel, "card" + step);
431 
432     }
