42  public class SplashFrame extends JWindow {
43  
44      private static Thread animator;
45  
46      public SplashFrame() {
47          super(new Frame());
48  
49          Toolkit toolkit = Toolkit.getDefaultToolkit();
50          Image image = toolkit.getImage(MainFrame.class.getResource("SplashBug1.png"));
51          Image image2 = toolkit.getImage(MainFrame.class.getResource("SplashBug2B.png"));
52          Image imageReverse = toolkit.getImage(MainFrame.class.getResource("SplashBug1reverse.png"));
53          Image image2Reverse = toolkit.getImage(MainFrame.class.getResource("SplashBug2reverseB.png"));
54  
55          JLabel l = new JLabel(new ImageIcon(MainFrame.class.getResource("spotbugs.png")));
56          JPanel p = new JPanel();
57          Viewer viewer = new Viewer(image, image2, imageReverse, image2Reverse);
58          final JPanel bottom = viewer;
59          p.setBackground(Color.white);
60          bottom.setBackground(Color.white);
61  
62          p.add(l);
63          getContentPane().add(p, BorderLayout.CENTER);
64          getContentPane().add(bottom, BorderLayout.SOUTH);
65          pack();
66          Dimension labelSize = l.getPreferredSize();
67          p.setPreferredSize(new Dimension(labelSize.width + 50, labelSize.height + 20));
68          p.setBorder(BorderFactory.createLineBorder(Color.BLACK, 1));
69          bottom.setBorder(BorderFactory.createLineBorder(Color.BLACK, 1));
70          Dimension panelSize = p.getPreferredSize();
71          bottom.setPreferredSize(new Dimension(panelSize.width, image.getHeight(null) + 2));
72  
73          setLocationRelativeTo(null);
74  
75          // g.drawImage(new ImageIcon("bugSplash3.png"),0 ,0 ,null);
76  
77          pack();
78          viewer.animate();
79  
80      }
81  
82      public static void main(String[] args) {
83          (new SplashFrame()).setVisible(true);
84  
85      }
86  
87      @Override
88      public void setVisible(boolean b) {
89          super.setVisible(b);
90          if (!b) {
91              animator.interrupt();
92          }
93      }
94  
95      private static class Viewer extends JPanel {
96          private Image image;
97  
98          private Image image2;
99  
100         private Image imageR;
101 
102         private Image image2R;
103 
104         boolean swap = false;
105 
106         boolean reverse = true;
107 
108         int callCount = 0;
109 
110         int xpos = 0;
111 
112         int ypos = 0;
113 
114         //        int farRight;
115 
116         public Viewer(Image i1, Image i2, Image i1r, Image i2r) {
117             image = i1;
118             image2 = i2;
119             imageR = i1r;
120             image2R = i2r;
121             MediaTracker mediaTracker = new MediaTracker(this);
122             mediaTracker.addImage(image, 0);
123             mediaTracker.addImage(image2, 1);
124             mediaTracker.addImage(imageR, 2);
125             mediaTracker.addImage(image2R, 3);
126             try {
127                 mediaTracker.waitForID(0);
128                 mediaTracker.waitForID(1);
129                 mediaTracker.waitForID(2);
130                 mediaTracker.waitForID(3);
131             } catch (InterruptedException ie) {
132                 System.err.println(ie);
133                 System.exit(1);
134             }
135             animator = new Thread((Runnable) () -> {
136                 int deltaX = 1;
137 
138                 while (true) {
139                     if (Thread.currentThread().isInterrupted()) {
140                         return;
141                     }
142 
143                     callCount++;
144                     if (callCount == 10) {
145                         swap = !swap;
146                         callCount = 0;
147                     }
148 
149                     xpos += deltaX;
150                     try {
151                         Thread.sleep(20);
152                     } catch (InterruptedException e) {
153                         break;
154                     }
155 
156                     if (xpos > Viewer.this.getSize().width - image.getWidth(null)) {
157                         deltaX = -1;
158                         reverse = !reverse;
159                     }
160                     if (xpos < 0) {
161                         deltaX = 1;
162                         reverse = !reverse;
163                     }
164 
165                     Viewer.this.repaint();
166                 }
167             }, "SpotBugs Splash screen thread");
168             animator.setDaemon(true);
169             animator.setPriority(Thread.MIN_PRIORITY);
170         }
171 
172         public void animate() {
173             animator.start();
174         }
175 
176         private Image imageToDraw() {
177             if (swap) {
178                 if (!reverse) {
179                     return image;
180                 }
181                 return imageR;
182             } else {
183                 if (!reverse) {
184                     return image2;
185                 }
186                 return image2R;
187             }
188         }
189 
190         @Override
191         public void paint(Graphics graphics) {
192             super.paint(graphics);
193 
194             // graphics.clearRect(0,0,500,500);
195             graphics.drawImage(imageToDraw(), xpos, ypos, null);
196         }
197     }
