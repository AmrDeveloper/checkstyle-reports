1   import java.util.concurrent.locks.Lock;
2   import java.util.concurrent.locks.ReadWriteLock;
3   import java.util.concurrent.locks.ReentrantLock;
4   import java.util.concurrent.locks.ReentrantReadWriteLock;
5   
6   class TryLock {
7       public static void main(String args[]) throws Exception {
8           ReentrantLock lock1 = new ReentrantLock();
9           ReadWriteLock rwLock = new ReentrantReadWriteLock();
10          Lock lock2 = rwLock.readLock();
11          Lock lock3 = rwLock.writeLock();
12          rwLock.readLock();
13  
14          lock1.newCondition();
15          lock2.newCondition();
16          lock1.tryLock();
17          lock2.tryLock();
18          lock3.tryLock();
19  
20          synchronized (lock1) {
21              System.out.println("Howdy");
22          }
23      }
24  }
