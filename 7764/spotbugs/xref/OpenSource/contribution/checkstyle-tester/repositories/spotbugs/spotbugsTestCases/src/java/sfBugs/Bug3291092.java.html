10      static Pattern CAP_PATTERN = Pattern.compile("[a-z_0-9]+");
11  
12      @NoWarning("IL_INFINITE_LOOP")
13      public static String constize(String field) {
14          StringBuilder sb = new StringBuilder();
15          sb.append(field.charAt(0));
16          Matcher m = CAP_PATTERN.matcher(field.substring(1));
17          int start = 1;
18          while (m.find()) {
19              sb.append(field.substring(start, m.end() + 1).toUpperCase());
20              sb.append("_");
21              start = m.end();
22          }
23          sb.append(field.substring(start).toUpperCase());
24          return sb.toString();
25      }
26  
27      public static void main(String args[]) {
28          String s = "camelCase";
29          System.out.println(constize(s));
30      }
