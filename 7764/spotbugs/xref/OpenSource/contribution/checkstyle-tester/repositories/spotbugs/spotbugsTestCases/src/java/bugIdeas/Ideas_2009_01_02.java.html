5       /**
6        * The Zune 30 loop of death From http://pastie.org/349916
7        *
8        * This code goes into an infinite loop on Dec 31st of a leap year
9        *
10       * @param days
11       *            since Jan 0, 1980
12       * @return the year
13       */
14      static int zuneLoopOfDeath(int days) {
15          int year = 1980;
16  
17          while (days > 365) {
18              if (year % 4 == 0 && (year % 100 != 0 || year % 400 == 0)) {
19                  /** Leap year */
20                  if (days > 366) {
21                      days -= 366;
22                      year += 1;
23                  }
24              } else {
25                  /** Non-leap year */
26                  days -= 365;
27                  year += 1;
28              }
29          }
30          return year;
31      }

35              System.out.printf("%5d %5d\n", d, zuneLoopOfDeath(d));
